100 def add_attribute_to_plot(self, attribute, cardinality=1, accessor=None):
101 """
102
103 Add an attribute to the plotter
104
105 You have to ensure that each attribute is added at most once.
106
107
108 ## Type casts for non-floating point types
109
110 The plotter supports only floating point data. If you have attributes that are
111 not of floating point type, then you have to convert them into floating point
112 data before they are dumped into VTK. For this, the routine offers the accessor
113 attribute. As long as it equals None, we assume that there is a pointer p pointing
114 to the particle, and that we can use getXXXX() to access the attribute data. The
115 XXXX is the name of the attribute with the first letter written uppercase. This
116 is how Peano 4 writes down accessors (using DaStGen - but that's a different story
117 how and why we use certain code generator conventions).
118
119 You can however inject any other accessor expression (using the pointer p) to
120 read out particle data. If your particle data is for example a vector over
121 integers, you might want to hand in
122
123 tarch::la::convertScalar<double>(p->getXXXX())
124
125 which converts the data into a proper format.
126
127
128 ## Arguments
129
130 attribute: Attribute
131 The actual attribute that we have to plot.
132
133 cardinality: Integer
134 The cardinality of the attribute. This is 1 for a scalar value, 2 for a 2d vector,
135 and so forth. We support arbitrary cardinalities, but the cardinality hsa to match
136 attribute.
137
138 accessor: String or None
139 If this term is None, then we use getXXX() constructs as they are generated by
140 DaStGen by default. If you hand in something not equal to None, then we use this
141 particular string to access a field of a pointer p, i.e. we use
142
143 p->accessor
144
145 to get the data out of the particle.
146
147 """
148 assert cardinality > 0, "cardinality has to be greater than 0"
149
150 self._attributes_to_plot.append(attribute)
151 writer_name = f"_{attribute._name}Writer"
152 delim = "\n "
153 if accessor != None:
154 accessor_name = accessor
155 elif cardinality <= 3:
156 accessor_name = (
157 "p->get" + attribute._name[0].title() + attribute._name[1:] + "()"
158 )
159 else:
160 accessor_name = (
161 "p->get"
162 + attribute._name[0].title()
163 + attribute._name[1:]
164 + "().data()"
165 )
166
167 self.d[
168 "ATTRIBUTE_WRITER_NULL_INITIALISERS"
169 ] += f"{delim}{writer_name} = nullptr;"
170 self.d["ATTRIBUTE_WRITER_CLOSES"] += f"{delim}{writer_name}->close();"
171 self.d["ATTRIBUTE_WRITER_DELETES"] += f"{delim}delete {writer_name};"
172 self.d[
173 "ATTRIBUTE_WRITER_INITIALISERS"
174 ] += f'{delim}{writer_name} = _writer->createPointDataWriter( "{attribute._name}", {cardinality} );'
175 if cardinality <= 3:
176 self.d[
177 "ATTRIBUTE_WRITER_PLOT_CALLS"
178 ] += f"{delim} {writer_name}->plot(particleNumber,{accessor_name});"
179 else:
180 self.d[
181 "ATTRIBUTE_WRITER_PLOT_CALLS"
182 ] += f"{delim} {writer_name}->plot(particleNumber,{accessor_name},{cardinality});"
