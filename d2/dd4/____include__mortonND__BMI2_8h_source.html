76 template<std::size_t Dimensions, typename T>
77 class MortonNDBmi
78 {
79 template<std::size_t... Idx>
80 static decltype(auto) apply_variadic_func(T *t, std::index_sequence<Idx...>) {
81 return EncodeInternal(t[Idx]...);
82 }
83
84 public:
85 static constexpr auto FieldBits = std::size_t(std::numeric_limits<T>::digits) / Dimensions;
86
87 static_assert(Dimensions > 0, "'Dimensions' must be > 0.");
88 static_assert(std::is_same<T, uint32_t>::value | std::is_same<T, uint64_t>::value,
89 "'T' must be either uint32_t or uint64_t.");
90
109 template<typename...Args>
110 static inline T Encode(T field1, Args... fields)
111 {
112 static_assert(sizeof...(Args) == Dimensions - 1, "'Encode' must be called with exactly 'Dimensions' arguments.");
113 return EncodeInternal(field1, fields...);
114 }
115
116 static inline T Encode(T *fields) {
117 return apply_variadic_func(fields, std::make_index_sequence<Dimensions>{});
118 }
119
129 static inline auto Decode(T encoding)
130 {
131 return DecodeInternal(encoding, std::make_index_sequence<Dimensions>{});
132 }
133
134 private:
135 MortonNDBmi() = default;
136
137 static const T Selector = BuildSelector<FieldBits>(Dimensions);
138
139 template<typename...Args>
140 static inline T EncodeInternal(T field1, Args... fields)
141 {
142 return EncodeInternal(fields...) | Deposit<Dimensions - sizeof...(fields) - 1>(field1);
143 }
144
145 static inline T EncodeInternal(T field)
146 {
147 return Deposit<Dimensions - 1>(field);
148 }
149
150 template<size_t... i>
151 static inline auto DecodeInternal(T encoding, std::index_sequence<i...>)
152 {
153 return NumVec<T, Dimensions>(Extract<i>(encoding)...);
154 }

156 template<size_t FieldIndex>

158 return _pdep_u32(field, Selector << FieldIndex);
