260void {{NAMESPACE | join("::")}}::{{CLASSNAME}}::endMeshSweep() {
261 if (_state == State::Solve) {
262 synchroniseGlobalResidualAndSolutionUpdate();
263 if ( tarch::mpi::Rank::getInstance().isGlobalMaster() )
264 logInfo( "endMeshSweep()", toString() );
265 }
266
267 // control flow
268 _state = State::Solve;
269
270 switch( _smoothingState ){
271 case SmoothingState::PreSmoothing:
272 // Always increment the counter, and change phase if we pass
273 // the number of steps we specified
274 {
275 if (
276 ++_preSmoothingStepsCounter >= PreSmoothingSteps
277 ) {
278 if ( _activeLevel == CoarsestLevel )
279 // On the coarsest level. Move to PostSmoothing
280 _smoothingState = SmoothingState::PostSmoothing;
281 else
282 // Not yet on coarsest level. Restrict.
283 _smoothingState = SmoothingState::RestrictToNextLevel;
284
285 // reset the counter
286 _preSmoothingStepsCounter = 0;
287 }
288
289 } break;
290
291 case SmoothingState::RestrictToNextLevel:
292 /*
293 Finished restriction. Time to move up one level
294 */
295 {
296 // Always decrement activeLevel. If we are on coarsest level
297 // then we move straight on to post-smoothing
298 if ( --_activeLevel == CoarsestLevel )
299 _smoothingState = SmoothingState::PostSmoothing;
300
301 else
302 _smoothingState = SmoothingState::PreSmoothing;
303 } break;
304
305 case SmoothingState::PostSmoothing:
306 {
307 // Always increment the counter, and change phase if we
308 // pass the number of steps we specified
309 if (
310 ++_postSmoothingStepsCounter >= PostSmoothingSteps
311 ) {
312 /*
313 If we are on the finest level, we want to do one more cycle
314 */
315 if (
316 _activeLevel == getFinestLevel()
317 and
318 _postSmoothingStepsCounter == PostSmoothingSteps
319 ) {
320 // DO NOTHING!! GIVE ME ANOTHER SWEEP
321 }
322
323 /*
324 This is the actual behaviour we want!
325
326 We ask for another sweep at the finest level (see the if statement above)
327 so that the reported numbers match exactly the vesion that was written before (with
328 updating solution at the end of the sweep, rather than the start of the following
329 sweep). Once we are satisifed with the development, we should delete the above.
330 */
331 else if (
332 _activeLevel == getFinestLevel()
333 and
334 _postSmoothingStepsCounter > PostSmoothingSteps
335 ) {
336 _smoothingState = SmoothingState::PreSmoothing;
337
338 // continue with updating solution; set the counter to 1
339 // so that we don't skip updating the solution from here onwards
340 // (we skip solution updates on the first presmoothing/postsmoothing)
341 // sweep to avoid 0 divison by diag
342 _preSmoothingStepsCounter = 1;
343
344 // reset this for next time
345 _postSmoothingStepsCounter = 0;
346
347 incrementVCycleCounter();
348 }
349
350 /*
351 Time to prolongate. First increment the active level, then when we touch a fine
352 vertex, we have the coarse vertices in scope so we can grab contributions we need.
353 */
354 else {
355 _activeLevel++;
356 _smoothingState = SmoothingState::Prolongation;
357 // reset this for next time
358 _postSmoothingStepsCounter = 0;
359 }
360 }
361
362 } break;
363
364 case SmoothingState::Prolongation:
365 {
366 // already increased active level. Time to do post-smoothing
367 _smoothingState = SmoothingState::PostSmoothing;
368 } break;
369
370 default:
371 assertion(false);
372 }
373}
