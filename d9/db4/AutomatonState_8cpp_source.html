115void peano4::grid::AutomatonState::initDatatype() {
116#if !defined(__MPI_ATTRIBUTES_LANGUAGE_EXTENSION__)
117 peano4::grid::AutomatonState instances[2];
118
119 int NumberOfAttributes = 0;
120 NumberOfAttributes++;
121 NumberOfAttributes++;
122 NumberOfAttributes++;
123 NumberOfAttributes++;
124 NumberOfAttributes++;
125 NumberOfAttributes++;
126
127 MPI_Datatype* subtypes = new MPI_Datatype[NumberOfAttributes];
128 int* blocklen = new int[NumberOfAttributes];
129 MPI_Aint* disp = new MPI_Aint[NumberOfAttributes];
130
131 int counter = 0;
132 subtypes[counter] = MPI_INT;
133 blocklen[counter] = 1;
134 counter++;
135 subtypes[counter] = MPI_DOUBLE;
136 blocklen[counter] = Dimensions;
137 counter++;
138 subtypes[counter] = MPI_DOUBLE;
139 blocklen[counter] = Dimensions;
140 counter++;
141 subtypes[counter] = MPI_BYTE;
142 blocklen[counter] = 1;
143 counter++;
144 subtypes[counter] = MPI_UNSIGNED_LONG;
145 blocklen[counter] = 1;
146 counter++;
147 subtypes[counter] = MPI_INT;
148 blocklen[counter] = DimensionsTimesTwo;
149 counter++;
150
151 MPI_Aint baseFirstInstance;
152 MPI_Aint baseSecondInstance;
153 MPI_Get_address(&instances[0], &baseFirstInstance);
154 MPI_Get_address(&instances[1], &baseSecondInstance);
155
156 counter = 0;
157 MPI_Get_address(&(instances[0]._level), &disp[counter]);
158 counter++;
159 MPI_Get_address(&(instances[0]._x.data()[0]), &disp[counter]);
160 counter++;
161 MPI_Get_address(&(instances[0]._h.data()[0]), &disp[counter]);
162 counter++;
163 MPI_Get_address(&(instances[0]._inverted), &disp[counter]);
164 counter++;
165 MPI_Get_address(&(instances[0]._evenFlags), &disp[counter]);
166 counter++;
167 MPI_Get_address(&(instances[0]._accessNumber.data()[0]), &disp[counter]);
168 counter++;
169
170 MPI_Aint offset = disp[0] - baseFirstInstance;
171 MPI_Aint extent = baseSecondInstance - baseFirstInstance - offset;
172 for (int i = NumberOfAttributes - 1; i >= 0; i--) {
173 disp[i] = disp[i] - disp[0];
174 }
175
176 int errorCode = 0;
177 MPI_Datatype tmpType;
178 errorCode += MPI_Type_create_struct(NumberOfAttributes, blocklen, disp, subtypes, &tmpType);
179 errorCode += MPI_Type_create_resized(tmpType, offset, extent, &Datatype);
180 errorCode += MPI_Type_commit(&Datatype);
181 errorCode += MPI_Type_free(&tmpType);
182 if (errorCode)
183 std::cerr << "error constructing MPI datatype in " << __FILE__ << ":" << __LINE__ << std::endl;
184
185 delete[] subtypes;
186 delete[] blocklen;
187 delete[] disp;
188
189#else
190 // invoke routine once to trigger lazy initialisation
191 getForkDatatype();
192 getJoinDatatype();
193 getBoundaryExchangeDatatype();
194 getMultiscaleDataExchangeDatatype();
195 getGlobalCommunciationDatatype();
196#endif
197}
