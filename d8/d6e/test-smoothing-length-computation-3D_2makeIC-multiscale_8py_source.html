74def get_coords(box_xmin, box_xmax, box_ymin, box_ymax, box_zmin, box_zmax, npart):
75 """
76 Generate coordinates for particles between box_min and box_max
77 npart: number of particles in each dimension
78 """
79
80 dx = (box_xmax - box_xmin) / npart
81 dy = (box_ymax - box_ymin) / npart
82 dz = (box_zmax - box_zmin) / npart
83
84 npart3 = npart**3
85
86 # First get uniform coordinates.
87 new_coords = np.zeros((npart3, 3))
88 ind = 0
89 for i in range(npart):
90 x = (i + 0.5) * dx
91 for j in range(npart):
92 y = (j + 0.5) * dy
93 for k in range(npart):
94 z = (k + 0.5) * dy
95 new_coords[ind, 0] = box_xmin + x
96 new_coords[ind, 1] = box_ymin + y
97 new_coords[ind, 2] = box_zmin + z
98 ind += 1
99
100 # Now randomize them a bit.
101 # Taking a completely randomly sampled particle distribution may lead
102 # to trouble the codes aren't set up to handle, such as particles which
103 # are either too close or too far from each other. So use a somewhat
104 # regular distribution instead, by randomly displacing an initially
105 # uniform particle distribution.
106
107 # rng.random returns random numbers in [0, 1). Scale that to [-1, 1)
108 displacement_x = (2.0 * rng.random((npart3, 1)) - 1.0) * max_displacement * dx
109 displacement_y = (2.0 * rng.random((npart3, 1)) - 1.0) * max_displacement * dy
110 displacement_z = (2.0 * rng.random((npart3, 1)) - 1.0) * max_displacement * dz
111 new_coords[:, 0] += displacement_x[:, 0]
112 new_coords[:, 1] += displacement_y[:, 0]
113 new_coords[:, 2] += displacement_z[:, 0]
114
115 return new_coords
