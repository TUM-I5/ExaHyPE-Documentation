183__attribute__((always_inline)) INLINE static int
184invert_dimension_by_dimension_matrix(float A[3][3]) {
185
186#if defined(HYDRO_DIMENSION_3D)
187
188 int pivot[3];
189 for (int i = 0; i < 3; i++) {
190 int imax = i;
191 float Smax = fabsf(A[imax][i]);
192 for (int j = i + 1; j < 3; j++) {
193 const float this_Smax = fabsf(A[j][i]);
194 if (this_Smax > Smax) {
195 Smax = this_Smax;
196 imax = j;
197 }
198 }
199
200 if (Smax < 1.e-8f) {
201 /* singular matrix. Early abort */
202 for (int j = 0; j < 3; j++) {
203 for (int k = 0; k < 3; k++) {
204 A[j][k] = 0.0f;
205 }
206 }
207 return 1;
208 }
209
210 pivot[i] = imax;
211 if (i != imax) {
212 for (int j = 0; j < 3; j++) {
213 const float temp = A[i][j];
214 A[i][j] = A[imax][j];
215 A[imax][j] = temp;
216 }
217 }
218
219 const float Aii_inv = 1.0f / A[i][i];
220 for (int j = i + 1; j < 3; j++) {
221 A[j][i] *= Aii_inv;
222 }
223
224 for (int j = i + 1; j < 3; j++) {
225 for (int k = i + 1; k < 3; k++) {
226 A[j][k] -= A[j][i] * A[i][k];
227 }
228 }
229 }
230
231 for (int i = 0; i < 3; i++) {
232 A[i][i] = 1.0f / A[i][i];
233 for (int j = i + 1; j < 3; j++) {
234 float Aij = 0.0f;
235 for (int k = i; k < j; k++) {
236 Aij -= A[i][k] * A[k][j];
237 }
238 A[i][j] = Aij / A[j][j];
239 }
240 }
241
242 float work[3];
243 for (int jp1 = 3; jp1 > 0; jp1--) {
244 const int j = jp1 - 1;
245 for (int i = 0; i < jp1; i++) {
246 work[i] = A[i][j];
247 }
248 for (int i = jp1; i < 3; i++) {
249 work[i] = 0.0f;
250 }
251 for (int k = jp1; k < 3; k++) {
252 for (int i = 0; i < 3; i++) {
253 work[i] -= A[i][k] * A[k][j];
254 }
255 }
256 for (int i = 0; i < 3; i++) {
257 A[i][j] = work[i];
258 }
259 }
260
261 for (int jp1 = 3; jp1 > 0; jp1--) {
262 const int j = jp1 - 1;
263 const int jp = pivot[j];
264 if (jp != j) {
265 for (int i = 0; i < 3; i++) {
266 const float temp = A[i][j];
267 A[i][j] = A[i][jp];
268 A[i][jp] = temp;
269 }
270 }
271 }
272
273 return 0;
274
275#elif defined(HYDRO_DIMENSION_2D)
276
277 float Ainv[2][2];
278
279 const float detA = A[0][0] * A[1][1] - A[0][1] * A[1][0];
280
281 const float detAinv = (detA != 0.0f) ? 1.0f / detA : 0.0f;
282
283 Ainv[0][0] = A[1][1] * detAinv;
284 Ainv[0][1] = -A[0][1] * detAinv;
285 Ainv[1][0] = -A[1][0] * detAinv;
286 Ainv[1][1] = A[0][0] * detAinv;
287
288 A[0][0] = Ainv[0][0];
289 A[0][1] = Ainv[0][1];
290 A[1][0] = Ainv[1][0];
291 A[1][1] = Ainv[1][1];
292
293 return 0;
294
295#elif defined(HYDRO_DIMENSION_1D)
296
297 if (A[0][0] && !isnan(A[0][0])) {
298 A[0][0] = 1.0f / A[0][0];
299 } else {
300 A[0][0] = 0.0f;
301 }
302
303 return 0;
304
305#else
306
307 error("The dimension is not defined !");
308
309#endif
310}
