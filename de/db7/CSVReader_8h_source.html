56 static T extractFundamentalTypeFromRow(const char separator, const int valueIndex, const std::string row) {
57 T extractedValue;
58 if (std::is_fundamental_v<T> == false) {
59 throw std::runtime_error("Template type is not fundamental.");
60 }
61 // length of string till last found match
62 size_t lastMatch = 0;
63 // length of string till next found match
64 size_t nextMatch = 0;
65 // count posts of the fence
66 int numberOfMatches = 0;
67
68 // loop over separator occurances in row and get substring of value, if possible
69 while ((nextMatch = row.find(separator, lastMatch)) != std::string::npos) {
70 ++numberOfMatches;
71 if (numberOfMatches > valueIndex) {
72 const std::string value = row.substr(lastMatch, nextMatch - lastMatch);
73 std::stringstream(value) >> extractedValue;
74 return extractedValue;
75 }
76
77 // if value not yet between lastMatch and nextMatch, increment lastMatch
78 lastMatch = nextMatch + 1;
79 }
80 std::stringstream ss;
