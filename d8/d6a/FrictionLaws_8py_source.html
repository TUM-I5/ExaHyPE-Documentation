4friction_coefficients = r"""
5 const double Zeta = 35.0 * tarch::la::PI / 180.0; // Conversion of degrees to radians
6 const double Zeta1 = 28.95 * tarch::la::PI / 180.0;
7 const double Zeta2 = 44.09 * tarch::la::PI / 180.0;
8 const double Zeta3 = 31.81 * tarch::la::PI / 180.0;
9
10 const double Mu1 = std::tan(Zeta1);
11 const double Mu2 = std::tan(Zeta2);
12 const double Mu3 = std::tan(Zeta3);
13
14 constexpr double Beta = 1.07;
15 constexpr double Gamma = 2.01;
16 constexpr double BetaStar = 0.06;
17 // Kappa in the friction law is assumed to be 1 and is just initialised as a
18 // const double here using Kappa. The user can apply std::pow((Fr/BetaStar),
19 // Kappa) for evaluating T here in case the value of Kappa is different
20 // from 1.
21
22 constexpr double Kappa = 1.0;
23 constexpr double FrictionLengthScale = 0.00035;
24
25 const double Nu = (2.0 / 9.0) * (FrictionLengthScale / Beta) * (GRAV * std::sin(Zeta) / std::sqrt(GRAV * std::cos(Zeta))) * (((Mu2 - Mu1) / (std::tan(Zeta) - Mu1)) - 1.0);
26 const double Hstart = FrictionLengthScale * (((Mu2 - Mu1) / (std::tan(Zeta) - Mu3)) - 1.0);
27 const double Hstop = FrictionLengthScale * (((Mu2 - Mu1) / (std::tan(Zeta) - Mu1)) - 1.0);
28"""

31 double result{0.0};
32
33 if (tarch::la::greater(Q[s::h], 0.0)) {
34 const double u = Q[s::hu + normal] / Q[s::h];
35 const double c = std::sqrt(GRAV * std::cos(Zeta) * Q[s::h]);
36 result = std::fmax(std::fabs(u - c), std::fabs(u + c));
37 }

64 BTimesDeltaQ[s::h] = 0.0;
65 BTimesDeltaQ[s::hu] = 0.0;
66 BTimesDeltaQ[s::hv] = 0.0;
67 BTimesDeltaQ[s::b] = 0.0;
68
69 const double hx = (Q[s::hx] - 0.5 * deltaQ[s::hx]) / (1.0 * h(0));
70 const double hy = (Q[s::hy] - 0.5 * deltaQ[s::hy]) / (1.0 * h(1));
71
72 // u
73 const double ux = (Q[s::ux] - 0.5 * deltaQ[s::ux]) / (1.0 * h(0));
74 const double uy = (Q[s::uy] - 0.5 * deltaQ[s::uy]) / (1.0 * h(1));
75 const double uxx = (Q[s::uxx] - 0.5 * deltaQ[s::uxx]) / (h(0) * h(0));
76 const double uyy = (Q[s::uyy] - 0.5 * deltaQ[s::uyy]) / (h(1) * h(1));
77 const double uxy = (Q[s::uxy] - 0.5 * deltaQ[s::uxy]) / (4.0 * h(0) * h(1));
78
79 // v
80 const double vx = (Q[s::vx] - 0.5 * deltaQ[s::vx]) / (1.0 * h(0));
81 const double vy = (Q[s::vy] - 0.5 * deltaQ[s::vy]) / (1.0 * h(1));
82 const double vxx = (Q[s::vxx] - 0.5 * deltaQ[s::vxx]) / (h(0) * h(0));
83 const double vyy = (Q[s::vyy] - 0.5 * deltaQ[s::vyy]) / (h(1) * h(1));
84 const double vxy = (Q[s::vxy] - 0.5 * deltaQ[s::vxy]) / (4.0 * h(0) * h(1));
85
86 switch (normal) {
87 case 0: // x
88 BTimesDeltaQ[s::hu] = -(h(0) * Nu * std::sqrt(Q[s::h]) * (1.5 * hx * ux + Q[s::h] * uxx));
89 BTimesDeltaQ[s::hv] = -(h(0) * 0.5 * Nu * std::sqrt(Q[s::h]) * (1.5 * hx * (uy + vx) + Q[s::h] * (uxy + vxx)));
90 break;
91 case 1: // y
92 BTimesDeltaQ[s::hu] = -(h(1) * 0.5 * Nu * std::sqrt(Q[s::h]) * (1.5 * hy * (uy + vx) + Q[s::h] * (uyy + vxy)));
93 BTimesDeltaQ[s::hv] = -(h(1) * Nu * std::sqrt(Q[s::h]) * (1.5 * hy * vy + Q[s::h] * vyy));
94 break;
95 }

111 const double friction{-GRAV * (Mu1 * Q[s::h] * c)};
112 const double velMagnitude{std::sqrt(velMagnitudeSqr)};
113 S[s::hu] += u / velMagnitude * friction;
114 S[s::hv] += v / velMagnitude * friction;

119material_dependent_basal_friction = friction_coefficients + r"""
120 if (tarch::la::greater(Q[s::h], 0.0)) {
121 const double ih = 1.0 / Q[s::h];
122 const double u = ih * Q[s::hu];
123 const double v = ih * Q[s::hv];
124 const double ubar = std::sqrt(Q[s::hu] * Q[s::hu] + Q[s::hv] * Q[s::hv]) * ih;
125 const double Fr = ubar / std::sqrt(GRAV * Q[s::h] * std::cos(Zeta));
126 const double Hstar = (BetaStar + Gamma) * Hstop / Beta;
127
128 double Mu = 1.0;
129 if (Fr > BetaStar) {
130 const double vstop = (Q[s::h] * Beta) / (Fr + Gamma);
131 const double Mustop = Mu1 + (Mu1 - Mu2) / (1.0 + vstop / FrictionLengthScale);
132 Mu = Mustop;
133 } else if (Fr > 0.0 && Fr <= BetaStar) {
134 const double vstart = Q[s::h];
135 const double vstop = Q[s::h] * Beta / (BetaStar + Gamma);
136 const double Mustart = Mu3 + (Mu1 - Mu2) / (1.0 + vstart / FrictionLengthScale);
137 const double Mustop = Mu1 + (Mu1 - Mu2) / (1.0 + vstop / FrictionLengthScale);
138 const double T = Mustart + (Mustop - Mustart) * std::pow((Fr / BetaStar), Kappa);
139 Mu = T;
140 } else if (tarch::la::equals(Fr, 0.0)) {
141 const double vstart = Q[s::h];
142 const double Mustart = Mu3 + (Mu1 - Mu2) / (1.0 + vstart / FrictionLengthScale);
143 Mu = Mustart;
144 }
145
146 if (tarch::la::equals(ubar, 0.0)) {
147 const double hx = Q[s::hx] / h(0);
148 const double hy = Q[s::hy] / h(1);
149 const tarch::la::Vector<Dimensions, double> frictionVector = {std::tan(Zeta) - hx, -hy};
150 const double frictionVectorMagnitude = tarch::la::norm2(frictionVector);
151 if (tarch::la::greater(Mu, frictionVectorMagnitude)) {
152 Mu = frictionVectorMagnitude;
153 }
154
155 if (tarch::la::equals(Mu, 0.0)) {
156 S[s::h] = 0.0;
157 S[s::hu] = Q[s::h] * GRAV * std::sin(Zeta);
158 S[s::hv] = 0.0;
159 } else {
160 S[s::h] = 0.0;
161 S[s::hu] = Q[s::h] * GRAV * std::sin(Zeta) - Mu * (frictionVector(0) / frictionVectorMagnitude) * Q[s::h] * GRAV * std::cos(Zeta);
162 S[s::hv] = -Mu * (frictionVector(1) / frictionVectorMagnitude) * Q[s::h] * GRAV * std::cos(Zeta);
163 }
164 } else if (tarch::la::greater(ubar, 0.0)) {
165 S[s::h] = 0.0;
166 S[s::hu] = Q[s::h] * GRAV * std::sin(Zeta) - Mu * (u / ubar) * Q[s::h] * GRAV * std::cos(Zeta);
167 S[s::hv] = -Mu * (v / ubar) * Q[s::h] * GRAV * std::cos(Zeta);
168 }
169 } else {
170 S[s::h] = 0.0;
171 S[s::hu] = 0.0;
172 S[s::hv] = 0.0;
173 }
