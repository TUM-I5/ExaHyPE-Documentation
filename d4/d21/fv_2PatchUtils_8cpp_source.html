42 static int previousZeroDataEntries = 0;
43
44 // first entry equals zero
45 if ( tarch::la::equals(value,0.0,1e-8) and isFirst ) {
46 previousZeroDataEntries = 1;

49 // first entry does not equal zero

84 const int PatchSize = numberOfVolumesPerAxisInPatch+2*haloSize;
85
86 std::ostringstream result;
87
88 if (prettyPrint) result << std::endl;
89
90 #if Dimensions==2
91 for (int y=0; y<PatchSize; y++) {
92 for (int x=0; x<PatchSize; x++) {
93 int index = (y*PatchSize+x) * (unknowns+auxiliaryVariables);
94
95 // It is a diagonal entry if this counter is bigger than 1. If it equals
96 // 0, it is interior. If it equals 1, then this is a face-connected halo
97 // entry.
98 bool isDiagonal = (x<haloSize and y<haloSize)
99 or (x>=numberOfVolumesPerAxisInPatch+haloSize and y<haloSize)
100 or (x<haloSize and y>=numberOfVolumesPerAxisInPatch+haloSize)
101 or (x>=numberOfVolumesPerAxisInPatch+haloSize and y>=numberOfVolumesPerAxisInPatch+haloSize);
102
103 result << "(";
104 for (int i=0; i<unknowns+auxiliaryVariables; i++) {
105 const int entry = index+i;
106
107 if (not isDiagonal)
108 result << prettyFormat( Q[entry], i==0);
109 else
110 result << "x";
111 }
112 result << ")";
113 }
114 result << std::endl;
115 }
116 #else
117 dfor (k,PatchSize) {
118 int index = peano4::utils::dLinearised(k,PatchSize) * (unknowns+auxiliaryVariables);
119
120 // It is a diagonal entry if this counter is bigger than 1. If it equals
121 // 0, it is interior. If it equals 1, then this is a face-connected halo
122 // entry.
123 bool isDiagonal = (tarch::la::count(k,0) + tarch::la::count(k,PatchSize-1))>1;
124
125 result << "(";
126 for (int i=0; i<unknowns+auxiliaryVariables; i++) {
127 const int entry = index+i;
128

132 result << Q[entry];

135 }

137 }

139
