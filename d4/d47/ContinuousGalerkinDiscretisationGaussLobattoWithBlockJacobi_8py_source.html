144 def _construct_narrow_to_interior_matrix(self):
145 """!
146
147 Construct the matrix which basically picks out the interior points.
148
149 """
150 result = []
151 if self._dimensions==2:
152 for image_x in range(0,self._polynomial_degree - 1):
153 for image_y in range(0,self._polynomial_degree - 1):
154 for image_unknown in range(0,self._unknowns_per_cell_node):
155 result.append([])
156 for preimage_x in range(0,self._polynomial_degree + 1):
157 for preimage_y in range(0,self._polynomial_degree + 1):
158 for preimage_unknown in range(0,self._unknowns_per_cell_node):
159 if (preimage_x-1==image_x and
160 preimage_y-1==image_y and
161 preimage_unknown==image_unknown):
162 result[-1].append( 1.0 )
163 else:
164 result[-1].append( 0.0 )
165 else:
166 for image_x in range(0,self._polynomial_degree - 1):
167 for image_y in range(0,self._polynomial_degree - 1):
168 for image_z in range(0,self._polynomial_degree - 1):
169 for preimage_unknown in range(0,self._unknowns_per_cell_node):
170 result.append([])
171 for preimage_x in range(0,self._polynomial_degree + 1):
172 for preimage_y in range(0,self._polynomial_degree + 1):
173 for preimage_z in range(0,self._polynomial_degree + 1):
174 for preimage_unknown in range(0,self._unknowns_per_cell_node):
175 if (preimage_x-1==image_x and
176 preimage_y-1==image_y and
177 preimage_z-1==image_z and
178 preimage_unknown==image_unknown):
179 result[-1].append( 1.0 )
180 else:
181 result[-1].append( 0.0 )
182 return result
