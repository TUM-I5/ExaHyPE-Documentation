12class SeparateSweeps(RungeKuttaDG):
13 """
14 Probably the simplest solver you could think off.
15 We run over the grid twice per Runge-Kutta step. In the first step,
16 we compute the linear reconstruction, project this one onto the faces
17 and also compute the volumetric prediction. In the second step, we
18 solve the Riemann problem and add it to the prediction.
19
20 In the attribute terminology, we call the two runs over the grid sweeps,
21 i.e., we work with two grid sweeps per Runge-Kutta step.
22 """

110 def create_action_sets(self):
111 """
112 We invoke the superclass which creates all the action sets, and then we
113 set the appropriate guards.
114
115 - _action_set_update_face_label - can be done always
116 - _action_set_update_cell_label - can be done always
117 - _action_set_linear_combination_of_estimates - primary sweeps only, as
118 these kick off the Runge-Kutta step.
119 - _action_set_couple_resolution_transitions_and_handle_dynamic_mesh_refinement -
120 primary only, as the secondary step is a sole clean-up
121 - _action_set_project_linear_combination_onto_faces - only required in
122 primary sweep. In the secondary, we collect the results together, i.e., we
123 don't immediately need the boundary data anymore.
124 - _action_set_solve_volume_integral - only done in the primary sweep.
125 - _action_set_handle_boundary - only in secondary sweep, as we now have
126 access to the left and right face data and hence can invoke the
127 Riemann solver.
128 - _action_set_solve_Riemann_problem - secondary sweep. See discussion of
129 previous item.
130 - _action_set_add_volume_and_face_solution - secondary sweep. The
131 Riemann solve plugs into the first touch of a face in the secondary
132 sweep, so now we can sum up the solutions.
133 - _action_set_compute_final_linear_combination_and_project_solution_onto_faces -
134 very last secondary sweep.
135 - _action_set_AMR - very last secondary sweep, i.e., after we have
136 the final solution.
137 - _action_set_postprocess_solution - same as for item above.
138
139 :: Guards around linear combination of estimates
140
141 We rely on the action set LinearCombination as it is initialised in the
142 superclass. So the only thing we have to do is to ensure that the
143 linear combination over the estimates is computed if and only if this
144 is required. For the separate sweeps, we compute the linear combinations
145 if the cell holds data, i.e., is on the finest level, and if it is the
146 primary of the two sweeps which realise a RK step, i.e., the one where
147 we later on project onto the faces.
148 """
149 super(SeparateSweeps, self).create_action_sets()
150 self._action_set_linear_combination_of_estimates.guards = self._primary_sweeps_of_Runge_Kutta_step_on_cell

152 self._action_set_project_linear_combination_onto_faces.guards = self._primary_sweeps_of_Runge_Kutta_step_on_cell
