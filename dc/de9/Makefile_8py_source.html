416 def add_cpp_file(self, filename, generated=False):
417 """
418 Add a new cpp filename. This is basically a set implementation, i.e., you can
419 add files multiple times, but they are not inserted multiple times. This
420 is important, as the steps add the cpp files. Multiple steps can hold the
421 same action, so this action would be created multiple times.
422
423 All the standard Peano 4 routines rely on this function to add their
424 generated files to the build environment. Nothing stops you however to
425 add more files yourself.
426
427 filename: String
428 Filename of a C++ file. They usually should have the .cpp/.cxx extension.
429 generated: Bool
430 Use this flag for generated files which can be tracked
431 by the cleaning routines 'distclean' or 'maintainer-clean' of the Makefile.
432 """
433 if generated:
434 # Non-generated file takes precedence over generated files.
435 # We only add a generated file if has not been added yet as a non-generated file instead.
436 if (
437 self.generated_cppfiles.count(filename) == 0
438 and self.cppfiles.count(filename) == 0
439 ):
440 self.generated_cppfiles.append(filename)
441 else:
442 # If a file is a non-generated file we definitely want to add it.
443 if self.cppfiles.count(filename) == 0:
444 self.cppfiles.append(filename)
445 # We ensure that we only add a .cpp file once.
446 # Otherwise we would get duplicated symbols errors.
447 # But we never delete a non-generated file.
448 if self.generated_cppfiles.count(filename) > 0:
449 self.generated_cppfiles.remove(filename)
450
451 def remove_cpp_file(self, filename):
452 if filename in self.generated_cppfiles:
453 self.generated_cppfiles.remove(filename)
454 if filename in self.cppfiles:
455 self.cppfiles.remove(filename)

457 def add_Fortran_file(self, filename, generated=False):
458 """
459 Add a new Fortran file.
460
461 All the standard Peano 4 routines rely on this function to add their
462 generated files to the build environment. Nothing stops you however to
463 add more files yourself. Don't add a file multiple times. This might
464 break the compiler.
465
466 Fortran is really picky about the translation order. So you have to add
467 the stuff in the right order. Otherwise, Fortran might complain. This is
468 your responsibility.
469
470 If your file defines a module, please do not use this routine, but use
471 add_Fortran_module() instead.
472
473 filename: String
474 Filename of a Fortran file. They usually should have the .f90 extension.
475 generated: Bool
476 Use this flag for generated files which can be tracked
477 by the cleaning routines 'distclean' or 'maintainer-clean' of the Makefile.
478 """

480 if (
481 self.generated_fortranfiles.count(filename) == 0

484 self.generated_fortranfiles.append(filename)
485 else:
486 if self.fortranfiles.count(filename) == 0:
487 self.fortranfiles.append(filename)
488 if self.generated_fortranfiles.count(filename) > 0:
489 self.generated_fortranfiles.remove(filename)
