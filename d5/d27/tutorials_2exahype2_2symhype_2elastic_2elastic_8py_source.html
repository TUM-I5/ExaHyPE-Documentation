78"""
79We want to define our PDE symbolically.
80For this we use the 'symhype' package (not to be confused with 'sympy') from 'ExaHyPE'.
81"""
82my_pde = exahype2.symhype.FirstOrderConservativePDEFormulation(
83 unknowns=dimensions + (3 if dimensions == 2 else 6), auxiliary_variables=0, dimensions=dimensions
84)
85
86# Define the unknowns.
87# v: velocity components [v_x, v_y, v_z]
88v = my_pde.name_Q_entries(0, dimensions, "v")
89# sigma: stress components [œÉ_xx, œÉ_yy, œÉ_zz, œÉ_xy, œÉ_xz, œÉ_yz]
90sigma = my_pde.name_Q_entries(dimensions, 3 if dimensions == 2 else 6, "sigma")
91
92# Define material parameters.
93rho = sympy.symbols("rho")
94cp = sympy.symbols("cp")
95cs = sympy.symbols("cs")
96
97# Lam√© parameters.
98mu = rho * cs * cs
99lamb = rho * cp * cp - 2.0 * mu
100
101if dimensions == 3:
102 """
103 Define the fluxes in each spatial direction.
104 The overall conservative form is:
105 ‚àÇ‚ÇúQ + ‚àÇ‚Çì(F_x(Q)) + ‚àÇ·µß(F_y(Q)) + ‚àÇùìè(F_z(Q)) = S
106 where Q = [v_x, v_y, v_z, œÉ_xx, œÉ_yy, œÉ_zz, œÉ_xy, œÉ_xz, œÉ_yz]^T.
107 """
108 # Flux [unknowns, dimensions]
109 # --- x-direction (direction index 0) ---
110 # Momentum equations:
111 my_pde.F[0, 0] = -1.0 / rho * sigma[0] # v_x: -œÉ_xx/œÅ
112 my_pde.F[1, 0] = -1.0 / rho * sigma[3] # v_y: -œÉ_xy/œÅ
113 my_pde.F[2, 0] = -1.0 / rho * sigma[4] # v_z: -œÉ_xz/œÅ
114
115 # Stress equations:
116 my_pde.F[3, 0] = -(lamb + 2.0 * mu) * v[0] # œÉ_xx: -(Œª+2Œº)v_x
117 my_pde.F[4, 0] = -lamb * v[0] # œÉ_yy: -Œª v_x
118 my_pde.F[5, 0] = -lamb * v[0] # œÉ_zz: -Œª v_x
119 my_pde.F[6, 0] = -mu * v[1] # œÉ_xy: -Œº v_y
120 my_pde.F[7, 0] = -mu * v[2] # œÉ_xz: -Œº v_z
121 my_pde.F[8, 0] = 0 # œÉ_yz: 0
122
123 # --- y-direction (direction index 1) ---
124 # Momentum equations:
125 my_pde.F[0, 1] = -1.0 / rho * sigma[3] # v_x: -œÉ_xy/œÅ
126 my_pde.F[1, 1] = -1.0 / rho * sigma[1] # v_y: -œÉ_yy/œÅ
127 my_pde.F[2, 1] = -1.0 / rho * sigma[5] # v_z: -œÉ_yz/œÅ
128
129 # Stress equations:
130 my_pde.F[3, 1] = -lamb * v[1] # œÉ_xx: -Œª v_y
131 my_pde.F[4, 1] = -(lamb + 2.0 * mu) * v[1] # œÉ_yy: -(Œª+2Œº)v_y
132 my_pde.F[5, 1] = -lamb * v[1] # œÉ_zz: -Œª v_y
133 my_pde.F[6, 1] = -mu * v[0] # œÉ_xy: -Œº v_x
134 my_pde.F[7, 1] = 0 # œÉ_xz: 0
135 my_pde.F[8, 1] = -mu * v[2] # œÉ_yz: -Œº v_z
136
137 # --- z-direction (direction index 2) ---
138 # Momentum equations:
139 my_pde.F[0, 2] = -1.0 / rho * sigma[4] # v_x: -œÉ_xz/œÅ
140 my_pde.F[1, 2] = -1.0 / rho * sigma[5] # v_y: -œÉ_yz/œÅ
141 my_pde.F[2, 2] = -1.0 / rho * sigma[2] # v_z: -œÉ_zz/œÅ
142
143 # Stress equations:
144 my_pde.F[3, 2] = -lamb * v[2] # œÉ_xx: -Œª v_z
145 my_pde.F[4, 2] = -lamb * v[2] # œÉ_yy: -Œª v_z
146 my_pde.F[5, 2] = -(lamb + 2.0 * mu) * v[2] # œÉ_zz: -(Œª+2Œº)v_z
147 my_pde.F[6, 2] = 0 # œÉ_xy: 0
148 my_pde.F[7, 2] = -mu * v[0] # œÉ_xz: -Œº v_x
149 my_pde.F[8, 2] = -mu * v[1] # œÉ_yz: -Œº v_y
150
151 """
152 Define the eigenvalues.
153 For a 3D elastic system the characteristic speeds in a given
154 direction are:
155 { cp, cs, cs, 0, 0, 0, -cs, -cs, -cp }
156 where:
157 cp = sqrt((Œª+2Œº)/œÅ) is the P-wave (compressional) speed,
158 cs = sqrt(Œº/œÅ) is the S-wave (shear) speed.
159 """
160 # Eigenvalues [unknowns, dimensions]
161 my_pde.eigenvalues[0] = cp
162 my_pde.eigenvalues[1] = cs
163 my_pde.eigenvalues[2] = cs
164 my_pde.eigenvalues[3] = 0
165 my_pde.eigenvalues[4] = 0
166 my_pde.eigenvalues[5] = 0
167 my_pde.eigenvalues[6] = -cs
168 my_pde.eigenvalues[7] = -cs
169 my_pde.eigenvalues[8] = -cp
170else:
171 # Flux [unknowns, dimensions]
172 my_pde.F[0, 0] = -1.0 / rho * sigma[0]
173 my_pde.F[1, 0] = -1.0 / rho * sigma[2]
174 my_pde.F[2, 0] = -(2.0 * mu + lamb) * v[0]
175 my_pde.F[3, 0] = -lamb * v[0]
176 my_pde.F[4, 0] = -mu * v[1]
177
178 my_pde.F[0, 1] = -1.0 / rho * sigma[2]
179 my_pde.F[1, 1] = -1.0 / rho * sigma[1]
180 my_pde.F[2, 1] = -lamb * v[1]
181 my_pde.F[3, 1] = -(2.0 * mu + lamb) * v[1]
182 my_pde.F[4, 1] = -mu * v[0]
183
184 # Eigenvalues [unknowns, dimensions]
185 my_pde.eigenvalues[0] = cp
186 my_pde.eigenvalues[1] = cs
187 my_pde.eigenvalues[2] = 0
188 my_pde.eigenvalues[3] = -cs
189 my_pde.eigenvalues[4] = -cp
190
191"""
192Since 'my_pde' only holds the PDE without initial- or boundary conditions,
193we still need to properly define initial- and boundary conditions.
194This gives us then a complete description of a 'scenario'.
195"""
196for i in range(dimensions + (3 if dimensions == 2 else 6)):
197 my_pde.initial_values[i] = 0
198 my_pde.boundary_values[i] = 0

200"""
201Define a source term.
202Here we add a time-dependent point force applied to one of the
203shear stress components (œÉ_xy in this case).
204"""
205t0 = sympy.symbols("t0")
206M0 = sympy.symbols("M0")
207t = sympy.symbols("t")
208force = M0 * t / (t0 * t0) * sympy.exp(-t / t0)
209
210max_h = sympy.symbols("MaxAdmissibleVolumeH")
211
212if dimensions == 3:
213 point_source = sympy.sqrt((10 - my_pde.x[0])**2 +
214 (10 - my_pde.x[1])**2 +
215 (10 - my_pde.x[2])**2)
216
217 # Apply the force to œÉ_xy, which is the 7th component in the overall Q (index 6).
218 my_pde.sources[0] = 0
219 my_pde.sources[1] = 0
220 my_pde.sources[2] = 0
221 my_pde.sources[3] = 0
222 my_pde.sources[4] = 0
223 my_pde.sources[5] = 0
224 my_pde.sources[6] = sympy.Piecewise((force, point_source <= max_h), (0, True))
225 my_pde.sources[7] = 0
226 my_pde.sources[8] = 0
227else:
228 point_source = sympy.sqrt((10 - my_pde.x[0]) ** 2 + (10 - my_pde.x[1]) ** 2)
229
230 my_pde.sources[0] = 0
231 my_pde.sources[1] = 0
232 my_pde.sources[2] = 0
233 my_pde.sources[3] = 0
234 my_pde.sources[4] = sympy.Piecewise((force, point_source <= max_h), (0, True))
235
236# Substitute numerical values for the material and source parameters.
237my_pde.substitute_expression(rho, 2.7)
238my_pde.substitute_expression(cp, 6.0)
239my_pde.substitute_expression(cs, 3.464)
240my_pde.substitute_expression(t0, 0.1)
241my_pde.substitute_expression(M0, 1000.0)
