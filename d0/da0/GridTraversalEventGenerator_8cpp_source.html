32std::bitset<TwoTimesD> peano4::grid::GridTraversalEventGenerator::areFacesLocal(
33 GridVertex vertices[TwoPowerD],
34 const SplitSpecification& splitTriggered,
35 const std::set<int>& splitting,
36 const std::set<int>& joinTriggered,
37 const std::set<int>& joining
38) const {
39 std::bitset<TwoTimesD> result;
40
41 for (int faceNumber = 0; faceNumber < TwoTimesD; faceNumber++) {
42 bool isLocal = false;
43
44 const int normal = faceNumber % Dimensions;
45 const bool upper = faceNumber >= Dimensions;
46
47 for (int i = 0; i < TwoPowerD; i++) {
48 std::bitset<Dimensions> studiedVertex(i);
49 studiedVertex.set(normal, upper);
50 const int vertexIndex = static_cast<int>(studiedVertex.to_ulong());
51 const GridVertex& vertex = vertices[vertexIndex];
52
53 std::bitset<Dimensions> studiedEntry = TwoPowerD - vertexIndex - 1;
54
55 studiedEntry.set(normal, 0);
56 int currentRank = vertex.getAdjacentRanks(static_cast<int>(studiedEntry.to_ulong()));
57 if (vertex.getState() != GridVertex::State::HangingVertex or upper) {
58 isLocal |= currentRank == _id or splitTriggered.count(currentRank) > 0 or splitting.count(currentRank) > 0;
59 }
60
61 studiedEntry.set(normal, 1);
62 currentRank = vertex.getAdjacentRanks(static_cast<int>(studiedEntry.to_ulong()));
63 if (vertex.getState() != GridVertex::State::HangingVertex or not upper) {
64 isLocal |= currentRank == _id or splitTriggered.count(currentRank) > 0 or splitting.count(currentRank) > 0;
65 }
66
67 if (isLocal) break;
68 }
69
70 result.set(faceNumber, isLocal);
71 }
72
73 return result;
74}

76bool peano4::grid::GridTraversalEventGenerator::isVertexAdjacentToLocalSpacetree(
77 GridVertex vertex,
78 const SplitSpecification& splitTriggered,
79 const std::set<int>& splitting,
80 const std::set<int>& joinTriggered,
81 const std::set<int>& joining,
82 bool splittingIsConsideredLocal,
83 bool joiningIsConsideredLocal
84) const {
85 for (int i = 0; i < TwoPowerD; i++) {
86 const auto adjacentRank = vertex.getAdjacentRanks(i);
87
88 assertion(splitTriggered.count(adjacentRank) <= 1);
89 assertion(splitting.count(adjacentRank) <= 1);
90
91 if (
92 adjacentRank == _id
93 or
94 splitTriggered.count(adjacentRank) == 1
95 or
96 (splittingIsConsideredLocal and splitting.count(adjacentRank) == 1)
97 or
98 (joiningIsConsideredLocal and joining.count(adjacentRank) == 1)
99 ) {
100 return true;
101 }
102 }
103
104 return false;
105}

107bool peano4::grid::GridTraversalEventGenerator::isSpacetreeNodeLocal(
108 GridVertex vertices[TwoPowerD],
109 const SplitSpecification& splitTriggered,
110 const std::set<int>& splitting,
111 const std::set<int>& joinTriggered,
112 const std::set<int>& joining,
113 bool splittingIsConsideredLocal,
114 bool joiningIsConsideredLocal
115) const {
116 dfor2(k)
117 const int rank = vertices[kScalar].getAdjacentRanks(TwoPowerD - kScalar - 1);
118 const auto isHanging = vertices[kScalar].getState() == GridVertex::State::HangingVertex;
119
120 if (
121 not isHanging and rank != _id and splitTriggered.find(rank) == splitTriggered.end()
122 and
123 (not splittingIsConsideredLocal or splitting.find(rank) == splitting.end())
124 and
125 (not joiningIsConsideredLocal or joining.find(rank) == joining.end())
126 ) {
127 return false;
128 }
129 enddforx
130
131 return true;
132}

134std::bitset<TwoPowerD> peano4::grid::GridTraversalEventGenerator::areVerticesAdjacentToParallelDomainBoundary(
135 GridVertex vertices[TwoPowerD],
136 const SplitSpecification& splitTriggered,
137 const std::set<int>& splitting,
138 const std::set<int>& joinTriggered,
139 const std::set<int>& joining,
140 bool calledByLeaveCell
141) const {
142 std::bitset<TwoPowerD> result;
143 for (int j = 0; j < TwoPowerD; j++) {
144 tarch::la::Vector<TwoPowerD, int> adjacency = vertices[j].getBackupOfAdjacentRanks();
145 bool oneLocal = false;
146 bool oneRemote = false;
147 for (int i = 0; i < TwoPowerD; i++) {
148 bool valid = adjacency(i) >= 0 or adjacency(i) == peano4::grid::Spacetree::RankOfPeriodicBoundaryCondition;
149 bool local = adjacency(i) == _id or splitting.count(adjacency(i)) > 0 or splitTriggered.count(adjacency(i)) > 0;
150 oneLocal |= (valid and local);
151 oneRemote |= (valid and not local);
152 }
153 result.set(j, oneLocal and oneRemote);
154 }
155 return result;
156}

201peano4::grid::GridTraversalEvent peano4::grid::GridTraversalEventGenerator::createGenericCellTraversalEvent(
202 GridVertex coarseGridVertices[TwoPowerD],
203 GridVertex fineGridVertices[TwoPowerD],
204 const AutomatonState& state,
205 const SplitSpecification& splitTriggered,
206 const std::set<int>& splitting,
207 const std::set<int>& joinTriggered,
208 const std::set<int>& joining,
209 const tarch::la::Vector<Dimensions, int>& relativePositionToFather,
210 bool spacetreeStateIsRunning
211) const {
212 GridTraversalEvent event;
213 event.setX(state.getX() + state.getH() * 0.5);
214 event.setH(state.getH());
215
216 event.setHasBeenRefined(haveVerticesBeenRefined(fineGridVertices));
217 event.setWillBeRefined(willVerticesBeRefined(fineGridVertices));
218 event.setRelativePositionToFather(relativePositionToFather);
219
220 event.setIsCellLocal(isSpacetreeNodeLocal(fineGridVertices, splitTriggered, splitting, joinTriggered, joining, true, true));
221 event.setIsParentCellLocal(state.getLevel() > 1 and isSpacetreeNodeLocal(coarseGridVertices, splitTriggered, splitting, joinTriggered, joining, true, true));
222 event.setIsFaceLocal(areFacesLocal(fineGridVertices, splitTriggered, splitting, joinTriggered, joining));
223 event.setIsVertexLocal(areVerticesLocal(fineGridVertices, splitTriggered, splitting, joinTriggered, joining));
224
225 event.setIsParentVertexLocal(areVerticesLocal(coarseGridVertices, splitTriggered, splitting, joinTriggered, joining));
226
227 event.setInvokingSpacetree(_id);
228 event.setInvokingSpacetreeIsNotInvolvedInAnyDynamicLoadBalancing(
229 spacetreeStateIsRunning and joinTriggered.empty() and joining.empty() and splitTriggered.empty()
230 and splitting.empty()
231 );
232
233 event.setIsAdjacentCellLocal(0);
234 dfor2(k) {
235 if (fineGridVertices[kScalar].getState() != GridVertex::State::HangingVertex) {
236 dfor2(j) {
237 tarch::la::Vector<Dimensions, int> entryIn3x3Patch = k + j;
238 int owningTree = fineGridVertices[kScalar].getAdjacentRanks(jScalar);
239 event.setIsAdjacentCellLocal(peano4::utils::dLinearised(entryIn3x3Patch, 3), owningTree == _id);
240 event.setIsVertexParentOfSubtree(
241 fineGridVertices[kScalar].getHasBeenParentOfSubtreeVertexInPreviousTreeSweep()
242 );
243 }
244 enddforx
245 }
246 }
247 enddforx
248
249 dfor2(k) {
250 std::set<int> neighbourIds;
251
252 for (int i = 0; i < TwoPowerD; i++) {
253 if (fineGridVertices[kScalar].getAdjacentRanks(i) >= 0) {
254 neighbourIds.insert(fineGridVertices[kScalar].getAdjacentRanks(i));
255 }
256 }
257
258 event.setNumberOfAdjacentTreesPerVertex(kScalar, neighbourIds.size());
259 }
260 enddforx
261
262 return event;
263}

265peano4::grid::GridTraversalEvent peano4::grid::GridTraversalEventGenerator::createLeaveCellTraversalEvent(
266 GridVertex coarseGridVertices[TwoPowerD],
267 GridVertex fineGridVertices[TwoPowerD],
268 const AutomatonState& state,
269 const SplitSpecification& splitTriggered,
270 const std::set<int>& splitting,
271 const std::set<int>& joinTriggered,
272 const std::set<int>& joining,
273 const std::set<int>& hasSplit,
274 const tarch::la::Vector<Dimensions, int>& relativePositionToFather,
275 bool spacetreeStateIsRunning
276) const {
277 GridTraversalEvent event = createGenericCellTraversalEvent(
278 coarseGridVertices,
279 fineGridVertices,
280 state,
281 splitTriggered,
282 splitting,
283 joinTriggered,
284 joining,
285 relativePositionToFather,
286 spacetreeStateIsRunning
287 );
288
289 event.setIsVertexAdjacentToParallelDomainBoundary(areVerticesAdjacentToParallelDomainBoundary(
290 fineGridVertices,
291 splitTriggered,
292 splitting,
293 joinTriggered,
294 joining,
295 true
296 ));
297 event.setIsFaceAdjacentToParallelDomainBoundary(
298 areFacesAdjacentToParallelDomainBoundary(fineGridVertices, splitTriggered, splitting, joinTriggered, joining, true)
299 );
300
301 const std::bitset<Dimensions> coordinates = PeanoCurve::getFirstVertexIndex(state);
302 for (int i = 0; i < TwoPowerD; i++) {
303 const std::bitset<Dimensions> vertexIndex(coordinates ^ std::bitset<Dimensions>(i));
304 const int stackNumber = PeanoCurve::getVertexWriteStackNumber(state, vertexIndex);
305
306 event.setVertexDataFrom(i, vertexIndex.to_ulong());
307 switch (fineGridVertices[vertexIndex.to_ulong()].getState()) {
308 case GridVertex::State::HangingVertex:
309 event.setVertexDataTo(i, TraversalObserver::CreateOrDestroyHangingGridEntity);
310 break;
311 case GridVertex::State::New:
312 case GridVertex::State::Unrefined:
313 case GridVertex::State::Refined:
314 case GridVertex::State::RefinementTriggered:
315 case GridVertex::State::Refining:
316 case GridVertex::State::EraseTriggered:
317 case GridVertex::State::Erasing:
318 event.setVertexDataTo(i, stackNumber);
319 break;
320 case GridVertex::State::Delete: {
321 if (PeanoCurve::isInOutStack(stackNumber)) {
322 event.setVertexDataTo(i, TraversalObserver::CreateOrDestroyPersistentGridEntity);
323 } else {
324 event.setVertexDataTo(i, stackNumber);
325 }
326 } break;
327 }
328
329 if (PeanoCurve::isInOutStack(event.getVertexDataTo(i)) and not event.getIsVertexLocal(vertexIndex.to_ulong())) {
330 logDebug(
331 "createLeaveCellTraversalEvent(...)",
332 "reset event entry "
333 << i << " to no-data as we have " << event.toString()
334 << ". vertex=" << fineGridVertices[vertexIndex.to_ulong()].toString()
335 );
336 event.setVertexDataTo(i, TraversalObserver::NoData);
337 }
338 }
339
340 for (int i = 0; i < 2 * Dimensions; i++) {
341 int faceIndex = PeanoCurve::getFaceNumberAlongCurve(state, i);
342 FaceType type = getFaceType(coarseGridVertices, relativePositionToFather, faceIndex);
343 const int stackNumber = PeanoCurve::getFaceWriteStackNumber(state, faceIndex);
344
345 event.setFaceDataFrom(i, faceIndex);
346
347 switch (type) {
348 case FaceType::Hanging:
349 event.setFaceDataTo(i, TraversalObserver::CreateOrDestroyHangingGridEntity);
350 break;
351 case FaceType::New:
352 case FaceType::Persistent:
353 event.setFaceDataTo(i, stackNumber);
354 break;
355 case FaceType::Delete:
356 if (PeanoCurve::isInOutStack(stackNumber)) {
357 event.setFaceDataTo(i, TraversalObserver::CreateOrDestroyPersistentGridEntity);
358 } else {
359 event.setFaceDataTo(i, stackNumber);
360 }
361 break;
362 }
363
364 if (PeanoCurve::isInOutStack(event.getFaceDataTo(i)) and not event.getIsFaceLocal(faceIndex)) {
365 event.setFaceDataTo(i, TraversalObserver::NoData);
366 }
367 }
368
369 {
370 CellType type = getCellType(coarseGridVertices, relativePositionToFather);
371 const int stackNumber = PeanoCurve::getCellWriteStackNumber(state);
372
373 switch (type) {
374 case CellType::New:
375 event.setCellData(stackNumber);
376 break;
377 case CellType::Persistent:
378 event.setCellData(stackNumber);
379 break;
380 case CellType::Delete:
381 event.setCellData(TraversalObserver::CreateOrDestroyPersistentGridEntity);
382 break;
383 }
384
385 if (
386 // always true here, but if I write it here explicitly, then it is consistent with faces/vertices
387 PeanoCurve::isInOutStack(event.getCellData())
388 and
389 not event.getIsCellLocal()
390 ) {
391 event.setCellData(TraversalObserver::NoData);
392 }
393 }
394
395#if !defined(SharedMemoryParallelisation)
396 for (int i = 0; i < TwoPowerD; i++)
397 assertion1(event.getIsVertexLocal(i) or not event.getIsVertexAdjacentToParallelDomainBoundary(i), event.toString());
398 for (int i = 0; i < TwoTimesD; i++)
399 assertion1(event.getIsFaceLocal(i) or not event.getIsFaceAdjacentToParallelDomainBoundary(i), event.toString());
400#endif
401
402 return event;
403}

462peano4::grid::GridTraversalEvent peano4::grid::GridTraversalEventGenerator::createEnterCellTraversalEvent(
463 GridVertex coarseGridVertices[TwoPowerD],
464 GridVertex fineGridVertices[TwoPowerD],
465 const AutomatonState& state,
466 const SplitSpecification& splitTriggered,
467 const std::set<int>& splitting,
468 const std::set<int>& joinTriggered,
469 const std::set<int>& joining,
470 const std::set<int>& hasSplit,
471 const tarch::la::Vector<Dimensions, int>& relativePositionToFather,
472 bool spacetreeStateIsRunning
473) const {
474 GridTraversalEvent event = createGenericCellTraversalEvent(
475 coarseGridVertices,
476 fineGridVertices,
477 state,
478 splitTriggered,
479 splitting,
480 joinTriggered,
481 joining,
482 relativePositionToFather,
483 spacetreeStateIsRunning
484 );
485
486 event.setIsVertexAdjacentToParallelDomainBoundary(areVerticesAdjacentToParallelDomainBoundary(
487 fineGridVertices,
488 splitTriggered,
489 splitting,
490 joinTriggered,
491 joining,
492 false
493 ));
494 event.setIsFaceAdjacentToParallelDomainBoundary(
495 areFacesAdjacentToParallelDomainBoundary(fineGridVertices, splitTriggered, splitting, joinTriggered, joining, false)
496 );
497
498#if !defined(SharedMemoryParallelisation)
499 for (int i = 0; i < TwoPowerD; i++)
500 assertion1(event.getIsVertexLocal(i) or not event.getIsVertexAdjacentToParallelDomainBoundary(i), event.toString());
501 for (int i = 0; i < TwoTimesD; i++)
502 assertion1(event.getIsFaceLocal(i) or not event.getIsFaceAdjacentToParallelDomainBoundary(i), event.toString());
503#endif
504
505 const std::bitset<Dimensions> coordinates = PeanoCurve::getFirstVertexIndex(state);
506 for (int i = 0; i < TwoPowerD; i++) {
507 const std::bitset<Dimensions> vertexIndex(coordinates ^ std::bitset<Dimensions>(i));
508 const int stackNumber = PeanoCurve::getVertexReadStackNumber(state, vertexIndex);
509 const int vertexPosition = vertexIndex.to_ulong();
510
511 switch (fineGridVertices[vertexPosition].getState()) {
512 case GridVertex::State::HangingVertex:
513 event.setVertexDataFrom(i, TraversalObserver::CreateOrDestroyHangingGridEntity);
514 break;
515 case GridVertex::State::New: {
516 if (PeanoCurve::isInOutStack(stackNumber)) {
517 event.setVertexDataFrom(i, TraversalObserver::CreateOrDestroyPersistentGridEntity);
518 } else {
519 event.setVertexDataFrom(i, stackNumber);
520 }
521 } break;
522 case GridVertex::State::Unrefined:
523 case GridVertex::State::Refined:
524 case GridVertex::State::RefinementTriggered:
525 case GridVertex::State::Refining:
526 case GridVertex::State::EraseTriggered:
527 case GridVertex::State::Erasing:
528 case GridVertex::State::Delete:
529 event.setVertexDataFrom(i, stackNumber);
530 break;
531 }
532 event.setVertexDataTo(i, vertexIndex.to_ulong());
533
534 bool mayResetToNoData = PeanoCurve::isInOutStack(event.getVertexDataFrom(i))
535 and not event.getIsVertexLocal(vertexPosition);
536
537 for (auto p : hasSplit) {
538 mayResetToNoData &= not tarch::la::contains(fineGridVertices[vertexPosition].getAdjacentRanks(), p);
539 }
540
541 if (mayResetToNoData) {
542 event.setVertexDataFrom(i, TraversalObserver::NoData);
543 }
544
545 logDebug(
546 "createEnterCellTraversalEvent(...)",
547 "vertex "
548 << i << " on on tree " << _id << ": " << event.toString() << ", mayResetToNoData=" << mayResetToNoData
549 << ", is-local=" << event.getIsVertexLocal(vertexPosition)
550 << ", is in-out=" << PeanoCurve::isInOutStack(event.getVertexDataFrom(i))
551 << ", vertex[0]=" << fineGridVertices[0].toString() << ", vertex[1]=" << fineGridVertices[1].toString()
552 << ", vertex[2]=" << fineGridVertices[2].toString() << ", vertex[3]=" << fineGridVertices[3].toString()
553 << ", vertex[4]=" << fineGridVertices[4].toString() << ", vertex[5]=" << fineGridVertices[5].toString()
554 << ", vertex[6]=" << fineGridVertices[6].toString() << ", vertex[7]=" << fineGridVertices[7].toString()
555 );
556 }
557
558 for (int i = 0; i < 2 * Dimensions; i++) {
559 int faceIndex = PeanoCurve::getFaceNumberAlongCurve(state, i);
560 FaceType type = getFaceType(coarseGridVertices, relativePositionToFather, faceIndex);
561 const int stackNumber = PeanoCurve::getFaceReadStackNumber(state, faceIndex);
562
563 switch (type) {
564 case FaceType::New: {
565 if (PeanoCurve::isInOutStack(stackNumber)) {
566 event.setFaceDataFrom(i, TraversalObserver::CreateOrDestroyPersistentGridEntity);
567 } else {
568 event.setFaceDataFrom(i, stackNumber);
569 }
570 } break;
571 case FaceType::Hanging:
572 event.setFaceDataFrom(i, TraversalObserver::CreateOrDestroyHangingGridEntity);
573 break;
574 case FaceType::Persistent:
575 case FaceType::Delete:
576 event.setFaceDataFrom(i, stackNumber);
577 break;
578 }
579 event.setFaceDataTo(i, faceIndex);
580
581 bool mayResetToNoData = PeanoCurve::isInOutStack(event.getFaceDataFrom(i)) and not event.getIsFaceLocal(faceIndex);
582
583 for (auto p : hasSplit) {
584 mayResetToNoData &= not tarch::la::contains(getAdjacentRanksOfFace(fineGridVertices, faceIndex, false), p);
585 }
586
587 if (mayResetToNoData) {
588 event.setFaceDataFrom(i, TraversalObserver::NoData);
589 }
590 }
591
592 {
593 CellType type = getCellType(coarseGridVertices, relativePositionToFather);
594 const int stackNumber = PeanoCurve::getCellReadStackNumber(state);
595 switch (type) {
596 case CellType::New:
597 event.setCellData(TraversalObserver::CreateOrDestroyPersistentGridEntity);
598 break;
599 case CellType::Persistent:
600 event.setCellData(stackNumber);
601 break;
602 case CellType::Delete:
603 event.setCellData(stackNumber);
604 break;
605 }
606 }
607
608 bool mayResetToNoData =
609 // always true here, but if I write it here explicitly, then it is consistent with faces/vertices
610 PeanoCurve::isInOutStack(event.getCellData()) and not event.getIsCellLocal();
611
612 for (auto p : hasSplit) {
613 mayResetToNoData &= getTreeOwningSpacetreeNode(fineGridVertices, splitTriggered, splitting, joinTriggered, joining)
614 != p;
615 }
616
617 if (mayResetToNoData) {
618 event.setCellData(TraversalObserver::NoData);
619 }
620
621#if !defined(SharedMemoryParallelisation)
622 for (int i = 0; i < TwoPowerD; i++)
623 assertion1(event.getIsVertexLocal(i) or not event.getIsVertexAdjacentToParallelDomainBoundary(i), event.toString());
624 for (int i = 0; i < TwoTimesD; i++)
625 assertion1(event.getIsFaceLocal(i) or not event.getIsFaceAdjacentToParallelDomainBoundary(i), event.toString());
626#endif
627
628 return event;
629}

631peano4::grid::FaceType peano4::grid::GridTraversalEventGenerator::getFaceType(
632 GridVertex coarseGridVertices[TwoPowerD],
633 tarch::la::Vector<Dimensions, int> positionOfCell,

711peano4::grid::VertexType peano4::grid::GridTraversalEventGenerator::
712 getVertexType(GridVertex coarseGridVertices[TwoPowerD], tarch::la::Vector<Dimensions, int> position, int dimension) {
713 // Helper lambda to determine the base case
714 auto baseCase = [](const GridVertex& vertex) -> VertexType {
715 switch (vertex.getState()) {
716 case GridVertex::State::Refined:
717 case GridVertex::State::EraseTriggered:
718 return VertexType::Persistent;
719 case GridVertex::State::Refining:
720 return VertexType::New;
721 case GridVertex::State::RefinementTriggered:
722 case GridVertex::State::Delete:
723 case GridVertex::State::New:
724 case GridVertex::State::Unrefined:
725 case GridVertex::State::HangingVertex:
726 return VertexType::Hanging;
727 case GridVertex::State::Erasing:
728 return VertexType::Delete;
729 }
730 assertion(false); // Should never happen
731 return VertexType::Hanging; // Default fallback
732 };
733
734 if (dimension < 0) {
735 int linearIndex = utils::dLinearised(position, 2);
736 return baseCase(coarseGridVertices[linearIndex]);
737 }
738
739 if (position(dimension) == 0) {
740 // position(dimension) = 0;
741 return getVertexType(coarseGridVertices, position, dimension - 1);
742 } else if (position(dimension) == 3) {
743 position(dimension) = 1;
744 return getVertexType(coarseGridVertices, position, dimension - 1);
745 }
746
747 tarch::la::Vector<Dimensions, int> positionCopy = position;
748 position(dimension) = 0;
749 VertexType lhs = getVertexType(coarseGridVertices, position, dimension - 1);
750 positionCopy(dimension) = 1;
751 VertexType rhs = getVertexType(coarseGridVertices, positionCopy, dimension - 1);
752
753 // Merge table to avoid branching
754 // Table rows: lhs
755 // Table columns: rhs
756 constexpr VertexType MergeTable[4][4] = {
757 {VertexType::New, VertexType::New, VertexType::Persistent, VertexType::Persistent},
758 {VertexType::New, VertexType::Hanging, VertexType::Persistent, VertexType::Delete},
759 {VertexType::Persistent, VertexType::Persistent, VertexType::Persistent, VertexType::Persistent},
760 {VertexType::Persistent, VertexType::Delete, VertexType::Persistent, VertexType::Delete},
761 };
762 return MergeTable[static_cast<int>(lhs)][static_cast<int>(rhs)];
763}
