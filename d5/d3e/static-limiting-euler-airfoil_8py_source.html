22init = """
23 // Parameters which define a symmetric NACA airfoil
24 constexpr double c = 100.;
25 constexpr double t = 0.12;
26
27 Q[0] = 1.0;
28 Q[1] = 0.1;
29 Q[2] = 0.0;
30 Q[3] = 1.0;
31
32 double yl = x[1] - h[1]/2;
33 double yu = x[1] + h[1]/2;
34 double volumePercentage = 1.0;
35
36 constexpr int nI = 50; // NumberOfIntegrationMeasurementPoints
37 const double hI = h[0] / nI;
38
39 for(int i=0; i<nI; i++){
40 double xI = x[0] + i * hI;
41 double yI = 5.0 * t * c * (
42 0.2969 * sqrt ( xI / c )
43 + ((((
44 - 0.1015 ) * ( xI / c )
45 + 0.2843 ) * ( xI / c )
46 - 0.3516 ) * ( xI / c )
47 - 0.1260 ) * ( xI / c ) ); // Specific function for a symmetric NACA airfoil
48
49 if( -yI<yl && yI>yu ){ // fully within the airfoil
50 volumePercentage -= hI ;
51 }
52 else if ( yI>yl && yI<yu && -yI<yl ){ // y in cell, -y below
53 volumePercentage -= hI*(yI - yl) / (yu - yl);
54 }
55 else if ( -yI<yu && -yI>yl && yI>yu){ // -y in cell, y above
56 volumePercentage -= hI*(yu + yI) / (yu - yl);
57 }
58 else if ( yI>yl && yI<yu && -yI<yu && -yI>yl ){ // -y and y both in cell
59 volumePercentage -= hI*( 2*yI ) / (yu - yl);
60 }
61 }
62
63 Q[4] = std::max(0.0, std::min(1.0, volumePercentage));
64"""

146 double LL[4];
147 double LR[4];
148 eigenvalues(QL, x, h, t, dt, normal, LL);
149 eigenvalues(QR, x, h, t, dt, normal, LR);
150 const double sMax = std::max(*std::max_element(LL, LL+4), *std::max_element(LR, LR+4));
151
152 double fluxLeft[4];
153 double fluxRight[4];
154 flux(QL, x, h, t, dt, normal, fluxLeft);
155 flux(QR, x, h, t, dt, normal, fluxRight);
156
157 const double vL = QL[4];
158 const double vR = QR[4];
159
160 /*
161 Assuming anything that can't get into the other is reflected back to me.
162 own + what's reflected from what I send + what was sent to me and not reflected
163 Rl = vl*Fl + vl*Fl*(1-vr) + vr*Fr*vl
164 Rr = vr*Fr + vr*Fr*(1-vl) + vl*Fl*vr
165 */
166 for(int i=0; i<4; i++){
167 const double FLe = 0.5*vL*(fluxLeft[i]+sMax*QL[i]);
168 const double FRe = 0.5*vR*(fluxRight[i]-sMax*QR[i]);
169
170 //left going update
171 FL[i] = FLe*(2-vR) + vL*FRe;
172 //right going update
173 FR[i] = FRe*(2-vL) + vR*FLe;
174 }
175
176 return sMax;""",
177)
