8class UnknownAttributes:
9 """!
10 Helper class to capture all of the attributes we need for each
11 unknown, but were originally fixed to the PatchFileParser class
12 itself.
13
14 For instance, at time of writing the vis scripts cannot handle
15 displaying separate vtu files for different data that is produced
16 in Peano patch files. An excerpt:
17
18 begin patch
19 offset 9.629630e-01 9.629630e-01
20 size 3.703704e-02 3.703704e-02
21 begin vertex-values "value"
22 0 0 0 0
23 end vertex-values
24 begin vertex-values "rhs"
25 0 0 0 0
26 end vertex-values
27 end patch
28
29 Previously, the only thing that would get parsed is the "value"
30 field, before moving on. We aim to get it to do the "rhs" part
31 as well.
32
33 However, we cannot assume that the number of dofs per axis or
34 unknowns per patch volume will be the same for both. So we
35 create this helper class to hold those.
36
37 We fix is_data_associated_to_cell to be the same as the
38 PatchFileParser object which, in turn, means they must
39 be the same for all of the unknowns
40
41 Attributes:
42 ----------
43
44 file_path: String
45 String of input file path (including filename and extension)
46
47 cell_data: list of patches
48 List of Patches with file data
49
50 dimensions: Integer
51 Dimension of domains
52
53 dof: Integer
54 Number of degrees of freedom per axis
55
56 unknowns: Integer
57 Number of unknowns per patch volume
58
59 mapping: series of d-dimensional tuples
60 Distorts the domain
61
62 set_identifier: String
63 Can be empty to parse everything.
64 """
65

99 def _initialise_default_mapping_if_no_mapping_specified(self):
100 assert not self.mapping, "attribute already set!"
101 vertices_per_axis = self.dof
102 if self.is_data_associated_to_cell:
103 vertices_per_axis += 1
104 if self.mapping == [] and self.dimensions == 2:
105 for y in range(vertices_per_axis):
106 for x in range(vertices_per_axis):
107 self.mapping.append(x * 1.0 / (vertices_per_axis - 1))
108 self.mapping.append(y * 1.0 / (vertices_per_axis - 1))
109 if self.mapping == [] and self.dimensions == 3:
110 for z in range(vertices_per_axis):
111 for y in range(vertices_per_axis):
112 for x in range(vertices_per_axis):
113 self.mapping.append(x * 1.0 / (vertices_per_axis - 1))
114 self.mapping.append(y * 1.0 / (vertices_per_axis - 1))
115 self.mapping.append(z * 1.0 / (vertices_per_axis - 1))

118 self.cell_data.append(patch)

121 return f"{self.unknown_name}: unknowns:{self.unknowns}, dof:{self.dof}"

124class PatchFileParser:
125 """!
126 Parser for Peano block file output.
127
128 This class should be instantiated from within the Visualiser class.
129 This class should be called once per patch file that contains
130 patch data that we want to render. The main method that is called
131 is parse_file().
132
133 This method doesn't return anything. Rather, it sets unknown_attributes
134 to capture all the patch data for each of the unknowns that we find.
135
136 Then, the Visualiser class will steal this data, and combine it into
137 its own helper class.
138
139 Attributes:
140 ----------
141
142 file_path: String
143 String of input file path (including filename and extension)
144
145 dimensions: Integer
146 Dimension of domains
147
148 mapping: series of d-dimensional tuples
149 Distorts the domain
150
151 set_identifier: String
152 Can be empty to parse everything.
153
154 unknown_attributes: Dict{ String, UnknownAttributes }
155 Dictionary that has the names of each unknown we wanna capture
156 as its keys, and the values are the helper class which captures
157 dimensions, cell data etc. Each of the unknowns should be
158 listed during the metadata part of the patch file, ie between
159 "begin cell-metadata" and "end cell-metadata".
160 """
161
162 def __init__(self, file_path, set_identifier, subdomain_number):
163 """ """
164 self.file_path = file_path
165 self.is_data_associated_to_cell = None
166 self.dimensions = -1
167
168 self.set_identifier = set_identifier
169 self.subdomain_number = subdomain_number
170 self.parsed = False
171 self.file_contains_patches = False
172
173 """
174 Here we expected the keys to be a list of unknown names that we
175 have associated with each cell/vertex, and the values to be
176 instances of the helper class UnknownAttributes.
177 """
178 self.unknown_attributes = {}
179
180 def __repr__(self):
181 return f"Parser for {self.file_path}, subdomain {self.subdomain_number}, identifier {self.set_identifier}"

183 def __parse_meta_data_region(self, current_line, file_object, end_condition):
184 """!
185 Pass in file object, move it along and return it
186 """
187 # unknown_name is contained at end of "begin x-metadata" line
188 unknown_name = current_line.strip().split()[-1][1:-1]
189
190 # create new helper class for this
191 # set the bool for data associated with cell to be the same as self.
192 # we set self.is_data_associated_to_cell already, during parse_file(),
193 # just before we came into this function.
194 patch_unknown = UnknownAttributes(
195 unknown_name, self.dimensions, self.is_data_associated_to_cell
196 )
197
198 current_line = file_object.readline().strip()
199
200 use_default_mapping = True
201
202 # do everything that we did in old line reading function
203 while not end_condition in current_line:
204 if "number-of-unknowns" in current_line:
205 patch_unknown.set_unknowns(int(current_line.strip().split()[1]))
206 if "number-of-dofs-per-axis" in current_line:
207 patch_unknown.set_dof(int(current_line.strip().split()[1]))
208 if "description" in current_line:
209 patch_unknown.set_description(
210 current_line.strip().split("description")[1]
211 )
212 if '"' in patch_unknown.description:
213 print(
214 "Warning: data description field "
215 + patch_unknown.description
216 + " holds \" which might lead to issues with VTK's XML export. Remove them"
217 )
218 patch_unknown.description = patch_unknown.description.replace(
219 '"', ""
220 )
221 if "mapping" in current_line:
222 use_default_mapping = False
223 values = current_line.strip().split("mapping")[1]
224 patch_unknown.set_mapping(np.fromstring(values, dtype=float, sep=" "))
225
226 current_line = file_object.readline().strip()
227
228 if use_default_mapping:
229 patch_unknown._initialise_default_mapping_if_no_mapping_specified()
230
231 # finally, add this unknown to our dictionary
232 self.unknown_attributes[unknown_name] = patch_unknown
233
234 return file_object, current_line

236 def __parse_patch_region(
237 self, current_line, file_object, end_condition="end patch"
238 ):
239 """!
240 Pass in file object, run it until we encounter the end condition,
241 and then return it
242
243 We assume that current_line == "begin patch file" at this stage
244
245 This is where we read the actual data
246 """
247 self.file_contains_patches = True
248 # Get patch offset
249 current_line = file_object.readline().strip()
250 line = current_line.strip().split()
251 if self.dimensions == 2:
252 offset = (float(line[1]), float(line[2]))
253 elif self.dimensions == 3:
254 offset = (float(line[1]), float(line[2]), float(line[3]))
255
256 # Get patch size
257 current_line = file_object.readline().strip()
258 line = current_line.strip().split()
259 if self.dimensions == 2:
260 size = (float(line[1]), float(line[2]))
261 elif self.dimensions == 3:
262 size = (float(line[1]), float(line[2]), float(line[3]))
263
264 # run until we have found all of the fields in this
265 # end condition is "end patch" by default
266 while end_condition not in current_line:
267 current_line = file_object.readline().strip()
268 values = None # prevents us adding empty patch to list at end of loop
269 unknown_name = ""
270
271 # get cell data
272 if current_line.startswith("begin cell-values") and (
273 self.set_identifier == ""
274 or current_line.endswith('"' + self.set_identifier + '"')
275 ):
276 assert (
277 self.is_data_associated_to_cell
278 ), "is_data_associated_to_cell flag set incorrectly"
279 unknown_name = current_line.strip().split()[-1][1:-1]
280 current_line = file_object.readline()
281 values = np.fromstring(current_line, dtype=float, sep=" ")
282
283 # do same for vertex data
284 if current_line.startswith("begin vertex-values") and (
285 self.set_identifier == ""
286 or current_line.endswith('"' + self.set_identifier + '"')
287 ):
288 assert (
289 not self.is_data_associated_to_cell
290 ), "is_data_associated_to_cell flag set incorrectly"
291 unknown_name = current_line.strip().split()[-1][1:-1]
292 current_line = file_object.readline()
293 values = np.fromstring(current_line, dtype=float, sep=" ")
294
295 # if we have read some data, let's add it to our list
296 if values is not None:
297 patch = Patch(offset, size, values, unknown_name, self.subdomain_number)
298
299 # by this point, all the unknown_names should be present
300 self.unknown_attributes[unknown_name].append_patch(patch)
301
302 return file_object, current_line

304 def parse_file(self):
305 """!
306 Read file and return cell data, dimensions, number of degrees of freedom and number of unknowns
307
308 Parameters:
309 ----------
310 set_identifier: String
311 Name of the set of unknowns we want to extract. If this is the empty string
312 then I parse all content.
313
314 @todo: clean up the loop below. we can separate into getting metadata, and then reading patches
315 """
316 # print("Reading " + self.file_path + " as subdomain " + str(self.subdomain_number))
317 current_line = ""
318 try:
319 with open(self.file_path, "r") as data_file:
320 current_line = data_file.readline()
321 while current_line:
322 # when we reach EOF, this_line will eval to False
323 """
324 the reason we keep track of 2 variables current_line and
325 line is because we want current_line to eval to False when we reach
326 EOF. However, we need to strip the whitespaces and newline
327 characters off the line for the data processing, without
328 exiting the loop if we have empty lines.
329 """
330 current_line = data_file.readline()
331 stripped_line = current_line.strip()
332 if stripped_line.startswith("dimensions"):
333 self.dimensions = int(stripped_line.strip().split()[1])
334 assert self.dimensions in [
335 2,
336 3,
337 ], "Only 2d and 3d patches are supported"
338
339 # Read out meta data
340 if stripped_line.startswith("begin cell-metadata") and (
341 self.set_identifier == ""
342 or stripped_line.endswith('"' + self.set_identifier + '"')
343 ):
344 self.is_data_associated_to_cell = True
345 # this function advances us along the data file
346 data_file, stripped_line = self.__parse_meta_data_region(
347 stripped_line, data_file, "end cell-metadata"
348 )
349
350 if stripped_line.startswith("begin vertex-metadata") and (
351 self.set_identifier == ""
352 or stripped_line.endswith('"' + self.set_identifier + '"')
353 ):
354 self.is_data_associated_to_cell = False
355 # this function advances us along the data file
356 data_file, stripped_line = self.__parse_meta_data_region(
357 stripped_line, data_file, "end vertex-metadata"
358 )
359
360 elif stripped_line.startswith("begin patch"):
361 # pass data_file into self.__parse_patch_region, which will modify and return it
362 # ie we move further along the file inside this function
363 # appending new patches happens in this function
364 data_file, stripped_line = self.__parse_patch_region(
365 stripped_line, data_file
366 )
367
368 # self._initialise_default_mapping_if_no_mapping_specified() # handled in meta data reader
369 self.parsed = True
370 except Exception as e:
371 print(
372 "Error: was not able to read "
373 + self.file_path
374 + ": "
375 + str(e)
376 + " in line "
377 + current_line
378 )
379 self.parsed = False
