47void toolbox::loadbalancing::Blacklist::update() {
48 for (auto& p : _blacklist) {
49 if (peano4::parallel::SpacetreeSet::getInstance().getGridStatistics(p.first).getRemovedEmptySubtree()) {
50 logDebug("update()", "Tree " << p.first << " has degenerated child. Set/keep on blacklist");
51 p.second.numberOfDegeneratedChildren++;
52 p.second.lifetime++;
53 } else if (p.second.lifetime > 0) {
54 const int currentNumberOfCells = peano4::parallel::SpacetreeSet::getInstance()
55 .getGridStatistics(p.first)
56 .getNumberOfLocalUnrefinedCells();
57 if (currentNumberOfCells == p.second.numberOfUnrefinedCells) {
58 logDebug(
59 "updateBlacklist()",
60 "Tree "
61 << p.first
62 << " is on blacklist but seems to have failed to fork off cells successfully, as number of cells did not decrease since last split. Keep on blacklist"
63 );
64 } else if (p.second.lifetime == 1) {
65 logDebug("updateBlacklist()", "Remove tree " << p.first << " from blacklist (keep but set at inactive)");
66 p.second.lifetime = 0;
67 } else if (p.second.lifetime > 0) {
68 p.second.lifetime--;
69 }
70 }
71 }
72}

74void toolbox::loadbalancing::Blacklist::triggeredSplit(int newParent) {
75 if (_blacklist.count(newParent) == 0) {
76 _blacklist.insert(std::pair<int, BlacklistData>(newParent, BlacklistData(newParent)));
77 } else if (_blacklist.at(newParent).lifetime == 0) {
78 _blacklist.at(newParent) = BlacklistData(newParent);
79 } else {
80 logDebug(
81 "triggerSplit()",
82 "Split local rank "
83 << newParent
84 << " though it had been on the blacklist. This happens usually if and only if a tree split multiple times in one iteration"
85 );
86 _blacklist.at(newParent).numberOfSplits++;
87 }
