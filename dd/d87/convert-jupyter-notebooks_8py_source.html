25def postprocess_markdown_line(line):
26 """!
27
28 currently_in_raw_data_box: Jupyter's markdown export does not preserve raw
29 data cells. But we use them quite frequently. So we follow the convention
30 that raw data cells should be indendented. And then we automatically
31 filter those lines out, and add the markup box around them.
32 """
33 global current_mode
34
35 if "\\begin{equation" in line and current_mode == Mode.Text:
36 current_mode = Mode.EquationSystem
37 line = """ @f{eqnarray*}{
38"""
39 elif "\\end{equation" in line:
40 current_mode = Mode.Text
41 line = """ @f}
42 """
43# elif line.startswith( " " ) and current_mode==Mode.Text:
44# current_mode = Mode.RawData
45# line = """
46#
47#~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
48#""" + line
49# elif not line.startswith( " " ) and current_mode==Mode.RawData:
50# current_mode = Mode.Text
51# line = """
52#~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
53#
54#""" + line
55 elif current_mode == Mode.Text:
56 line = line.replace( "$$", "\n @f$ ")
57 # This replacement messes up any dollar sign in raw data cells (which I cannot detect atm)
58 # line = line.replace( "$", "@f$ ")
59 return line

62def postprocess_notebook_markdown(filename: str):
63 """!
64
65 Takes the markdown produced by nbconvert into something that plays along
66 nicely with our Doxygen.
67
68 filename: File without extension
69
70 """
71 ouput_file_name = filename + ".dox"
72 output_file = open( ouput_file_name, "w" )
73
74 output_file.write( "/**\n" )
75
76 subpages = []
77
78 have_created_header = False
79 with open(filename + ".md") as file:
80 for line in file:
81 if line.startswith("#") and not have_created_header:
82 line = line.replace("#", "")
83 path = filename.replace( "./", "" ).replace( "/", "_" ).replace( ".", "_" )
84 output_file.write( "@page " + path + " " + line)
85 output_file.write( """
86<div style="background-color: #fcc ; padding: 10px; border: 1px solid green;">
87This documentation is autogenerated from {}.ipynb. Once you clone the Peano repository, please work against this notebook or study
88Peano's @ref page_tutorials_overview "generic description how to generate web docu from Jupyter notebooks".
89</div>
90
91
92 """.format(filename))
93 have_created_header = True
94 elif "@subpage" in line:
95 new_subpage = line.split( "@subpage" )[1].strip().split( " ")[0]
96 subpages.append( new_subpage )
97 output_file.write( line.replace( "@subpage", "@ref" ))
98 elif "\\subpage" in line:
99 new_subpage = line.split( "\\subpage" )[1].strip().split( " ")[0]
100 subpages.append( new_subpage )
101 output_file.write( line.replace( "\\subpage", "@ref" ))
102 else:
103 output_file.write( postprocess_markdown_line(line) )
104
105
106 subpages = list(dict.fromkeys(subpages))
107 for subpage in subpages:
108 if subpage.endswith("."):
109 subpage = subpage[0:-1]
110 output_file.write( """
111@subpage {}
112 """.format(subpage) )
113 print ("add subpage {}".format(subpage) )
114 output_file.write( "*/ \n ")
115

119 for filename in [f for f in filenames if f.endswith(".ipynb")]:

127 print( "postprocess markdown file {}.md".format(name_without_extension) )
128 postprocess_notebook_markdown( name_without_extension )
129 print( "delete markdown file" )
