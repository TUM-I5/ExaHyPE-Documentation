94 MPI_Datatype* subtypes = new MPI_Datatype[NumberOfAttributes];

105 subtypes[counter] = MPI_DOUBLE;
106 blocklen[counter] = Dimensions;
107 counter++;
108 subtypes[counter] = MPI_DOUBLE;
109 blocklen[counter] = Dimensions;
110 counter++;
111
112 MPI_Aint baseFirstInstance;
113 MPI_Aint baseSecondInstance;
114 MPI_Get_address(&instances[0], &baseFirstInstance);
115 MPI_Get_address(&instances[1], &baseSecondInstance);
116
117 counter = 0;
118 MPI_Get_address(&(instances[0]._refinementControl), &disp[counter]);
119 counter++;
120 MPI_Get_address(&(instances[0]._offset.data()[0]), &disp[counter]);
121 counter++;
122 MPI_Get_address(&(instances[0]._width.data()[0]), &disp[counter]);
123 counter++;
124 MPI_Get_address(&(instances[0]._h.data()[0]), &disp[counter]);
125 counter++;
126
127 MPI_Aint offset = disp[0] - baseFirstInstance;
128 MPI_Aint extent = baseSecondInstance - baseFirstInstance - offset;
129 for (int i = NumberOfAttributes - 1; i >= 0; i--) {
130 disp[i] = disp[i] - disp[0];
131 }
132
133 int errorCode = 0;
134 MPI_Datatype tmpType;
135 errorCode += MPI_Type_create_struct(NumberOfAttributes, blocklen, disp, subtypes, &tmpType);
136 errorCode += MPI_Type_create_resized(tmpType, offset, extent, &Datatype);
137 errorCode += MPI_Type_commit(&Datatype);
138 errorCode += MPI_Type_free(&tmpType);
139 if (errorCode)
140 std::cerr << "error constructing MPI datatype in " << __FILE__ << ":" << __LINE__ << std::endl;
141
142 delete[] subtypes;
143 delete[] blocklen;
144 delete[] disp;
145
146#else
147 // invoke routine once to trigger lazy initialisation
148 getForkDatatype();
149 getJoinDatatype();
150 getBoundaryExchangeDatatype();
151 getMultiscaleDataExchangeDatatype();
152 getGlobalCommunciationDatatype();
153#endif
154}
