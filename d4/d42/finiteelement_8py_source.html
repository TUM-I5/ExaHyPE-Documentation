82 def evaluate(self, k, x_local):
83 """Evaluate the k-th basis function at the point x_local = (xi,eta)
84 2D basis functions are enumerated as follows
85 !!! So far, for 2D only
86
87 y
88 |
89 2 | 6 7 8 k = i + j * (self.basis.degree+1)
90 j 1 | 3 4 5 => i = k % (self.basis.degree+1)
91 0 | 0 1 2 j = k // (self.basis.degree+1)
92 - - - - x
93 0 1 2
94 i
95
96 :arg k: index of basis function
97 :arg x_local: position (xi,eta) in [-1,+1] x [-1,+1]
98 """
99 i, j = k % (self.degree + 1), k // (self.degree + 1)
100 return self.basis.evaluate(self.degree, i, x_local[0]) * self.basis.evaluate(self.degree, j, x_local[1])
101 # i, j = k % (self.basis.degree + 1), k // (self.basis.degree + 1)
102 # return self.basis.evaluate(i, x_local[0]) * self.basis.evaluate(j, x_local[1])

104 def evaluate_derivative(self, k, direction, x_local):
105 """Evaluate the derivative of k-th basis function at the point x_local = (xi,eta) in the
106 direction dir which can be 0 (x-direction) or 1 (y-direction)
107
108 :arg k: index of basis function
109 :arg direction: direction (0 or 1)
110 :arg x_local: position (xi,eta) in [-1,+1] x [-1,+1]
111 """
112 i, j = k % (self.degree + 1), k // (self.degree + 1)
113 if direction == 0:
114 return self.basis.evaluate_derivative(self.degree, i, x_local[0]) * self.basis.evaluate(self.degree, j, x_local[1])
115 elif direction == 1:
116 return self.basis.evaluate(self.degree, i, x_local[0]) * self.basis.evaluate_derivative(self.degree, j, x_local[1])
117 else:
118 raise ValueError("Direction can only be 0 (x-direction) or 1 (y-direction)")
119
120 @property
121 def nodes(self):
122 """Compute nodal points on reference entity"""
123
124 nodes = []
125
126 index_set = itertools.product(range(self.degree + 1), repeat=self.dim) # create tuples for possible indices combinations
127 for index in index_set:
128 # reverse (0,1,2) to (2,1,0)
129 # required as itertools.product() generates tuples like (0,0), (0,1), (0,2)
130 # -- with the last index advancing faster, but we need (0,0), (1,0), (2,0)
131 #index = next(index_set)
132 reverse_index = tuple(reversed(index))
133 node_coordinates = [self.basis.node1d(self.degree, idx) for idx in reverse_index]
134 nodes.append(np.asarray(node_coordinates))
135
136 return nodes
