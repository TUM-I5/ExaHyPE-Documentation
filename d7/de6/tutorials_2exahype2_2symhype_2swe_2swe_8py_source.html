3import sympy

5"""
6We import the required modules for this project.
7We always need the 'peano4' module as this is our core project.
8Since we are creating an ExaHyPE 2 application, we additionally
9need to import the 'exahype2' module.

11import peano4
12import exahype2
13
14"""
15The number of finite volumes per axis in one patch.
16"""
17patch_size = 16
18
19"""
20The number of levels the mesh is refined.
21"""
22depth = 4
23
24"""
25The simulation end time.
26"""
27end_time = 1.0
28
29"""
30Choose domain size and offset.
31"""
32size = [2.0, 2.0]
33offset = [0.0, 0.0]
34
35"""
36Choose how often a snapshot is written.
37"""
38time_in_between_two_snapshots = end_time / 20
39# time_in_between_two_snapshots = 0 # Set to 0 to disable I/O
40
41"""
42Switch between 'Release', 'Debug', 'Asserts', 'Trace', 'Stats'.
43"""
44compile_mode = "Release"
45
46"""
47We first create a new ExaHyPE 2 project.
48For this, we specify the (nested) namespaces, the name of our main file and our executable name.
49"""
50my_project = exahype2.Project(
51 namespace=["tutorials", "exahype2", "symhype", "swe"],
52 project_name="SWE",
53 directory=".",
54 executable="SWE." + str(compile_mode),
55)
56
57"""
58Add the solver using named arguments.
59This is the way you can add further PDE terms.
60This requires the 'BlockStructured' toolbox and 'ExaHyPE' to be built.
61"""
62my_solver = exahype2.solvers.fv.godunov.GlobalAdaptiveTimeStep(
63 name="FVSolver",
64 patch_size=patch_size,
65 unknowns=2 + 1, # [h, hu, hv]
66 auxiliary_variables=1, # Bathymetry (b)
67 min_volume_h=(1.1 * min(size) / (3.0**depth)),
68 max_volume_h=(1.1 * min(size) / (3.0**depth)),
69 time_step_relaxation=0.5,
70)
71
72"""
73We want to define our PDE symbolically.
74For this we use the 'symhype' package (not to be confused with 'sympy') from 'ExaHyPE'.
75"""
76my_pde = exahype2.symhype.FirstOrderConservativePDEFormulation(
77 unknowns=2 + 1, auxiliary_variables=1, dimensions=2
78)
79h = my_pde.name_Q_entry(0, "h")
80hu = my_pde.name_Q_entry(1, "hu")
81hv = my_pde.name_Q_entry(2, "hv")
82b = my_pde.name_auxiliary_variable(0, "b")
83
84# Gravitational constant
85g = sympy.symbols("g")
86
87# Speed in x-direction
88u = hu / h
89
90# Speed in y-direction
91v = hv / h
92
93"""
94Define the equation system
95"""
96# Flux [unknowns, dimensions]
97my_pde.F[0, 0] = hu
98my_pde.F[1, 0] = hu * u + (g * h**2) / 2
99my_pde.F[2, 0] = h * u * v
100
101my_pde.F[0, 1] = hv
102my_pde.F[1, 1] = h * u * v
103my_pde.F[2, 1] = hv * v + (g * h**2) / 2
104
105# Eigenvalues [unknowns, dimensions]
106my_pde.eigenvalues[0, 0] = u
107my_pde.eigenvalues[0, 1] = v
108
109my_pde.eigenvalues[1, 0] = u + sympy.sqrt(g * h)
110my_pde.eigenvalues[1, 1] = v + sympy.sqrt(g * h)
111my_pde.eigenvalues[2, 0] = u - sympy.sqrt(g * h)
112my_pde.eigenvalues[2, 1] = v - sympy.sqrt(g * h)
113
114# NCP [unknowns, dimensions]
115delta_b = my_pde.grad(b)
116
117my_pde.ncp[0, 0] = 0.0
118my_pde.ncp[1, 0] = g * h * delta_b
119my_pde.ncp[2, 0] = 0.0
120
121my_pde.ncp[0, 1] = 0.0
122my_pde.ncp[1, 1] = 0.0
123my_pde.ncp[2, 1] = g * h * delta_b
124
125"""
126Since 'my_pde' only holds the PDE without initial- or boundary conditions,
127we still need to properly define initial- and boundary conditions.
128This gives us then a complete description of a 'scenario'.
129"""
130
131# Initial conditions as specified in the documentation.
132my_pde.initial_values[0] = 1 # Water height (h)
133my_pde.initial_values[1] = 0 # x-momentum (hu)
134my_pde.initial_values[2] = 0 # y-momentum (hv)
135
136volume_centre = sympy.sqrt((1.0 - my_pde.x[0]) ** 2 + (1.0 - my_pde.x[1]) ** 2)
137my_pde.initial_values[3] = sympy.Piecewise((0.0, volume_centre < 0.4), (0.1, True)) # Bathymetry (b)
138
139# Boundary conditions
140my_pde.boundary_values[0] = h
141my_pde.boundary_values[1] = -hu
142my_pde.boundary_values[2] = -hv
143my_pde.boundary_values[3] = b
144
145my_pde.substitute_expression(g, 9.81)
146
147"""
148Specify which implementation our solvers uses.
149Here we want to set the implementation we get from our symbolically defined PDE,
150i.e., we get the C++ implementation which is generated by ExaHyPE's 'symhype' package.
151"""
152my_solver.set_implementation(
153 initial_conditions=my_pde.implementation_of_initial_conditions(),
154 boundary_conditions=my_pde.implementation_of_boundary_conditions(),
155 flux=my_pde.implementation_of_flux(),
156 max_eigenvalue=my_pde.implementation_of_max_eigenvalue(),
157 ncp=my_pde.implementation_of_ncp(),
158)
159
160"""
161To see which variables (unknowns + auxiliary variables) we can visualise,
162let's add a plot description for the variables to our solver.
163"""
164my_solver.plot_description = my_pde.unknown_identifier_for_plotter()
165
166"""
167Add the solver to our project
168"""
169my_project.add_solver(my_solver)
170
171"""
172Configure some global parameters
173"""
174my_project.set_global_simulation_parameters(
175 dimensions=2, # The dimensionality is fixed for SWE
176 size=size,
177 offset=offset,
178 min_end_time=end_time,
179 max_end_time=end_time,
180 first_plot_time_stamp=0.0,
181 time_in_between_plots=time_in_between_two_snapshots,
182 periodic_BC=[False, False, False],
183)
184
185"""
186This defines where the output files should go.
187If you omit this, output files are automatically put into the application's folder
188"""
189my_project.set_output_path("solutions")
190
191"""
192Configure load balancer for parallel execution.
193"""
194my_project.set_load_balancer("new ::exahype2::LoadBalancingConfiguration")
195
196"""
197We need to set the location of our core libraries ('Peano4').
198This helps us to resolve any dependencies.
199Additionally, we specify the build mode which you can also change to a different mode.
200"""
201my_project.set_Peano4_installation("../../../../", mode=peano4.output.string_to_mode(compile_mode))
202
203"""
204We generate and grab the underlying core project of 'Peano4'.
205This gives us access to some functions we want to use to finalise and build this project.
206"""
207my_project = my_project.generate_Peano4_project(verbose=False)
208
209"""
210Finally, we want to build our project.
211First, all of the necessary glue code is generated in the application folder,
212then 'make' is invoked automatically which compiles the generated code and links against our core libraries
213and toolboxes which have been built before.
214You can also always invoke 'make' yourself to compile, or cleanup with 'make clean'.
215"""
216my_project.build(make=True, make_clean_first=True, throw_away_data_after_build=True)
