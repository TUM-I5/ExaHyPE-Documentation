51 assertion(lifetime >= 1);
52
53 switch (command) {
54 case exahype2::RefinementCommand::Refine: {
55 tarch::la::Vector<Dimensions, double> expandedWidth = (1.0 + _tolerance) * h;
56 tarch::la::Vector<Dimensions, double> refinedMeshSize = (1.0 + _tolerance) * 1.0 / 3.0 * h;
57
58 peano4::grid::GridControlEvent newEvent(
59 peano4::grid::GridControlEvent::RefinementControl::Refine,
60 x - 0.5 * expandedWidth,
61 expandedWidth,
62 refinedMeshSize
63 );
64
65 assertion1(newEvent.getH(0) > 0.0, newEvent.toString());
66 assertion1(newEvent.getWidth(0) > 0.0, newEvent.toString());
67 _newEvents.push_back(std::pair<peano4::grid::GridControlEvent, int>(newEvent, lifetime));
68 logDebug(
69 "addCommand()",
70 "Added refinement for x="
71 << x << ", h=" << h << ": " << newEvent.toString() << " (total of " << _newEvents.size() << " instructions)"
72 );
73 } break;
74 case exahype2::RefinementCommand::Keep:
75 break;
76 case exahype2::RefinementCommand::Erase: {
77 tarch::la::Vector<Dimensions, double> expandedWidth = (1.0 + _tolerance) * h;
78 tarch::la::Vector<Dimensions, double> shift = 0.5 * (expandedWidth - h);

80
81 peano4::grid::GridControlEvent newEvent(

85 refinedMeshSize
86 );

91 logDebug(
92 "addCommand()",
93 "Added erase for x="
94 << x << ", h=" << h << ": " << newEvent.toString() << " (total of " << _newEvents.size() << " instructions)"
95 );
96 } break;
97 }
98
99 logTraceOutWith1Argument("addCommand()", _newEvents.size());
