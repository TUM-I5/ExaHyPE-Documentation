38void exahype2::EnclaveBookkeeping::fusePendingTasks() {
39 for (auto& [taskType, taskList] : pendingTasks) {
40 if (!taskList.empty()) {
41 // Get the first task and remove it from the list
42 tarch::multicore::Task* firstTask = taskList.front();
43 taskList.pop_front();
44
45 // Fuse the first task with the remaining tasks in the list
46 // The fuse also deletes every task in the list
47 firstTask->fuse(taskList);
48
49 // Clear the remaining tasks since they've been fused
50 taskList.clear();
51
52 delete firstTask;
53 }
54 }
