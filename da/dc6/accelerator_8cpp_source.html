40void tarch::accelerator::freeRawData(void* data, MemoryLocation, int) {
41 if (!data) {
42 return;
43 }
44
45#if PeanoDebug > 0
46 {
47 tarch::multicore::Lock lock(Device::getInstance().getMemoryBlockSemaphore());
48 auto itr = Device::getInstance().getMemoryBlockTable().find(data);
49 if (itr == Device::getInstance().getMemoryBlockTable().end()) {
50 logDebug("freeRawData()", "Attempt to free invalid pointer")
51 return;
52 }
53 Device::getInstance().getMemoryBlockTable().erase(itr);
54 }
55#endif
56
57 std::free(data);
58}

92void* tarch::accelerator::reallocateRawData(
93 void* data,
94 std::size_t sizeInBytes,

96 int device
97) {
98 assertion(data);
99 assertion2(sizeInBytes > 0, sizeInBytes, toString(location));

101#if PeanoDebug > 0

104 tarch::multicore::Lock lock(Device::getInstance().getMemoryBlockSemaphore());

107 if (currentSizeInBytes >= sizeInBytes) {

110#endif
111
112 std::size_t newSizeInBytes = sizeInBytes + sizeInBytes / 2;
113 void* newData = allocateRawData(newSizeInBytes, location, device);
114 assertion(newData);

116 // TODO: Copy host to host
117 freeRawData(data, location, device);
118

126std::size_t tarch::padSizeToAlignment(std::size_t size, std::size_t alignment) {
127 assertion1(size > 0, size);
128 assertion1(alignment > 0, alignment);
129 // Aligned alloc requires size to be a multiple of alignment, therefore we
130 // need to check it. Also see:
131 // https://en.cppreference.com/w/c/memory/aligned_alloc
132 // Pad to nearest multiple of alignment.
133 int padding = size % alignment == 0 ? 0 : alignment - (size % alignment);
134 return size + padding;
135}
