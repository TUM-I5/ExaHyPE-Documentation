60void YAMLInput::parseGrid(uint a_grid[3], uint ez, uint ey , uint ex){
61 std::vector<uint> grid;
62
63 if(!config["domain"]["grid"] && !config["domain"]["mode"]){
64 std::stringstream ss;
65 ss << "Grid specifiation not valid: Missing keys are: " << std::endl;
66 ss << "domain:" << std::endl;
67 ss << "grid: [int, int ,int]" << std::endl;
68 ss << "or" << std::endl;
69 ss << "mode: [int, int ,int]" << std::endl;
70 }
71 if(bool(config["domain"]["grid"])){
72 grid = config["domain"]["grid"] .as<std::vector<uint>> ();
73 }else{
74 grid = config["domain"]["mode"] .as<std::vector<uint>> ();
75 grid[0] = (grid[0]-1) * ez + 1;
76 grid[1] = (grid[1]-1) * ey + 1;
77 grid[2] = (grid[2]-1) * ex + 1;
78 }
79 std::copy_n(grid.data(),3 ,a_grid);
80}
