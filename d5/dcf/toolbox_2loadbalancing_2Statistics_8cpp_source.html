30void toolbox::loadbalancing::Statistics::waitForGlobalDataExchange() {
31#ifdef Parallel
32 if (_globalNumberOfInnerUnrefinedCellsRequest != nullptr) {
33 MPI_Wait(_globalNumberOfInnerUnrefinedCellsRequest, MPI_STATUS_IGNORE);
34 MPI_Wait(_globalNumberOfSplitsRequest, MPI_STATUS_IGNORE);
35 MPI_Wait(_globalNumberOfTreesRequest, MPI_STATUS_IGNORE);
36 MPI_Wait(_globalNumberOfRanksWithEnabledLoadBalancingRequest, MPI_STATUS_IGNORE);
37
38 delete _globalNumberOfInnerUnrefinedCellsRequest;

44 _globalNumberOfSplitsRequest = nullptr;

46 _globalNumberOfRanksWithEnabledLoadBalancingRequest = nullptr;

51void toolbox::loadbalancing::Statistics::updateGlobalView() {
52 _localNumberOfInnerUnrefinedCells = peano4::parallel::SpacetreeSet::getInstance()
53 .getGridStatistics()
54 .getNumberOfLocalUnrefinedCells();
55 _localNumberOfTrees = peano4::parallel::SpacetreeSet::getInstance().getLocalSpacetrees().size();
56
57 _isViewConsistent = true;
58
59 if (tarch::mpi::Rank::getInstance().getNumberOfRanks() <= 1) {
60 _globalNumberOfInnerUnrefinedCells = _localNumberOfInnerUnrefinedCells;
61 _globalNumberOfSplits = _localNumberOfSplits;
62 _globalNumberOfTrees = _localNumberOfTrees;
63 _globalNumberOfRanksWithEnabledLoadBalancing = _isLocalLoadBalancingEnabled ? 1 : 0;
64 } else {
65#ifdef Parallel
66 waitForGlobalDataExchange();
67
68 _globalNumberOfInnerUnrefinedCells = _globalNumberOfInnerUnrefinedCellsBufferIn;
69 _globalNumberOfSplits = _globalNumberOfSplitsIn;
70 _globalNumberOfTrees = _globalNumberOfTreesIn;
71 _globalNumberOfRanksWithEnabledLoadBalancing = _globalNumberOfRanksWithEnabledLoadBalancingIn;
72
73 if (_globalNumberOfInnerUnrefinedCells < _localNumberOfInnerUnrefinedCells) {
74 logDebug(
75 "updateGlobalView()",
76 "Local number of cells ("
77 << _localNumberOfInnerUnrefinedCells << ") is bigger than global cell count ("
78 << _globalNumberOfInnerUnrefinedCells
79 << "). This usually happens if a forking tree has some pending refinement events and cannot refine anymore, as it has already spawned cells. Statistics might have inconsistent view of world"
80 );
81 _isViewConsistent = false;
82 _globalNumberOfInnerUnrefinedCells = _localNumberOfInnerUnrefinedCells;
83 }
84
85 _globalNumberOfInnerUnrefinedCellsRequest = new MPI_Request();
86 _globalNumberOfSplitsRequest = new MPI_Request();
87 _globalNumberOfTreesRequest = new MPI_Request();
88 _globalNumberOfRanksWithEnabledLoadBalancingRequest = new MPI_Request();
89
90 _globalNumberOfInnerUnrefinedCellsBufferOut = _localNumberOfInnerUnrefinedCells;
91 _globalNumberOfSplitsOut = _localNumberOfSplits;
92 _globalNumberOfTreesOut = _localNumberOfTrees;
93 _globalNumberOfRanksWithEnabledLoadBalancingOut = _isLocalLoadBalancingEnabled ? 1 : 0;
94
95 MPI_Iallreduce(
96 &_globalNumberOfRanksWithEnabledLoadBalancingOut, // send
97 &_globalNumberOfRanksWithEnabledLoadBalancingIn, // receive
98 1, // count
99 MPI_INT,
100 MPI_SUM,
101 tarch::mpi::Rank::getInstance().getCommunicator(),
102 _globalNumberOfRanksWithEnabledLoadBalancingRequest
103 );
104 MPI_Iallreduce(
105 &_globalNumberOfTreesOut, // send
106 &_globalNumberOfTreesIn, // receive
107 1, // count
108 MPI_INT,
109 MPI_SUM,
110 tarch::mpi::Rank::getInstance().getCommunicator(),
111 _globalNumberOfTreesRequest
112 );
113 MPI_Iallreduce(
114 &_globalNumberOfInnerUnrefinedCellsBufferOut, // send
115 &_globalNumberOfInnerUnrefinedCellsBufferIn, // receive
116 1, // count
117 MPI_INT,
118 MPI_SUM,
119 tarch::mpi::Rank::getInstance().getCommunicator(),
120 _globalNumberOfInnerUnrefinedCellsRequest
121 );
122 MPI_Iallreduce(
123 &_globalNumberOfSplitsOut, // send
124 &_globalNumberOfSplitsIn, // receive
125 1, // count
126 MPI_INT,
127 MPI_SUM,
128 tarch::mpi::Rank::getInstance().getCommunicator(),
129 _globalNumberOfSplitsRequest
130 );
131#endif
132 }
133
134 if (_globalNumberOfSplits == 0 and _localNumberOfSplits == 0 and _numberOfStateUpdatesWithoutAnySplit < 65536) {
135 _numberOfStateUpdatesWithoutAnySplit++;
136 } else if (_globalNumberOfSplits > 0 or _localNumberOfSplits > 0) {
137 _numberOfStateUpdatesWithoutAnySplit = 0;
138 }
139
140 _localNumberOfSplits = 0;
141}
