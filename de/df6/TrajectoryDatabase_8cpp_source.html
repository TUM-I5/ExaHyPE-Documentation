211 toolbox::particles::TrajectoryDatabase::AddSnapshotAction result;
212

214 _data.insert( std::pair<std::pair<int,int>, std::list<Entry>>(number,std::list<Entry>()) );

223 _maxPositionDelta = std::max(delta,_maxPositionDelta);

228 ) {

230 result=toolbox::particles::TrajectoryDatabase::AddSnapshotAction::Replace;
231 }
232 else if (
233 tarch::la::equals( _data.at(number).front().timestamp, timestamp )
234 ) {
235 result=toolbox::particles::TrajectoryDatabase::AddSnapshotAction::Ignore;
236 }
237 else if ( _deltasAreRelative and _maxPositionDelta>=_deltaCutOffThreshold and delta/_maxPositionDelta>=_positionDelta ) {
238 result=toolbox::particles::TrajectoryDatabase::AddSnapshotAction::Append;
239 }

241 result=toolbox::particles::TrajectoryDatabase::AddSnapshotAction::Append;
