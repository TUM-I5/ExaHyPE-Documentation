159void petsc::LinearEquationSystem::solve(PreconditionerType pctype, KSPSolverType ksptype){
160 //assemble matrices
161 MatAssemblyBegin(_A,MAT_FINAL_ASSEMBLY);
162 MatAssemblyEnd(_A,MAT_FINAL_ASSEMBLY);
163
164 //assemble vectors
165 VecAssemblyBegin(_x);
166 VecAssemblyEnd(_x);
167 VecAssemblyBegin(_b);
168 VecAssemblyEnd(_b);
169
170 Vec u; // exact soln
171 KSP ksp; /* KSP context */
172 PC pc; /* PC context */
173 PetscReal norm; /* norm of solution error */
174
175 /*
176 Create linear solver context
177 */
178 KSPCreate(PETSC_COMM_WORLD, &ksp);
179
180 /*
181 Set operators. Here the matrix that defines the linear system
182 also serves as the preconditioning matrix.
183 */
184 KSPSetOperators(ksp, _A, _A);
185 KSPGetPC(ksp, &pc);
186
187 //set runtime options
188 KSPSetFromOptions(ksp);
189
190 // convert each of the enums into Petsc macros
191 PCType preconditionerType = getPreconditionerTypeFromEnum(pctype);
192 KSPType kspsolvertype = getKSPSolverTypeFromEnum(ksptype);
193
194 PCSetType(pc, preconditionerType);
195 KSPSetType(ksp, kspsolvertype);
196 KSPSetUp(ksp);
197 KSPSetTolerances(ksp, 1.e-7, PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT);
198
199 //solve against the "exact" solution u
200 KSPSolve(ksp, _b, _x);
201
202 KSPConvergedReason reason;
203 KSPGetConvergedReason(ksp, &reason);
204
205 PetscInt its;
206 KSPGetIterationNumber(ksp, &its);
207 logInfo("petscSolve()", "used preconditioner " << preconditionerType << " and solver " << kspsolvertype);
208 logInfo("petscSolve()", "number of iterations was " << (int)its);
209 logInfo("petscSolve()", "converged for reason " << KSPConvergedReasons[reason]);
210
211 //tidy up temp objects
212 KSPDestroy(&ksp);
213
214}
