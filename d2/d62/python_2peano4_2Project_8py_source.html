107 def generate(self,
108 overwrite=peano4.output.Overwrite.Default,
109 throw_away_data_after_generation=False):
110 """!
111 Generate all code. If you add stuff to your project after a
112 build, you have to (re-)generate the code. If you compile only
113 once, you don't have to invoke this routine explicitly. It is
114 lazily called by the other project routines - the latest before
115 you run the code.
116
117 It is important that I reset the output after each generate
118 call before you change parameter settings and call generate
119 again. To do so, invoke cleanup(). If you forget this, two
120 subsequent generate calls enrich the output twice.
121
122 throw_away_data_after_generation: Bool
123 The Peano 4 memory footprint can become quite substantial
124 effectively reducing the translation capabilities (as compilers
125 tend to require a lot of memory, too). So pass in True if you
126 want the script to throw away the data structures (and run a
127 garbage collection) after all files have been generated. Please
128 note that it invalidates both this object (and maybe another
129 object that you've used to generate the present one - such as
130 ExaHyPE). It really strips everything down to the stuff you
131 absolutely need to translate and run the code.
132 """
133 print( "generate all code ..." )
134 self.is_generated = True
135 self.is_built = False
136 if len(self.output.artefacts)>0:
137 print( "some artefacts have already been added to repository ... assume this is intentional (by higher abstraction layer, e.g.)")
138
139 self.output.readme.add_package_description( self.constants.readme_entry() )
140
141 self.datamodel.construct_output(self.output)
142 self.solversteps.construct_output(self.output)
143
144 self.main.construct_output(self.output,self.project_name + "-main")
145 self.output.generate(overwrite, self.directory, self.subdirectories)
146 if self.subconstants:
147 for i in range(len(self.subconstants)):
148 self.subconstants[i].generate(overwrite, self.directory, self.subdirectories[i])
149 else:
150 self.constants.generate(overwrite, self.directory)
151
152 print("generation complete")
153
154 if throw_away_data_after_generation:
155 self.datamodel = None
156 self.solversteps = None
157 self.output = None
158 gc.collect()
159 print("threw away all data and ran garbage collection")

161 def build(self,
162 make=True,
163 make_clean_first=True,
164 throw_away_data_after_build=False,
165 number_of_parallel_builds=-1):
166 """!
167 Invokes the underlying make build mechanism on the project.
168
169 number_of_parallel_builds: int
170 This is mapped onto make -jnumber_of_parallel_builds, i.e., it
171 determines how many parallel make instances the code spawns.
172 Usually, a lot of the generated code is quite lengthy. Therefore
173 compile time can be high. If you pass something smaller or equal
174 to zero, it will use the core count as guideline how to compile.
175
176 throw_away_data_after_build: Bool
177 see generate()
178 """
179 if number_of_parallel_builds <= 0:
180 number_of_parallel_builds = multiprocessing.cpu_count()
181
182 if not self.is_generated:
183 self.generate()
184
185 if make:
186 if make_clean_first:
187 print("clean up project ... in {}".format(self.directory))
188 try:
189 if make_clean_first:
190 subprocess.check_call(["make", "clean"], cwd=self.directory)
191 self.is_built = False
192 print("clean complete")
193 except Exception as e:
194 print("clean failed (" + str(e) + ") - continue anyway")
195
196 if not self.is_built:
197 print("start to compile with concurrency level of " + str(number_of_parallel_builds) + " ...")
198 try:
199 process = subprocess.run(["make", "-j"+str(number_of_parallel_builds)], cwd=self.directory, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
200 # Extract warnings and errors from error message
201 error_message = process.stderr.decode().strip()
202 warnings = re.findall(r"(?i)warning:.*", error_message)
203 errors = re.findall(r"(?i)error:.*", error_message)
204 # Print warnings into the terminal
205 if warnings:
206 print("\nWarnings ({}):".format(len(warnings)))
207 print("\n".join(warnings))
208 if errors:
209 print("\nErrors ({}):".format(len(errors)))
210 error_message = "\n".join(errors)
211 # Check the return code for linker errors
212 # Raise exception with errors
213 if process.returncode != 0:
214 linker_error_message = process.stderr.decode().strip() + "\n" + error_message
215 raise Exception(linker_error_message)
216 print("compile completed successfully")
217 self.is_built = True
218 self.build_was_successful = True
219 except Exception as e:
220 self.is_built = True
221 self.build_was_successful = False
222 print(str(e))
223 print("compile was not successful")
224 sys.exit(1)
225 else:
226 print("cannot build as code generation has not been successful")
227
228 if throw_away_data_after_build:
229 self.cleanup()
230 self.datamodel = None
231 self.solversteps = None
232 self.output = None
233 gc.collect()
234 print("threw away all data and ran garbage collection")

236 def run(self, executable, args=[], prefix=None, pipefile=None, rebuild_if_required=True):
237 """!
238 Runs the code. args should be a list of strings or the empty list.
239 prefix is an array, too. A typical invocation looks alike
240 project.run( ["16.0", "32.0"], ["/opt/mpi/mpirun", "-n", "1"] )
241 The operation returns True if the run had been successful
242
243 pipefile: string or None
244 """
245 success = False
246 if rebuild_if_required and not self.is_built and not self.build_was_successful:
247 self.build()
248
249 if not rebuild_if_required or (self.is_built and self.build_was_successful):
250 print( "run executable " + str(executable))
251
252 invocation = []
253 if prefix!=None:
254 invocation += prefix
255 invocation += ["./" + executable]
256 invocation += args
257
258 try:
259 result = None
260 if pipefile==None:
261 result = subprocess.run( invocation, stdout=subprocess.PIPE, stderr=subprocess.PIPE )
262 else:
263 result = subprocess.run( invocation, stdout=open( pipefile, "w" ), stderr=subprocess.PIPE )
264 if result.returncode==0:
265 if pipefile==None:
266 print( result.stdout.decode("utf-8") )
267 if result.stderr!=None:
268 print( result.stderr.decode("utf-8") )
269 print( "run completed without error code, but check for warnings and numerical assertions/errors" )
270 else:
271 print( result.stderr.decode("utf-8") )
272 print( "run failed" )
273 success = True
274 except Exception as e:
275 print( "run of application was not successful: " + str(e) )
276 print( "invocation: " + str(invocation) )
277 else:
278 print( "cannot run as code compilation has not been successful" )
279 return success
