26bool mghype::matrixfree::solvers::SolverStatistics::terminationCriterionHolds() {
27 int residualReducedUnderThreshold = 0;
28 int residualGrows = 0;
29
30 // The criterion holds if ALL three norms reach the tolerance
31 // or if the total numberOfResidualGrowths exceeds a limit
32
33 if ( tarch::la::equals(_initialGlobalResidualMaxNorm,0.0) ) {
34 _initialGlobalResidualMaxNorm = _globalResidualMaxNorm;
35 }
36 else if (_previousGlobalResidualMaxNorm * 1.2 < _globalResidualMaxNorm) {
37 logWarning( "terminationCriterionHolds()", "max residual grows from " << _previousGlobalResidualMaxNorm << " to " << _globalResidualMaxNorm );
38 residualGrows++;
39 }
40 else if (_globalResidualMaxNorm / _initialGlobalResidualMaxNorm < _tolerance) {
41 residualReducedUnderThreshold++;
42 }
43
44 _previousGlobalResidualMaxNorm = _globalResidualMaxNorm;
45 _previousGlobalResidualEukledianNormSquared = _globalResidualEukledianNormSquared;
46 _previousGlobalResidualHNormSquared = _globalResidualHNormSquared;
47
48 logDebug( "terminationCriterionHolds()", "residualReducedUnderThreshold=" << residualReducedUnderThreshold );
49 static int numberOfResidualGrowths = 0;
50 const int numberOfChecks = 1; // reset to 3 for all-norms check and uncomment the blocks above...
51 const int residualGrowthsLimit = 50; // stop if the residual norm increases over several consecutive iterations
52
53 if (residualGrows==numberOfChecks) numberOfResidualGrowths++;
54
55 bool output = (residualReducedUnderThreshold==numberOfChecks or numberOfResidualGrowths > residualGrowthsLimit);
56
57 if (residualReducedUnderThreshold==numberOfChecks)
58 logInfo( "terminationCriterionHolds()", "ONLY CHECKING MAX-NORM: terminating because residualReducedUnderThreshold==" << residualReducedUnderThreshold );
59
60 if (numberOfResidualGrowths > residualGrowthsLimit)
61 logInfo( "terminationCriterionHolds()", "terminating because numberOfResidualGrowths==" << numberOfResidualGrowths );
62
63 return output;
64}
