10class Constants(object):
11 """!
12 Representative of generated Constants.h holding all user-defined constants
13
14 Represents all constants that a Project exports from the Python
15 script into C++. I do provide routines to export defines or
16 constants (via constexpr). For the latter, I rely on the auto
17 type word unless you use a specialised routine for a particular
18 type.
19
20 Each project has one instance of Constants, so you can always
21 add/export new constants with
22
23 ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
24 my_project.constants.export ...
25 ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
26
27 Constants also allows you to add defines (precompiler statements).
28 """

52 def export( self, name, value ):
53 """
54 Tell the C++ code underlying the project that a certain variable with a
55 name has a certain value. The passed arguments are mapped onto an
56 constexpr. Therefore, name has to be a string, while value can be an
57 integer, a float or a string as well. If you want to export booleans
58 or just define variants, you have to use the other routines.
59 """
60 new_entry = "constexpr auto " + str(name) + " = " + str(value) + ";"
61 self.d[ "ADD_CONSTANTS" ] += " " + new_entry + "\n"
