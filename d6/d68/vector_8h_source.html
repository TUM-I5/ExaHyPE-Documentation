242/* Takes an integer mask and forms a left-packed integer vector
243 * containing indices of the set bits in the integer mask.
244 * Also returns the total number of bits set in the mask. */
245#define VEC_FORM_PACKED_MASK(mask, packed_mask) \
246 { \
247 unsigned long expanded_mask = _pdep_u64(mask, 0x0101010101010101); \
248 expanded_mask *= 0xFF; \
249 unsigned long wanted_indices = _pext_u64(identity_indices, expanded_mask); \
250 __m128i bytevec = _mm_cvtsi64_si128(wanted_indices); \
251 packed_mask.m = _mm256_cvtepu8_epi32(bytevec); \
252 }

272/* Takes an integer mask and forms a left-packed integer vector
273 * containing indices of the set bits in the integer mask.
274 * Also returns the total number of bits set in the mask. */
275#define VEC_FORM_PACKED_MASK(mask, v_mask, pack) \
276 { \
277 for (int i = 0; i < VEC_SIZE; i++) \
278 if ((mask & (1 << i))) v_mask.i[pack++] = i; \
279 }
280
281/* Takes two integer masks and forms two left-packed integer vectors
282 * containing indices of the set bits in each corresponding integer mask.
283 * Also returns the total number of bits set in the mask. */
284#define VEC_FORM_PACKED_MASK_2(mask, v_mask, pack, mask2, v_mask2, pack2) \
285 { \
286 for (int i = 0; i < VEC_SIZE; i++) { \
287 if ((mask & (1 << i))) v_mask.i[pack++] = i; \
288 if ((mask2 & (1 << i))) v_mask2.i[pack2++] = i; \
289 } \
290 }
291#endif

293/* Performs a left-pack on a vector based upon a mask and returns the result. */
294/* This uses AVX intrinsics, but this is slower than performing the left-pack
295 * manually by looping over the vectors. */
296#ifndef VEC_LEFT_PACK
297#define VEC_LEFT_PACK(a, mask, result) \
298 { \
299 __m256 t1 = _mm256_castps128_ps256(_mm256_extractf128_ps(a, 1)); \
300 __m256 t2 = _mm256_insertf128_ps(t1, _mm256_castps256_ps128(a), 1); \
301 __m256 r0 = _mm256_permutevar_ps(a, mask); \
302 __m256 r1 = _mm256_permutevar_ps(t2, mask); \
303 __m128i k1 = _mm_slli_epi32( \
304 (__m128i)(_mm_xor_si128((__m128i)VEC_GET_HIGH((__m256)mask), \
305 (__m128i)_mm_set1_epi32(4))), \
306 29); \
307 __m128i k0 = _mm_slli_epi32((__m128i)(VEC_GET_LOW((__m256)mask)), 29); \
308 __m256 kk = \
309 _mm256_insertf128_ps(_mm256_castps128_ps256(_mm_castsi128_ps(k0)), \
310 _mm_castsi128_ps(k1), 1); \
311 *((__m256 *)(result)) = _mm256_blendv_ps(r0, r1, kk); \
312 }
313#endif /* HAVE_AVX2 */
