281 def get_min_number_of_spacetree_levels(self, domain_size):
282 coarsest_tree_level = 0
283 while (
284 domain_size * 3 ** (-coarsest_tree_level) / self._patch_size
285 > self._max_meshcell_h

287 coarsest_tree_level += 1
288 return coarsest_tree_level

292 while (
293 domain_size * 3 ** (-finest_tree_level) / self._patch_size
294 > self._min_meshcell_h
295 ):
296 finest_tree_level += 1

440 def create_data_structures(self):
441 """
442
443 Recall in subclasses if you wanna change the number of unknowns
444 or auxiliary variables. See class description's subsection on
445 data flow.
446
447 :: Call order and ownership
448
449 This operation can be called multiple times. However, only the very
450 last call matters. All previous calls are wiped out.
451
452 If you have a hierarchy of solvers, every create_data_structure()
453 should first(!) call its parent version. This way, you always ensure
454 that all data are in place before you continue to alter the more
455 specialised versions. So it is (logically) a top-down (general to
456 specialised) run through all create_data_structure() variants
457 within the inheritance tree.
458
459
460 :: Arguments
461
462 _patch: Patch (NxNxN)
463 Actual patch data. We use Finite Volumes, so this is
464 always the current snapshot, i.e. the valid data at one point.
465
466 _patch_overlap_old, _patch_overlap_new: Patch (2xNxN)
467 This is a copy/excerpt from the two adjacent finite volume
468 snapshots plus the old data as backup. If I want to implement
469 local timestepping, I don't have to backup the whole patch
470 (see _patch), but I need a backup of the face data to be able
471 to interpolate in time.
472
473 _patch_overlap_update: Patch (2xNxN)
474 This is hte new update. After the time step, I roll this
475 information over into _patch_overlap_new, while I backup the
476 previous _patch_overlap_new into _patch_overlap_old. If I
477 worked with regular meshes only, I would not need this update
478 field and could work directly with _patch_overlap_new. However,
479 AMR requires me to accumulate data within new while I need

481 accumulation/roll-over data which usually is not stored

483

681 @abstractmethod

683 """!
684
685 Create required action sets
686
687 Overwrite in subclasses if you wanna create different
688 action sets.
689
690 ## Call order and ownership
691
692 This operation can be called multiple times. However, only the very
693 last call matters. All previous calls are wiped out.
694
695 If you have a hierarchy of solvers, every create_data_structure()
696 should first(!) call its parent version. This way, you always ensure
697 that all data are in place before you continue to alter the more
698 specialised versions. So it is (logically) a top-down (general to
699 specialised) run through all create_data_structure() variants
700 within the inheritance tree.
701
702 :: Recreation vs backup (state discussion)
703
704 We faced some issues with action sets that should not be
705 overwritten. For example, the postprocessing should not be overwritten
706 as users might want to set it and then later on reset the number of
707 unknowns, e.g. In this case, you would loose your postprocessing if
708 create_action_sets() recreated them. So I decided to make an exception
709 here: the postprocessing step is never overwritten by the standard
710 classes.
711
712 There are further action sets which have a state, which users might
713 want to alter. The most prominent one is the AMR routines, where users
714 often alter the interpolation and restriction scheme. Here, things are
715 tricky: If we keep the default one, the action set would not pick up
716 if you changed the number of unknowns, e.g. However, if we recreated
717 the action set, we'd miss out on any changed interpolation/restriction
718 scheme. Therefore, I have to hold the interpolation and restriction
719 scheme seperatae.

722

724 predicate, i.e. _store_cell_data_default_guard() and

727 study carefully if a predicate accepts a unique guard or a set of
728 guards.

730

732

1018 def add_actions_to_init_grid(self, step, restart_from_checkpoint=False):
1019 """!
1020
1021 Add your actions to init grid
1022
1023 The AMR stuff has to be the very first thing. Actually, the AMR routines'
1024 interpolation doesn't play any role here. But the restriction indeed is
1025 very important, as we have to get the face data for BCs et al. The action
1026 set order is inverted while we ascend within the tree again. Therefore, we
1027 add the AMR action set first which means it will be called last when we go
1028 from fine to coarse levels within the tree.
1029
1030 The projection onto the faces is a postprocessing step. This is different
1031 to DG, where we need face data for the current time step's solution. Here,
1032 we ensure that all halos are valid for the subsequent time step again.
1033
1034 ## Ordering
1035
1036 The order of the action sets is preserved throughout the steps down within
1037 the tree hierarchy. It is inverted throughout the backrolling.
1038
1039 This is what we want to achieve:
1040
1041 - Project solution onto faces. This happens in touchCellLastTime(). See
1042 exahype2.solvers.rkfd.actionsets.ProjectPatchOntoFaces for comments.
1043 The project will end up in QUpdate.
1044 - Roll updates over on the faces from QUpdate into Q_new. This is done
1045 by RollOverUpdateFace, which requires in turn that the getUpdated()
1046 flag is set. As the roll-over plugs into touchFaceLastTime(), it will
1047 always be called after the projection, since the latter is a cell
1048 operation.
1049 - Copy new face data Q_new into old face data Q_old, as this is the
1050 initial sweep, i.e. the old face data otherwise might hold rubbish.
1051 This is a backup operation realised through the action set
1052 BackupPatchOverlap. This one plugs into touchFaceLastTime() too.
1053 Therefore, it is important that its priority is smaller than the one
1054 of the roll-over, since we invert the call order for the touch-last
1055 events.
1056 - Restrict the data to the coarser level if we are on a hanging face.
1057
1058
1059 """
1060 assert (
1061 self._action_set_copy_new_faces_onto_old_faces.descend_invocation_order
1062 < self._action_set_roll_over_update_of_faces.descend_invocation_order
1063 )
1064
1065 if restart_from_checkpoint:
1066 self._action_set_initial_conditions = (
1067 exahype2.solvers.rkfd.actionsets.InitialCondition(
1068 self, self._store_cell_data_default_guard(), "true", restart_from_checkpoint=True
1069 )
1070 )
1071
1072 step.add_action_set(self._action_set_preprocess_solution)
1073 step.add_action_set(
1074 self._action_set_couple_resolution_transitions_and_handle_dynamic_mesh_refinement
1075 )
1076 step.add_action_set(self._action_set_copy_new_faces_onto_old_faces)
1077 step.add_action_set(self._action_set_roll_over_update_of_faces)
1078 step.add_action_set(self._action_set_initial_conditions)
1079 step.add_action_set(self._action_set_project_patch_onto_faces)
1080 step.add_action_set(self._action_set_update_face_label)
1081 step.add_action_set(self._action_set_update_cell_label)
1082 step.add_action_set(self._action_set_AMR)
1083 step.add_action_set(self._action_set_postprocess_solution)
