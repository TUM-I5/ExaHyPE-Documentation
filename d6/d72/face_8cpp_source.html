152void Face::reverseDimension(Coordinate coord){
153 Coordinate coord_other = coord == coords[0] ? coords[1] : coords[0];
154 std::vector<int> range; getIndexRange(coord,range);
155 std::vector<int> range_other; getIndexRange(coord_other,range);
156 for (int c_indx = 0 ; c_indx < 3 ; c_indx++){
157 for (int j : range_other){
158 std::vector<double> temp; temp.resize(range.size());
159 for (int i : range){
160 int index_temp = range.size()-(i+1-range[0]);
161 temp[index_temp] = (*this)(static_cast<Coordinate>(c_indx),coord_other,j,coord,i);
162 }
163 for (int i : range){
164 (*this)(static_cast<Coordinate>(c_indx),coord_other,j,coord,i) = temp[i - range[0]];
165 }
166 }

277Edge Face::split(Face& other, Face& left, Face& right) const{
278 /* We assume this face lies in the X Y plane, the other face in the Y Z plane*/
279 /* The index cut happens in the Y coordinate */
280
281 //Find shared coordiante:
282 Coordinate x,y,z;
283 if(coords[0] == other.coords[0] || coords[0] == other.coords[1] ){
284 y=coords[0];
285 x=coords[1];
286 }else if(coords[1] == other.coords[0] || coords[1] == other.coords[1] ){
287 y=coords[1];
288 x=coords[0];
289 }else{
290 std::cout << "Couldn't find shared coordinate" << std::endl;
291 }
292
293 Edge intersection_path(y);
294
295 z = getNormal(x,y);
296
297 /*** Cut Indeces ***/
298 int index_intersection_x = other.index_offset[x];
299 int index_intersection_z = index_offset[z];
300
301 if( index_intersection_x < index_offset[x] ||
302 index_intersection_x >= index_offset[x] + index_size[x]){
303 std::stringstream ss;
304 ss << "Intersection index in coord "<< x << " at "<<index_intersection_x
305 <<" lies outside of face "<< index_offset[x] << " "
306 << index_offset[x] + index_size[x] << std::endl;
307 throw std::domain_error(ss.str());
308 }
309
310 if( index_intersection_z < other.index_offset[z] ||
311 index_intersection_z >= other.index_offset[z] + other.index_size[z]){
312 std::stringstream ss;
313 ss << "Intersection index in coord "<< z << " at "<<index_intersection_z
314 <<" lies outside of face "<< index_offset[z] << " "
315 << index_offset[z] + index_size[z] << std::endl;
316 throw std::domain_error(ss.str());
317 }
318 /******************/
319
320
322 // We have to find intersecting planes on each column in y
323 // The results is a intersection path
324 std::vector<int> range_x; getIndexRange(x,range_x); range_x.pop_back();
325 std::vector<int> range_y; getIndexRange(y,range_y);
326 std::vector<int> range_z; other.getIndexRange(z,range_z); range_z.pop_back();
327
328 intersection_path.resize(NCOORDS * range_y.size());
329 std::vector<int> intersection_index_path;
330 intersection_index_path.resize(NCOORDS * range_y.size());
331
332 idx2 idx_path(NCOORDS,range_y.size());
333 for(int ind_y : range_y){
334 bool found = false;
335 double x_int,z_int;
336 int x_int_index,z_int_index;
337 for(int ind_x : range_x){
338 for(int ind_z : range_z){
339 double f0 = other (x, y, ind_y, z, ind_z);
340 double f1 = other (x, y, ind_y, z, ind_z + 1);
341 double y0 = other (z, y, ind_y, z ,ind_z);
342 double y1 = other (z, y, ind_y, z ,ind_z + 1);
343
344 double t0 = (*this)(z, y, ind_y, x, ind_x);
345 double t1 = (*this)(z, y, ind_y, x, ind_x+1);
346 double x0 = (*this)(x, y, ind_y, x, ind_x);
347 double x1 = (*this)(x, y, ind_y, x, ind_x+1);
348
349 if(intersect(x0,t0,
350 f0,y0,
351 x1-x0,t1-t0,
352 f1-f0,y1-y0,
353 x_int,z_int)){
354 found = true;
355 }
356
357 if(found){
358 x_int_index = ind_x;
359 z_int_index = ind_z;
360 break;
361 }
362
363 }
364 if(found) break;
365 }
366 if(found){
367 intersection_path(x,ind_y-range_y[0]) = x_int;
368 intersection_path(y,ind_y-range_y[0]) = (*this)(y,y,ind_y,x,x_int_index);
369 intersection_path(z,ind_y-range_y[0]) = z_int;
370
371 intersection_index_path[idx_path(x,ind_y-range_y[0])] = x_int_index;
372 intersection_index_path[idx_path(y,ind_y-range_y[0])] = ind_y;
373 intersection_index_path[idx_path(z,ind_y-range_y[0])] = z_int_index;
374 }else{
375 std::stringstream ss;
376 ss << "Couldn't find intersection at "<< coordToString(y) <<" = " << ind_y << std::endl;
377 ss << coordToString(z) <<" " << coordToString(y) <<" " << coordToString(x) <<std::endl;
378 ss << "This face: " << std::endl;
379 ss << (*this) << std::endl;
380 ss << "Other face: " << std::endl;
381 ss << other << std::endl;
382 throw std::out_of_range(ss.str());
383 }
384 }
385
386 FaceBoundary thisBoundary ((*this));
387 FaceBoundary otherBoundary(other);
388
389 //copy the boundary of this face
390 FaceBoundary left_boundary = thisBoundary;
391 FaceBoundary right_boundary = thisBoundary;
392
393 thisBoundary.split(x,
394 index_intersection_x-index_offset[x],
395 intersection_path,
396 left_boundary,right_boundary);
397
398 int index_offset_left[3];
399 std::copy_n(index_offset,3,index_offset_left);
400 int index_offset_right[3];
401 std::copy_n(index_offset,3,index_offset_right);
402 index_offset_right[x] = index_intersection_x;
403
404 left = Face(index_offset_left ,left_boundary , perturbation);
405 right = Face(index_offset_right,right_boundary, perturbation);
406
407 return intersection_path;
408}

411bool Face::intersect(double a_0, double a_1,
412 double b_0, double b_1,
413 double c_0, double c_1,
414 double d_0, double d_1,
415 double& x, double& y) const{
416 double mu = (a_1 - b_1 + c_1/c_0 * (b_0 -a_0)) / (d_1 - d_0 * c_1/c_0);
417 double lambda = (b_0 - a_0 + mu * d_0) / c_0;
418
419 if(!std::isfinite(mu) || mu < 0 || mu > 1 || lambda < 0 || lambda > 1 || !std::isfinite(lambda)){
420 return false;
421 }else{
422
423 x= a_0 + lambda * c_0;
424 y= a_1 + lambda * c_1;
425
426 if(std::isnan(x) || std::isnan(y)){
427 std::stringstream ss;
428 ss << "Intersection resulted in nan";
429 ss << lambda << " " << mu << std::endl;
430 ss << x << " " << y << std::endl;
431 ss << "for" << std::endl;
432 ss << a_0 << " " << a_1<< std::endl;
433 ss << b_0 << " " << b_1<< std::endl;
434 ss << c_0 << " " << c_1<< std::endl;
435 ss << d_0 << " " << d_1<< std::endl;
436 throw std::range_error(ss.str());
437 }
438
439 return true;
440 }
