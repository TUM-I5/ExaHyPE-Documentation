19void* tarch::accelerator::MemoryManager::allocate(
20 std::size_t sizeInBytes,
21 MemoryLocation location,
22 int device
23) {
24 tarch::multicore::Lock lock(_semaphore);
25
26 // Is device initialised?
27 if (_memoryPages.count(device) == 0) {
28 _memoryPages.insert(std::make_pair(device, MemoryPages()));
29 }
30
31 // Is a free page left? If not, insert one.
32 if (_memoryPages.at(device).freePages[sizeInBytes].empty()) {
33 // TODO: We should also check for the memory location here.
34 // Not only the size but also the memory location should
35 // be used for a lookup.
36 _memoryPages.at(device).freePages[sizeInBytes].push_back(std::make_pair(
37 accelerator::allocateRawData(sizeInBytes, location, device),
38 location
39 ));
40 }
41
42 // Grab fitting page
43 void* resultPtr = _memoryPages.at(device).freePages.at(sizeInBytes).back().first;
44 const MemoryLocation resultLocation = _memoryPages.at(device).freePages.at(sizeInBytes).back().second;
45 assertion(resultPtr);
46 _memoryPages.at(device).freePages.at(sizeInBytes).pop_back();
47
48 // Memorise page meta data, i.e. size and location
49 assertion(_memoryPages.at(device).usedPages.count(resultPtr) == 0);
50 _memoryPages.at(device).usedPages.insert(
51 std::make_pair(resultPtr, std::make_pair(sizeInBytes, resultLocation))
52 );
53
54 // Return result
55 return resultPtr;
56}

58void tarch::accelerator::MemoryManager::free(void* pointer, int device) {
59 tarch::multicore::Lock lock(_semaphore);
60 assertion(pointer);
61
62 assertion(_memoryPages.count(device) == 1);
63 assertion(_memoryPages.at(device).usedPages.count(pointer) == 1);
64
65 const std::size_t size = _memoryPages.at(device).usedPages.at(pointer).first;
66 const MemoryLocation location = _memoryPages.at(device)
67 .usedPages.at(pointer)
68 .second;
69 _memoryPages.at(device).usedPages.erase(pointer);
70
71 assertion(_memoryPages.at(device).freePages.count(size) == 1);
72 _memoryPages.at(device).freePages.at(size).push_back(
73 std::make_pair(pointer, location)
74 );
75}

81void tarch::accelerator::MemoryManager::clearWithoutLock() {
82 for (auto& memoryPage : _memoryPages) {
83 int device = memoryPage.first;
84 logDebug("clear()", "Free data of device " << device);
85
86 if (not memoryPage.second.usedPages.empty()) {
87 logWarning(
88 "clear()",
89 "Used pages on device "
90 << device << " are not empty. " << memoryPage.second.usedPages.size()
91 << " memory blocks still in use. Likely a memory leak (missing free())."
92 );
93 }
94
95 for (auto& p : memoryPage.second.freePages)
96 for (auto& pp : p.second) {
97 accelerator::freeRawData(pp.first, pp.second, device);
98 }
99 }
100
101 _memoryPages.clear();
102}
