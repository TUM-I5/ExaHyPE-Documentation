36/*
37 The blocked_rangeNd_impl uses make_index_sequence<N> to automatically generate a ctor with
38 exactly N arguments of the type tbb::blocked_range<Value>. Such ctor provides an opportunity
39 to use braced-init-list parameters to initialize each dimension.
40 Use of parameters, whose representation is a braced-init-list, but they're not
41 std::initializer_list or a reference to one, produces a non-deduced context
42 within template argument deduction.
43
44 NOTE: blocked_rangeNd must be exactly a templated alias to the blocked_rangeNd_impl
45 (and not e.g. a derived class), otherwise it would need to declare its own ctor
46 facing the same problem that the impl class solves.
47*/
48
49template<typename Value, unsigned int N, typename = detail::make_index_sequence<N>>
50 __TBB_requires(blocked_range_value<Value>)
51class blocked_rangeNd_impl;

162 template<typename split_type>
163 void do_split(blocked_rangeNd_impl& r, split_type proportion) {
164 static_assert((std::is_same<split_type, split>::value || std::is_same<split_type, proportional_split>::value), "type of split object is incorrect");
165 __TBB_ASSERT(r.is_divisible(), "can't split not divisible range");
166
167 auto my_it = std::max_element(my_dims.begin(), my_dims.end(), [](const tbb::blocked_range<value_type>& first, const tbb::blocked_range<value_type>& second) {
168 return (first.size() * second.grainsize() < second.size() * first.grainsize());
169 });
170
171 auto r_it = r.my_dims.begin() + (my_it - my_dims.begin());
172
173 my_it->my_begin = tbb::blocked_range<value_type>::do_split(*r_it, proportion);
174
175 // (!(my_it->my_begin < r_it->my_end) && !(r_it->my_end < my_it->my_begin)) equals to
176 // (my_it->my_begin == r_it->my_end), but we can't use operator== due to Value concept
177 __TBB_ASSERT(!(my_it->my_begin < r_it->my_end) && !(r_it->my_end < my_it->my_begin),
178 "blocked_range has been split incorrectly");
179 }
