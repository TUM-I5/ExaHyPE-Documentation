29class _forbidInvalidKernelsDict(dict):
30 """!
31 Modify the default dict to raise an error if a user tries to
32 access kernels which aren't defined. Undefined kernels are
33 specified via blocked_keys kwarg during initialization.
34
35 Also limit the available kernels to those specified in the
36 sph_kernel_list list.
37 """
38
39 def __init__(self, *args, blocked_keys=(), ndim=0, **kwargs):
40 super().__init__(*args, **kwargs)
41 for key in blocked_keys:
42 if key not in sph_kernel_list and key not in sph_vectorized_kernel_list:
43 raise KeyError(
44 f"Invalid kernel key {key}. Only {sph_kernel_list} are permitted"
45 )
46 self.blocked_keys = blocked_keys
47 self.ndim = ndim
48
49 def __setitem__(self, key, value):
50 if key not in sph_kernel_list and key not in sph_vectorized_kernel_list:
51 raise KeyError(
52 f"Invalid kernel key {key}. Only {sph_kernel_list} are permitted"
53 )
54 super().__setitem__(key, value)
55
56 def __getitem__(self, key):
57 if key in self.blocked_keys:
58 raise ValueError(f"SPH kernel {key} is not defined for ndim={self.ndim}")
59 return super().__getitem__(key)

169def _sph_kernel_eval_cubic_spline(q: float):
170 """!
171 Evaluates the cubic spline kernel
172 """
173 if q < 0.5:
174 res = 3.0 * q**2.0 * (q - 1.0) + 0.5
175 elif q < 1.0:
176 # res = q * (q * (3 - q) - 3) + 1
177 res = -(q**3.0) + 3.0 * q**2.0 - 3.0 * q + 1.0
178 else:
179 res = 0.0
180 return res

183def _sph_kernel_eval_quartic_spline(q: float):
184 """!
185 Evaluates the quartic spline kernel
186 """
187 if q < 0.2:
188 res = 6 * q**4 - 2.4 * q**2 + 46 / 125
189 elif q < 0.6:
190 res = -4 * q**4 + 8 * q**3 - 4.8 * q**2 + 8 / 25 * q + 44.0 / 125
191 elif q < 1:
192 res = q**4 - 4 * q**3 + 6 * q**2 - 4 * q + 1
193 else:
194 res = 0.0
195 return res

198def _sph_kernel_eval_quartic_spline_vectorized(q: np.ndarray):
199 """!
200 Evaluates the quartic spline kernel
201 """
202
203 res = np.zeros(q.shape)
204 mask1 = q <= 0.2
205 mask2 = np.logical_and(q > 0.2, q <= 0.6)
206 mask3 = np.logical_and(q > 0.6, q <= 1.0)
207
208 qa = q[mask1]
209 qa2 = qa**2
210 res[mask1] = 6.0 * qa2**2 - 2.4 * qa2 + 46.0 / 125.0
211
212 qb = q[mask2]
213 qb2 = qb**2
214 res[mask2] = (
215 -4.0 * qb2**2 + 8.0 * qb2 * qb - 4.8 * qb2 + 8.0 / 25.0 * qb + 44.0 / 125.0
216 )
217
218 qc = q[mask3]
219 qc2 = qc**2
220 res[mask3] = qc2**2 - 4.0 * qc2 * qc + 6.0 * qc2 - 4.0 * qc + 1.0
221
222 return res

225def _sph_kernel_eval_quintic_spline(q: float):
226 """!
227 Evaluates the quintic spline kernel
228 """
229 if q < 1.0 / 3.0:
230 res = 10.0 * (q**4 * (1.0 - q) - 0.2222222222 * q**2) + 0.2716049382716049
231 elif q < 0.666666666666:
232 qsq = q**2
233 q4 = qsq**2
234 res = (
235 5
236 * (
237 q4 * (q - 3)
238 + qsq * (3.333333333333333 * q - 1.5555555555555)
239 + 0.18518518518518517 * q
240 )
241 + 0.20987654320987653
242 )
243 elif q < 1.0:
244 qsq = q**2
245 res = qsq * (qsq * (5 - q) + 10 * (1 - q)) - 5 * q + 1
246 else:
247 res = 0.0
248 return res

487def sph_kernel_W(
488 r: Union[float, np.ndarray], h: Union[float, np.ndarray], kernel: str, ndim: int
489):
490 """!
491 Evaluates various kernels.
492 The kernels are scaled such that W(q > 1) = 0.
493 Currently implemented:
494 cubic_spline,
495 quartic_spline,
496 quintic_spline,
497 wendland_C2,
498 wendland_C4, [not with ndim=1]
499 wendland_C6, [not with ndim=1]
500
501
502 Parameters
503 ----------
504
505 r: float or ndarray
506 particle distance(s)
507
508 h: float or ndarray
509 smoothing length(s)
510
511 kernel: str
512 which kernel to use
513
514 ndim: int
515 what dimension to use. [1-3]
516
517
518 Returns
519 -------
520
521 W: float
522 evaluated kernel
523
524 """
525
526 sigma = sph_kernel_normalization[ndim - 1][kernel]
527 H = sph_kernel_get_H(h, kernel, ndim)
528 kernel_W = sph_kernel_eval_function[kernel]
529 q = r / H
530 W = sigma * kernel_W(q) / H**ndim
531
532 return W

535def sph_kernel_dWdr(
536 r: Union[float, np.ndarray], h: Union[float, np.ndarray], kernel: str, ndim: int
537):
538 """
539 Evaluates kernel derivatives for various kernels:
540 returns dW/dr = dW/dq dq/dr = 1/H * dW/dq
541
542 The kernels are scaled such that W(q > 1) = 0.
543 Currently implemented:
544 cubic_spline,
545 quartic_spline,
546 quintic_spline,
547 wendland_C2,
548 wendland_C4, [not with ndim=1]
549 wendland_C6, [not with ndim=1]
550
551
552 Parameters
553 ----------
554
555 r: float or ndarray
556 particle distance(s)
557
558 h: float or ndarray
559 smoothing length(s)
560
561 kernel: str
562 which kernel to use
563
564 ndim: int
565 what dimension to use. [1-3]
566
567 Returns
568 -------
569
570 dWdr: float or ndarray
571 evaluated kernel derivative
572
573 """
574
575 sigma = sph_kernel_normalization[ndim - 1][kernel]
576 kernel_dWdr = sph_kernel_deval_function[kernel]
577 H = sph_kernel_get_H(h, kernel, ndim)
578 q = r / H
579 dWdr = sigma * kernel_dWdr(q) / H ** (ndim + 1)
580
581 return dWdr

589 """!

592

650 """!
651 Evaluate kernel root, i.e. W(0, h).
652 We don't actually need `h` here, because the quantity used in the kernels,

656 ----------
657
658 kernel: str
659 which kernel to use
660
661 ndim: int

664

667
668 W: float
669 evaluated kernel
670
671 """
