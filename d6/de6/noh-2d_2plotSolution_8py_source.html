67# Bin the data
68r_bin_edge = np.arange(0.0, 1.0, 0.02)
69r_bin = 0.5 * (r_bin_edge[1:] + r_bin_edge[:-1])
70rho_bin, _, _ = stats.binned_statistic(r, rho, statistic="mean", bins=r_bin_edge)
71v_bin, _, _ = stats.binned_statistic(r, v, statistic="mean", bins=r_bin_edge)
72P_bin, _, _ = stats.binned_statistic(r, P, statistic="mean", bins=r_bin_edge)
73# S_bin, _, _ = stats.binned_statistic(r, S, statistic="mean", bins=r_bin_edge)
74u_bin, _, _ = stats.binned_statistic(r, u, statistic="mean", bins=r_bin_edge)
75rho2_bin, _, _ = stats.binned_statistic(r, rho**2, statistic="mean", bins=r_bin_edge)
76v2_bin, _, _ = stats.binned_statistic(r, v**2, statistic="mean", bins=r_bin_edge)
77P2_bin, _, _ = stats.binned_statistic(r, P**2, statistic="mean", bins=r_bin_edge)
78# S2_bin, _, _ = stats.binned_statistic(r, S ** 2, statistic="mean", bins=r_bin_edge)
79u2_bin, _, _ = stats.binned_statistic(r, u**2, statistic="mean", bins=r_bin_edge)
80rho_sigma_bin = np.sqrt(rho2_bin - rho_bin**2)
81v_sigma_bin = np.sqrt(v2_bin - v_bin**2)
82P_sigma_bin = np.sqrt(P2_bin - P_bin**2)
83# S_sigma_bin = np.sqrt(S2_bin - S_bin ** 2)
84u_sigma_bin = np.sqrt(u2_bin - u_bin**2)
85
86# Analytic solution
87N = 1000 # Number of points
88
89x_s = np.arange(0, 2.0, 2.0 / N) - 1.0
90rho_s = np.ones(N) * rho0
91P_s = np.ones(N) * rho0
92v_s = np.ones(N) * v0
93
94# Shock position
95u0 = rho0 * P0 * (gas_gamma - 1)
96us = 0.5 * (gas_gamma - 1) * v0
97rs = us * time
98
99# Post-shock values
100rho_s[np.abs(x_s) < rs] = rho0 * ((gas_gamma + 1) / (gas_gamma - 1)) ** 2
101P_s[np.abs(x_s) < rs] = 0.5 * rho0 * v0**2 * (gas_gamma + 1) ** 2 / (gas_gamma - 1)
102v_s[np.abs(x_s) < rs] = 0.0
103
104# Pre-shock values
105rho_s[np.abs(x_s) >= rs] = rho0 * (1 + v0 * time / np.abs(x_s[np.abs(x_s) >= rs]))
106P_s[np.abs(x_s) >= rs] = 0
107v_s[x_s >= rs] = -v0
108v_s[x_s <= -rs] = v0
109
110# Additional arrays
111u_s = P_s / (rho_s * (gas_gamma - 1.0)) # internal energy
112s_s = P_s / rho_s**gas_gamma # entropic function
