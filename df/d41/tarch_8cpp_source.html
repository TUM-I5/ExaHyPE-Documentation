37int tarch::convertMemorySize(long long value, MemoryUsageFormat format) {
38 constexpr long long MegaByte = 1024 * 1024;
39 constexpr long long GigaByte = MegaByte * 1024;
40 int result = 0;
41 switch (format) {
42 case tarch::MemoryUsageFormat::MByte:
43 {
44 long long usageMB ((value + (MegaByte / 2)) / MegaByte);
45 if (usageMB > std::numeric_limits<int>::max()) {
46 logError("convertMemorySize()", "cannot cast result to return value: " << usageMB);
47 }
48 result = static_cast<int>(usageMB);
49 break;
50 }
51 case tarch::MemoryUsageFormat::GByte:
52 {
53 long long usageGB ((value + (GigaByte / 2)) / GigaByte);
54 if (usageGB > std::numeric_limits<int>::max()) {
55 logError("convertMemorySize()", "cannot cast result to return value: " << usageGB);
56 }
57 result = static_cast<int>(usageGB);
58 break;
59 }
60 }
61 return result;
62}

82int tarch::getTotalMemory(MemoryUsageFormat format) {
83 int mib[] = { CTL_HW, HW_MEMSIZE };
84 int64_t physical_memory;
85 size_t length;
86
87 // Get the physical memory size
88 length = sizeof(int64_t);
89 sysctl(mib, 2, &physical_memory, &length, NULL, 0);
90 return convertMemorySize(physical_memory, format);
91}

93int tarch::getFreeMemory(MemoryUsageFormat format) {
94 struct rusage usage{};
95 if (0 == getrusage(RUSAGE_SELF, &usage)) {
96 auto freeMem = tarch::getTotalMemory(tarch::MemoryUsageFormat::MByte) - usage.ru_maxrss;
97 return convertMemorySize(freeMem, format);
98 } else {
99 return 0;
100 }
