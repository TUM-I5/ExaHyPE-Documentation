87 def _assemble_A_CC(self, fs_pair):
88 """Assemble weak form for A_{CC}^{(k,ell)}
89 :arg fs_pair: pair of cell function space indices that couple
90 k = fs_pair[0]: first function space (to-space): a cell function space
91 ell = fs_pair[1]: second function space (from-space): a cell function space
92 """
93 #assert fs_pair in self._coupling_CC
94
95 k = fs_pair[0]
96 ell = fs_pair[1]
97 ndof_k = self._fe_C[k].ndof
98 ndof_ell = self._fe_C[ell].ndof
99 A_k_ell = np.empty((ndof_k, ndof_ell))
100 gamma = (0, 0) # no sacling, as the contribution h^2 from dxdy is compensated by 1/h^2 from derivatives \nambla u \cdot \nabla v
101
102 # Quadrature rule: based on the max degree of basis in the function spaces pair
103 max_degree = max(self._fe_C[k].degree, self._fe_C[ell].degree)
104 quad2d = Quadrature2d(GaussianQuadrature(max_degree + 1))
105
106 if fs_pair == (0, 0):
107 #func_k = self._fe_C[k].evaluate
108 #func_ell = self._fe_C[ell].evaluate
109 factor = 1.0
110
111 # Assemble local matrix A^(k,ell)
112 for i in range(ndof_k):
113 for j in range(ndof_ell):
114 A_k_ell[i, j] = 0.0
115 for q, w in zip(quad2d.points, quad2d.weights):
116 A_k_ell[i, j] += factor * w * ( self._fe_C[k].evaluate_derivative(i, 0, q) * self._fe_C[ell].evaluate_derivative(j, 0, q)
117 +
118 self._fe_C[k].evaluate_derivative(i, 1, q) * self._fe_C[ell].evaluate_derivative(j, 1, q)
119 )
120 else:
121 A_k_ell = np.zeros((k, ell))
122
123 return {gamma: A_k_ell}

128 def _assemble_A_CF(self, fs_pair, relative_position):
129 """Assemble weak form for A_{CF}^{(k,ell)}
130 :arg fs_pair: pair of function space indices that couple
131 k = fs_pair[0]: first function space (to-space): a cell function space
132 ell = fs_pair[1]: second function space (from-space): a facet function space
133 :arg relative_position: relative position of the facet within a cell given by
134 the "type of entity"
135 """
136 k = fs_pair[0]
137 ell = fs_pair[1]
138 ndof_k = self._fe_C[k].ndof
139 ndof_ell = self._fe_F[ell].ndof
140 A_k_ell = np.empty((ndof_k, ndof_ell))
141 gamma = (0, 0)
142
143 # Quadrature rule: based on the max degree of basis in the function spaces pair
144 max_degree = max(self._fe_C[k].degree, self._fe_F[ell].degree)
145 quad = GaussianQuadrature(max_degree + 1)
146
147 # Matrix restr applied to [1, q] gives [-1, q], [1, q], [q, -1] or [q, 1] --
148 # 2D coordinates on a facet depending on its relative location
149 # Is's used to later restrict a cell-function to a facet for integrating over facet
150 if relative_position == "left":
151 restr = np.array([[-1.0, 0.0],
152 [ 0.0, 1.0]])
153 # nx, ny = -1.0, 0.0 # outward unit normal components
154 elif relative_position == "right":
155 restr = np.array([[1.0, 0.0],
156 [0.0, 1.0]])
157 # nx, ny = +1.0, 0.0
158 elif relative_position == "bottom":
159 restr = np.array([[ 0.0, 1.0],
160 [-1.0, 0.0]])
161 # nx, ny = 0.0, -1.0
162 elif relative_position == "top":
163 restr = np.array([[0.0, 1.0],
164 [1.0, 0.0]])
165 # nx, ny = 0.0, +1.0
166 else:
167 raise ValueError("Invalid value of relative_position")
168
169 # The following sets factors and integrands for all the pairs (k,ell);
170 if fs_pair == (0, 2): # -int_F q^f v^p dF
171 factor = -0.5 # minus befor the boundary int. in the weak formulation
172
173 if relative_position in ["left", "right"]:
174 gamma = (0, 1) # scale by hy
175 else: # ["top", "bottom"]
176 gamma = (1, 0) # scale by hx
177
178 # Define the integrand function for different
179 def func_k(i, q):
180 facet_coord = np.dot(restr, np.array([1.0, q])) # gives [-1, q], [1, q], [q, -1] or [q, 1]
181 return self._fe_C[k].evaluate(i, facet_coord) # restriction of the cell-function to a facet
182 func_ell = self._fe_F[ell].evaluate
183
184 # Assemble local matrix A^(k,ell)
185 for i in range(ndof_k):
186 for j in range(ndof_ell):
187 A_k_ell[i, j] = 0.0
188 for q, w in zip(quad.points, quad.weights):
189 A_k_ell[i, j] += factor * w * func_k(i, q) * func_ell(j, q)
190
191 else:
192 A_k_ell = np.zeros((ndof_k, ndof_ell))
193
194 #print(k, ell, relative_position, "\n", A_k_ell, "\n")
195 return {gamma: A_k_ell}

297 def _assemble_A_FF(self, fs_pair, relative_position):
298 """Assemble weak form for A_{FF}^{(k,ell)}
299 :arg fs_pair: pair of function space indices that couple
300 k = fs_pair[0]: first function space (to-space): a facet function space
301 ell = fs_pair[1]: second function space (from-space): a facet function space
302 :arg relative_position: relative position of the facet within a cell
303 """
304 k = fs_pair[0]
305 ell = fs_pair[1]
306 ndof_k = self._fe_F[k].ndof
307 ndof_ell = self._fe_F[ell].ndof
308 A_k_ell = np.empty((ndof_k, ndof_ell))
309 gamma = (0, 0)
310
311 # Quadrature rule: based on the max degree of basis in the function spaces pair
312 max_degree = max(self._fe_F[k].degree, self._fe_F[ell].degree)
313 quad = GaussianQuadrature(max_degree + 1)
314
315 if fs_pair in [(0,0), (1,1)]: # Facet space mass matrices A_q+_q+, A_q-_q-
316 func_k = self._fe_F[k].evaluate
317 func_ell = self._fe_F[ell].evaluate
318 factor = 0.5
319
320 if relative_position in ["left", "right"]:
321 gamma = (0, 1) # scale by hy
322 else:
323 gamma = (1, 0) # scale by hx
324
325 # Assemble local matrix A^(k,ell)
326 for i in range(ndof_k):
327 for j in range(ndof_ell):
328 A_k_ell[i, j] = 0.0
329 for q, w in zip(quad.points, quad.weights):
330 A_k_ell[i, j] += factor * w * func_k(i, q) * func_ell(j, q)
331
332 elif fs_pair == (2, 2): # A_qf_qf
333 gamma = (0, 0)
334 A_k_ell = np.eye(ndof_k)
335
336 elif fs_pair in [(2,0), (2,1)]: # A_q+_qf, A_q-_qf
337 gamma = (0, 0)
338 A_k_ell = -0.5*np.eye(ndof_k)
339
340 else:
341 gamma = (0, 0)
342 A_k_ell = np.zeros((ndof_k, ndof_ell))
343
344 #print(k, ell, relative_position, "\n", A_k_ell, "\n")
