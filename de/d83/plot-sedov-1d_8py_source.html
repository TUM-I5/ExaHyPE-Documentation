104def calc_a(g, nu=3):
105 """
106 exponents of the polynomials of the sedov solution
107 g - the polytropic gamma
108 nu - the dimension
109 """
110 a = [0] * 8
111
112 a[0] = 2.0 / (nu + 2)
113 a[2] = (1 - g) / (2 * (g - 1) + nu)
114 a[3] = nu / (2 * (g - 1) + nu)
115 a[5] = 2 / (g - 2)
116 a[6] = g / (2 * (g - 1) + nu)
117
118 a[1] = (((nu + 2) * g) / (2.0 + nu * (g - 1.0))) * (
119 (2.0 * nu * (2.0 - g)) / (g * (nu + 2.0) ** 2) - a[2]
120 )
121 a[4] = a[1] * (nu + 2) / (2 - g)
122 a[7] = (2 + nu * (g - 1)) * a[1] / (nu * (2 - g))
123 return a

126def calc_beta(v, g, nu=3):
127 """
128 beta values for the sedov solution (coefficients of the polynomials of the similarity variables)
129 v - the similarity variable
130 g - the polytropic gamma
131 nu- the dimension
132 """
133
134 beta = (
135 (nu + 2)
136 * (g + 1)
137 * np.array(
138 (
139 0.25,
140 (g / (g - 1)) * 0.5,
141 -(2 + nu * (g - 1))
142 / 2.0
143 / ((nu + 2) * (g + 1) - 2 * (2 + nu * (g - 1))),
144 -0.5 / (g - 1),
145 ),
146 )
147 )
148
149 beta = np.outer(beta, v)
150
151 beta += (g + 1) * np.array(
152 (
153 0.0,
154 -1.0 / (g - 1),
155 (nu + 2) / ((nu + 2) * (g + 1) - 2.0 * (2 + nu * (g - 1))),
156 1.0 / (g - 1),
157 ),
158 ).reshape((4, 1))
159
160 return beta

163def sedov(t, E0, rho0, g, n=1000, nu=3):
164 """
165 solve the sedov problem
166 t - the time
167 E0 - the initial energy
168 rho0 - the initial density
169 n - number of points (10000)
170 nu - the dimension
171 g - the polytropic gas gamma
172 """
173 # the similarity variable
174 v_min = 2.0 / ((nu + 2) * g)
175 v_max = 4.0 / ((nu + 2) * (g + 1))
176
177 v = v_min + np.arange(n) * (v_max - v_min) / (n - 1.0)
178
179 a = calc_a(g, nu)
180 beta = calc_beta(v, g=g, nu=nu)
181 lbeta = np.log(beta)
182
183 r = np.exp(-a[0] * lbeta[0] - a[2] * lbeta[1] - a[1] * lbeta[2])
184 rho = ((g + 1.0) / (g - 1.0)) * np.exp(
185 a[3] * lbeta[1] + a[5] * lbeta[3] + a[4] * lbeta[2]
186 )
187 p = np.exp(
188 nu * a[0] * lbeta[0] + (a[5] + 1) * lbeta[3] + (a[4] - 2 * a[1]) * lbeta[2]
189 )
190 u = beta[0] * r * 4.0 / ((g + 1) * (nu + 2))
191 p *= 8.0 / ((g + 1) * (nu + 2) * (nu + 2))
192
193 # we have to take extra care at v=v_min, since this can be a special point.
194 # It is not a singularity, however, the gradients of our variables (wrt v) are.
195 # r -> 0, u -> 0, rho -> 0, p-> constant
196
197 u[0] = 0.0
198 rho[0] = 0.0
199 r[0] = 0.0
200 p[0] = p[1]
201
202 # volume of an n-sphere
203 vol = (np.pi ** (nu / 2.0) / Gamma(nu / 2.0 + 1)) * np.power(r, nu)
204
205 # note we choose to evaluate the integral in this way because the
206 # volumes of the first few elements (i.e near v=vmin) are shrinking
207 # very slowly, so we dramatically improve the error convergence by
208 # finding the volumes exactly. This is most important for the
209 # pressure integral, as this is on the order of the volume.
210
211 # (dimensionless) energy of the model solution
212 de = rho * u * u * 0.5 + p / (g - 1)
213 # integrate (trapezium rule)
214 q = np.inner(de[1:] + de[:-1], np.diff(vol)) * 0.5
215
216 # the factor to convert to this particular problem
217 fac = (q * (t**nu) * rho0 / E0) ** (-1.0 / (nu + 2))
218
219 # shock speed
220 shock_speed = fac * (2.0 / (nu + 2))
221 rho_s = ((g + 1) / (g - 1)) * rho0
222 r_s = shock_speed * t * (nu + 2) / 2.0
223 p_s = (2.0 * rho0 * shock_speed * shock_speed) / (g + 1)
224 u_s = (2.0 * shock_speed) / (g + 1)
225
226 r *= fac * t
227 u *= fac
228 p *= fac * fac * rho0
229 rho *= rho0
230 return r, p, rho, u, r_s, p_s, rho_s, u_s, shock_speed
