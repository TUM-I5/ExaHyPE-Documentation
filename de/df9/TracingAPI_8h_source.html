289 template <typename Particle> std::string pruneTypeName() {
290 std::string result = typeid(Particle).name();
291 std::size_t lastDigit = result.find_last_of("0123456789");
292 if (lastDigit != std::string::npos) {
293 result = result.substr(lastDigit + 1);
294 }
295 assertion2(result.size() > 1, typeid(Particle).name(), result);
296 return result.size() > 1 ? result.substr(0, result.size() - 1) : result;
297 }
