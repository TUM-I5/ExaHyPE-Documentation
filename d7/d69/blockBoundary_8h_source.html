37 BlockBoundary(Face topography, bool top, int n_depth, double depth){
38 /*We asume topography is in the eta xi plane at the bottom of the domain*/
39
40 Coordinate coords_topo[2]; topography.getCoord(coords_topo);
41 Coordinate eta = topography.getCoord(1);
42 Coordinate xi = topography.getCoord(0);
43 Coordinate mu = getNormal(eta,xi);
44
45 // set coordiantes for block
46 coords[2] = eta;
47 coords[1] = xi;
48 coords[0] = mu;
49
50 //set index_size of block
51 topography.getIndexSize(index_size); // in eta and xi we copy from the topography
52 index_size[mu] = n_depth; // in mu the size is given
53
54 //we copy the index offset from the topography
55 topography.getIndexOffset(index_offset);
56
57 // in case it is on top we have subtract the index depth
58 if(top) index_offset[mu] = index_offset[mu] - n_depth + 1;
59
60 //--------------------//
61
62 //--Next we span the edges of the block--//
63 std::vector<int> range_eta; topography.getIndexRange(eta,range_eta);
64 std::vector<int> range_xi; topography.getIndexRange(xi,range_xi);
65
66
67
68 double offset_block[3];
69 offset_block[eta] = topography(eta,eta,range_eta.front(),xi,range_xi.front());
70 offset_block[xi] = topography(xi ,eta,range_eta.front(),xi,range_xi.front());
71 offset_block[mu] = topography.getPosition();
72 // in case the topography is on top we have to subtract the depth
73 if(top) offset_block[mu] -= depth;
74
75 double size_block[3];
76 size_block[eta] = topography(eta,eta,range_eta.back(),xi,range_xi.back()) - offset_block[eta];
77 size_block[xi] = topography(xi, eta,range_eta.back(),xi,range_xi.back()) - offset_block[xi];
78 size_block[mu] = depth;
79 //---------------------------------------//
80
81
82 double offset_other[3];
83 double size_other[3];
84 int index_offset_other[3];
85 int index_size_other[3];
86
87 std::copy_n(index_offset, 3, index_offset_other);
88 std::copy_n(index_size , 3, index_size_other);
89 std::copy_n(offset_block, 3, offset_other);
90 std::copy_n(size_block , 3, size_other);
91
92 if(!top){
93 index_offset_other[mu] += index_size_other[mu] - 1;
94 offset_other[mu] += size_other[mu];
95 }
96 index_size_other[mu] = 1;
97 size_other[mu] = 0.0;
98
99 FaceBoundary face_boundary_mu_other(offset_other, size_other, index_size_other, coords_topo);
100 Face face_mu_other(index_offset_other,face_boundary_mu_other);
101 //--set faces in mu --//
102 setFace(mu, top, topography); // set topography as first face
103 setFace(mu, !top, face_mu_other);
104 /*---------------*/
105
106 /*--Init edges in corners of xi eta---*/
107 double start_eta_false_xi_false[3]; std::copy_n(offset_block,3,start_eta_false_xi_false);
108 double start_eta_true_xi_false[3]; std::copy_n(offset_block,3,start_eta_true_xi_false);
109 start_eta_true_xi_false[eta] += size_block[eta];
110 double start_eta_false_xi_true[3]; std::copy_n(offset_block,3,start_eta_false_xi_true);
111 start_eta_false_xi_true[xi] += size_block[xi];
112 double start_eta_true_xi_true[3]; std::copy_n(offset_block,3,start_eta_true_xi_true);
113 start_eta_true_xi_true[eta] += size_block[eta];
114 start_eta_true_xi_true[xi] += size_block[xi];
115 double stop_eta_false_xi_false[3];
116 std::copy_n(start_eta_false_xi_false,3,stop_eta_false_xi_false);
117 double stop_eta_true_xi_false[3];
118 std::copy_n(start_eta_true_xi_false,3,stop_eta_true_xi_false);
119 double stop_eta_false_xi_true[3];
120 std::copy_n(start_eta_false_xi_true,3,stop_eta_false_xi_true);
121 double stop_eta_true_xi_true[3];
122 std::copy_n(start_eta_true_xi_true,3,stop_eta_true_xi_true);
123
124 stop_eta_false_xi_false[mu] += size_block[mu];
125 stop_eta_false_xi_true[mu] += size_block[mu];
126 stop_eta_true_xi_false[mu] += size_block[mu];
127 stop_eta_true_xi_true[mu] += size_block[mu];
128
129
130 Edge edge_eta_false_xi_false(mu, start_eta_false_xi_false, stop_eta_false_xi_false, n_depth);
131 Edge edge_eta_true_xi_false (mu, start_eta_true_xi_false , stop_eta_true_xi_false , n_depth);
132 Edge edge_eta_false_xi_true (mu, start_eta_false_xi_true , stop_eta_false_xi_true , n_depth);
133 Edge edge_eta_true_xi_true (mu, start_eta_true_xi_true , stop_eta_true_xi_true , n_depth);

163 offset_block[eta],
164 edge_eta_false_xi_false,
165 edge_eta_false_xi_true,
166 edge_eta_false_mu_false,
167 edge_eta_false_mu_true
168 );

171 edge_eta_true_xi_false,
172 edge_eta_true_xi_true,
173 edge_eta_true_mu_false,
174 edge_eta_true_mu_true
175 );
176 setFace(eta,false,Face(index_offset_neta_false, face_boundary_neta_false));
177 setFace(eta,true, Face(index_offset_neta_true , face_boundary_neta_true));

190 int index_offset_nxi_false[3];

192 getFace(mu,false).getIndexOffset(index_offset_nxi_true);

198 edge_eta_false_xi_false,
199 edge_eta_true_xi_false,
200 edge_xi_false_mu_false,
201 edge_xi_false_mu_true
202 );

206 edge_eta_false_xi_true,
207 edge_eta_true_xi_true,
208 edge_xi_true_mu_false,
209 edge_xi_true_mu_true
210 );

343 void split(Face& face, BlockBoundary& left, BlockBoundary& right){
344
345 /* We assume we split in the xi-mu plane*/
346 Coordinate xi = face.getCoord(1);
347 Coordinate mu = face.getCoord(0);
348 Coordinate eta = getNormal(xi,mu);
349
350 // Check if face is inside the block
351 int face_index_offset[3];
352 int face_index_range[3];
353
354 //face.toNetCDF("fault_in.nc");
355 face.getIndexOffset(face_index_offset);
356 face.getIndexSize (face_index_range);
357
358 bool face_is_in_block = face_index_offset[eta] > getFace(eta,false).getIndexOffset(eta);
359 face_is_in_block &= face_index_offset[eta] < getFace(eta,true ).getIndexOffset(eta);
360
361 if(!face_is_in_block){
362 std::cout << "Warning splitting face is outside of block in Coordinate "<< coordToString(eta) << std::endl;
363 std::cout << "Spliting face index: " << face_index_offset[eta] << std::endl;
364 std::cout << "Block index range" << getFace(eta,false ).getIndexOffset(eta) << "-"
365 << getFace(eta,true ).getIndexOffset(eta) << std::endl;
366
367 std::cout << "Face:" << std::endl;
368 std::cout << face << std::endl;
369 std::cout << "Boundary:" << std::endl;
370 std::cout << toString() << std::endl;
371 std::cout << getFace(eta,false) << std::endl;
372 std::cout << getFace(eta,true ) << std::endl;
373 }
374
375 //Check if face covers mu and xi
376 bool face_covers_mu = face_index_offset[mu] < getFace(mu,false).getIndexOffset(mu);
377 face_covers_mu &=
378 face_index_offset[mu] + face_index_range[mu] > getFace(mu,true).getIndexOffset(mu);
379
380 bool face_covers_xi = face_index_offset[xi] < getFace(xi ,false).getIndexOffset(xi );
381 face_covers_xi &=
382 face_index_offset[xi ] + face_index_range[xi ] > getFace(xi ,true).getIndexOffset(xi );
383
384 // Set unchanged faces in resulting blocks
385 left .setFace(eta,false,getFace(eta,false));
386 right.setFace(eta,true ,getFace(eta,true ));
387
388 //split upperFace
389 Face upperLeft, upperRight;
390 Edge upperPath = getFace(mu, true).split(face,upperLeft,upperRight);
391
392 //split lowerFace
393 Face lowerLeft, lowerRight;
394 Edge lowerPath = getFace(mu, false).split(face,lowerLeft,lowerRight);
395
396 //split backFace
397 Face backLeft, backRight;
398 Edge backPath = getFace(xi, true) .split(face,backLeft,backRight);
399
400 //split frontFace
401 Face frontLeft, frontRight;
402 Edge frontPath = getFace(xi, false).split(face,frontLeft,frontRight);
403
404 Coordinate coords_fault[2] = {mu,xi};
405
406 FaceBoundary fault_boundary(coords_fault,face.getPosition(),
407 frontPath,backPath,lowerPath,upperPath);
408
409 int index_offset_fault[3]; std::copy_n(index_offset,3,index_offset_fault);
410 index_offset_fault[eta] = face_index_offset[eta];
411 Face fault(index_offset_fault,fault_boundary,face.getPerturbation(),true);
412 fault.toNetCDF("fault_cut.nc");
413
414 //Set split faces in resulting blocks
415 left .setFace(mu, false, lowerLeft);
416 right.setFace(mu, false, lowerRight);
417
418 left .setFace(mu, true, upperLeft);
419 right.setFace(mu, true, upperRight);
420
421 left .setFace(xi, false, frontLeft );
422 right.setFace(xi, false, frontRight);
423
424 left .setFace(xi, true, backLeft );
425 right.setFace(xi, true, backRight);
426
427 left .setFace(eta, true, fault);
428 right.setFace(eta, false, fault);
429
430 int index_offset_left[3]=
431 {left.getFace(Coordinate::Z,false).getIndexOffset(Coordinate::Z),
432 left.getFace(Coordinate::Y,false).getIndexOffset(Coordinate::Y),
433 left.getFace(Coordinate::X,false).getIndexOffset(Coordinate::X)};
434
435 left.setIndexOffset(index_offset_left);
436
437 int index_offset_right[3];
438 std::copy_n(index_offset,3,index_offset_right);
439 index_offset_right[eta] = face.getIndexOffset(eta);
440 right.setIndexOffset(index_offset_right);
441
442 int index_size_left[3];
443 std::copy_n(index_size,3,index_size_left);
444 index_size_left[eta] = face.getIndexOffset(eta) - index_offset[eta] + 1;
445 left.setIndexSize(index_size_left);
446
447 int index_size_right[3];
448 std::copy_n(index_size,3,index_size_right);
449 index_size_right[eta] = index_size[eta] - index_size_left[eta] + 1;
450 right.setIndexSize(index_size_right);
451
452 }

492 void getBoundingBox(double offset[3], double size[3]){
493 double coord_max[3] = {std::numeric_limits<double>::lowest(),
494 std::numeric_limits<double>::lowest(),
495 std::numeric_limits<double>::lowest()};
496 double coord_min[3] = {std::numeric_limits<double>::max(),
497 std::numeric_limits<double>::max(),
498 std::numeric_limits<double>::max()};
499
500 for(int f_ind = 0; f_ind < 6; f_ind++){
501 BlockFace f = static_cast<BlockFace>(f_ind);
502 for(int c_ind = 0; c_ind < 3; c_ind++){
503 Coordinate c = static_cast<Coordinate>(c_ind);
504 Face face = getFace(f);
505 Coordinate face_coord[2];
506 face.getCoord(face_coord);
507 std::vector<int> index_range_0; face.getIndexRange(face_coord[0],index_range_0);
508 std::vector<int> index_range_1; face.getIndexRange(face_coord[1],index_range_1);
509 for(int i : index_range_0){
510 for(int j : index_range_1){
511 coord_max[c] = std::max(coord_max[c],face(c,i,j));
512 coord_min[c] = std::min(coord_min[c],face(c,i,j));
513 }
514 }
515 }
516 }
517
518 for(int c_ind = 0; c_ind < 3; c_ind++){
519 offset[c_ind] = coord_min[c_ind];
520 size [c_ind] = coord_max[c_ind] - coord_min[c_ind];
521 }
522 }

535 FaceBoundary setupSplittingFace(Coordinate cycx[2],
536 int fault_index,
537 double fault_position,
538 int indexOffset_fault[3]
539 ){
540 int dmesh = 8;
541 Coordinate c_depth_fault = getNormal(cycx[0], cycx[1]);
542
543 double offset[3];
544 double size [3];
545
546 double offset_fault[3];
547 double size_fault[3];
548 getBoundingBox(offset,size); //get bounding box of boundary
549
550 int indexSize_fault[3];
551 std::copy_n(index_size,3,indexSize_fault);
552 std::copy_n(index_offset,3,indexOffset_fault);
553
554 std::copy_n(offset,3,offset_fault);
555 offset_fault[cycx[0]] -= size[cycx[0]]/indexSize_fault[cycx[0]] * dmesh/2; // add buffer
556 offset_fault[cycx[1]] -= size[cycx[1]]/indexSize_fault[cycx[1]] * dmesh/2; // add buffer
557 offset_fault[c_depth_fault] = fault_position;
558
559 std::copy_n(size,3,size_fault);
560 size_fault[cycx[0]] += size[cycx[0]]/indexSize_fault[cycx[0]] * dmesh; // add buffer
561 size_fault[cycx[1]] += size[cycx[1]]/indexSize_fault[cycx[1]] * dmesh; // add buffer
562
563
564 indexOffset_fault[cycx[0]]-=dmesh/2; // add buffer
565 indexOffset_fault[cycx[1]]-=dmesh/2; // add buffer
566 indexOffset_fault[c_depth_fault] = fault_index;
567
568 indexSize_fault[c_depth_fault] = 1;
569 indexSize_fault[cycx[0]] += dmesh; // add buffer
570 indexSize_fault[cycx[1]] += dmesh; // add buffer
571
572 return FaceBoundary(offset_fault,
573 size_fault,
574 indexSize_fault,
575 cycx);
576 }
