48 struct offsetHash {
49 std::size_t operator()(const std::tuple<double, double, double>& offset) const {
50 double tolerance = 1e-4;
51 auto roundToTolerance = [tolerance](double value) {
52 return std::round(value / tolerance) * tolerance;
53 };
54
55 double x = roundToTolerance(std::get<0>(offset));
56 double y = roundToTolerance(std::get<1>(offset));
57 double z = roundToTolerance(std::get<2>(offset));
58
59 std::size_t hash1 = std::hash<double>()(x);
60 std::size_t hash2 = std::hash<double>()(y);
61 std::size_t hash3 = std::hash<double>()(z);
62
63 return hash1 ^ (hash2 << 1) ^ (hash3 << 2);
64 }
65 };
