67 static double lagrangePolynomial(double position, int i, std::vector<double> range){
68 double nominator = 1;
69 double denominator = 1;
70 for (int j = 0 ; j < i ; j++){
71 nominator *= (position - range[j]);
72 denominator *= (range[i] - range[j]);
73 }
74
75 for (int j = i+1 ; j < range.size() ; j++){
76 nominator *= (position - range[j]);
77 denominator *= (range[i] - range[j]);
78 }
79
80 return nominator/denominator;
81 }
