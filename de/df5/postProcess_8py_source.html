3def parse_patch_file(file_path, solver_type):
4 """
5 Transform data from Peano patch file into numpy array
6 in the format [x_vertex, y_vertex, value]
7
8 :arg file_path: path to Peano patch file
9 :solver_type: "CG" or "DG"
10 """
11
12 if solver_type not in ["CG", "DG"]: raise ValueError("Invalid solver type. Expected 'CG' or 'DG'.")
13
14 with open(file_path, 'r') as file:
15 lines = file.readlines()
16
17 if solver_type == 'CG':
18 # For CG, convert the patch file format into a list:
19 data = [] # the output format is [x, y, val], where (x,y) is vertex coordinates
20 i = 0
21 while i < len(lines):
22 if lines[i].startswith("begin patch"):
23 # Read 'offset x_left_bottom_node_coordinate y_left_bottom_node_coordinate'
24 _, xlb, ylb = lines[i+1].split()
25 xlb = float(xlb)
26 ylb = float(ylb)
27
28 # Read 'size x_cell_size y_cell_size'
29 _, hx, hy = lines[i+2].split()
30 hx = float(hx)
31 hy = float(hy)
32
33 # Read vertex values in the order: left-bottom, right-bottom, left-top, right-top
34 values = [float(value) for value in lines[i+4].split()]
35
36 # Write in the output
37 data.extend([[xlb, ylb, values[0]],
38 [xlb+hx, ylb, values[1]],
39 [xlb, ylb+hy, values[2]],
40 [xlb+hx, ylb+hy, values[3]]])
41
42 # jump to the next 'begin patch'
43 i += 8
44
45 else:
46 i += 1
47
48 # keep only unique rows (one vertex -- one entry)
49 tolerance = 1e-9
50 data_unique = []
51
52 for row in data:
53 # Check if row=[x,y,val] is close to any existing row
54 if not any( np.all(np.isclose(row, unique_row, atol=tolerance)) for unique_row in data_unique ):
55 data_unique.append(row)
56
57 # assert:
58 # size of data = 4*N^2
59 # size of data_unique = (N+1)^2, where N is the number of cells
60 assert np.sqrt(np.shape(data)[0]) == (np.sqrt(np.shape(data_unique)[0]) - 1) * 2, f'{np.shape(data) = }, {np.shape(data_unique) = }'
61 result = np.array(data_unique)
62
63 elif solver_type == 'DG':
64 # For DG, convert the patch file format into a dictionary:
65 # {(x*,y*): [val0, val1, ...], ...}, where
66 # (x*,y*) are the coordinates of the left bottom vertex of the cell and
67 # val0, val1, ... are values assigned to the cell dofs
68
69 result = {}
70
71 i = 0
72 while i < len(lines):
73 if lines[i].startswith("begin patch"):
74 # Read line 'offset x_left_bottom_node_coordinate y_left_bottom_node_coordinate'
75 _, xlb, ylb = lines[i+1].split()
76 # Read values
77 values = [float(value) for value in lines[i+4].split()]
78 result[( float(xlb), float(ylb) )] = values
79
80 # jump to the next 'begin patch'
81 i += 8
82 else:
83 # check next line
84 i += 1
85
86 return result
