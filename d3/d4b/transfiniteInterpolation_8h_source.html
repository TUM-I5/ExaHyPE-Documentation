35 static void interpolate2D( FaceBoundary& boundary,
36 std::vector<double>& face ){
37
38 Coordinate eta = boundary.getCoord(1);
39 Coordinate xi = boundary.getCoord(0);
40
41 int neta = boundary.getIndexSize(eta);
42 int nxi = boundary.getIndexSize(xi);
43
44 double mesh_size_eta = 1.0/(neta-1);
45 double mesh_size_xi = 1.0 /(nxi-1);
46
47 double r;
48 double q;
49
50 idx3 id_dxy(NCOORDS, nxi, neta);
51 idx2 id_dx (NCOORDS, neta);
52 idx2 id_dy (NCOORDS, nxi);
53
54 Edge& left_bnd = boundary.getEdge(eta,false);
55 Edge& right_bnd = boundary.getEdge(eta,true);
56 Edge& bottom_bnd = boundary.getEdge(xi ,false);
57 Edge& top_bnd = boundary.getEdge(xi ,true);
58
59 for(int d = 0 ; d < NCOORDS ; d++) {
60 for(int j =0 ; j < nxi ; j++) {
61 for(int i = 0 ; i < neta ; i++) {
62 q = (i)*mesh_size_eta;
63 r = (j)*mesh_size_xi;
64 //From https://en.wikipedia.org/wiki/Transfinite_interpolation
65 face[id_dxy(d,j,i)]=
66 (1-q) * left_bnd(d,j) + q * right_bnd(d,j)
67 +(1-r) * bottom_bnd(d,i) + r * top_bnd(d,i)
68 -(1-q) * (1-r) * left_bnd(d,0) //intersection left bottom
69 -q * (1-r) * right_bnd(d,0) //intersection right bottom
70 -(1-q) * r * top_bnd(d,0) //intersection left top
71 -q * r * top_bnd(d,neta-1); //intersection right top
72 }
73 }
74 }
75 }

92 static void interpolate3D( BlockBoundary& boundary,
93 int o_mu ,int s_mu,
94 int o_xi ,int s_xi,
95 int o_eta ,int s_eta,
96 Block& block){
97
98 Coordinate coords[3];
99 boundary.getCoord(coords);
100 Coordinate eta = coords[2];
101 Coordinate xi = coords[1];
102 Coordinate mu = coords[0];
103
104 double bs_eta = boundary.getIndexSize(eta);
105 double bs_xi = boundary.getIndexSize(xi);
106 double bs_mu = boundary.getIndexSize(mu);
107
108 double bo_eta = boundary.getIndexOffset(eta);
109 double bo_xi = boundary.getIndexOffset(xi);
110 double bo_mu = boundary.getIndexOffset(mu);
111
112 double mesh_size_eta = 1.0/double(bs_eta - 1.0);
113 double mesh_size_xi = 1.0/double(bs_xi - 1.0);
114 double mesh_size_mu = 1.0/double(bs_mu - 1.0);
115
116 int bp_eta = bo_eta + bs_eta;
117 int bp_xi = bo_xi + bs_xi;
118 int bp_mu = bo_mu + bs_mu;
119
120
121 int p_eta = o_eta + s_eta;
122 int p_xi = o_xi + s_xi;
123 int p_mu = o_mu + s_mu;
124
125 double u,v,w;
126 double uv,vw,uw,uvw1,uvw2;
127
128 double d_eta,d_xi,d_mu;
129
130 Face left_bnd = boundary.getFace(eta,false); // mu xi
131 Face right_bnd = boundary.getFace(eta,true); // mu xi
132 Face bottom_bnd = boundary.getFace(mu, false); // xi eta
133 Face top_bnd = boundary.getFace(mu, true); // xi eta
134 Face front_bnd = boundary.getFace(xi, false); // mu eta
135 Face back_bnd = boundary.getFace(xi, true); // mu eta
136
137 for(int d = 0 ; d < NCOORDS ; d++) { //eta
138 for(int i_eta = o_eta ; i_eta< p_eta ; i_eta++){
139 d_eta = mesh_size_eta * (i_eta - bo_eta);
140 for(int i_xi = o_xi ; i_xi< p_xi ; i_xi++){ //xi
141 d_xi = mesh_size_xi * (i_xi - bo_xi);
142 for(int i_mu = o_mu ; i_mu< p_mu ; i_mu++){ //mu
143 d_mu = mesh_size_mu * (i_mu - bo_mu);
144
145 //(1-eta) * l(0,i_xi) + eta * r(0,i_xi)
146 u=(1-d_eta) * left_bnd (d,mu,i_mu,xi,i_xi) +
147 d_eta * right_bnd (d,mu,i_mu,xi,i_xi);
148
149 //d_mu = 0 => i_mu = 0
150 v=(1-d_mu) * bottom_bnd(d,xi,i_xi,eta,i_eta)+
151 d_mu * top_bnd (d,xi,i_xi,eta,i_eta);
152
153 //(1-xi) * f(0,i_eta) + xi * b(0,i_eta)
154 w=(1-d_xi) * front_bnd(d,mu,i_mu,eta,i_eta)+
155 d_xi * back_bnd (d,mu,i_mu,eta,i_eta);
156
157 // (1-eta) * l(0,i_xi) + eta * l(0,0) * (1-eta) * l(0,i_xi) + eta * l(0,0)
158 uw=(1-d_eta) * (1-d_xi) * left_bnd (d,mu,i_mu,xi,bo_xi ) +
159 (1-d_eta) * d_xi * left_bnd (d,mu,i_mu,xi,bp_xi-1 ) +
160 d_eta * (1-d_xi) * right_bnd(d,mu,i_mu,xi,bo_xi ) +
161 d_eta * d_xi * right_bnd(d,mu,i_mu,xi,bp_xi-1 );
162
163 // (1-xi) * B(0,i_eta) + xi * B(n_xi-1,i_eta)
164 uv=(1-d_mu)* (1-d_eta) * bottom_bnd(d,xi,i_xi,eta,bo_eta ) +
165 (1-d_mu)* d_eta * bottom_bnd(d,xi,i_xi,eta,bp_eta-1) +
166 d_mu * (1-d_eta) * top_bnd (d,xi,i_xi,eta,bo_eta ) +
167 d_mu * d_eta * top_bnd (d,xi,i_xi,eta,bp_eta-1);
168
169 // (1-xi) * f(0,i_eta) + ix * b(0,i_eta)
170 vw= (1-d_xi) * (1-d_mu) * front_bnd(d,mu,bo_mu ,eta,i_eta) +
171 (1-d_xi) * d_mu * front_bnd(d,mu,bp_mu-1,eta,i_eta) +
172 d_xi * (1-d_mu) * back_bnd (d,mu,bo_mu ,eta,i_eta) +
173 d_xi * d_mu * back_bnd (d,mu,bp_mu-1,eta,i_eta);
174
175 //(1-xi)*(1-eta) B(0,0) + (xi)*(1-eta) B(n_xi-1,0)
176 uvw1= (1-d_mu)*(1-d_xi)*(1-d_eta) * bottom_bnd(d,xi,bo_xi ,eta,bo_eta) +
177 (1-d_mu)* d_xi *(1-d_eta) * bottom_bnd(d,xi,bp_xi-1,eta,bo_eta) +
178 d_mu *(1-d_xi)*(1-d_eta) * top_bnd (d,xi,bo_xi ,eta,bo_eta) +
179 d_mu * d_xi *(1-d_eta) * top_bnd (d,xi,bp_xi-1,eta,bo_eta);
180
181 //(1-xi)*(1-eta) B(0,n_eta-1) + (xi)*(1-eta) B(n_xi-1,n_eta-1)
182 uvw2= (1-d_mu)*(1-d_xi)* d_eta * bottom_bnd (d,xi,bo_xi ,eta,bp_eta-1) +
183 (1-d_mu)* d_xi * d_eta * bottom_bnd (d,xi,bp_xi-1,eta,bp_eta-1) +
184 d_mu *(1-d_xi)* d_eta * top_bnd (d,xi,bo_xi ,eta,bp_eta-1) +
185 d_mu * d_xi * d_eta * top_bnd (d,xi,bp_xi-1,eta,bp_eta-1);
186
187 block(d,coords[0],i_mu,coords[1],i_xi,coords[2],i_eta) =
188 u + v + w - uv - uw - vw + uvw1 + uvw2;
189 }
190 }
191 }
192 }
