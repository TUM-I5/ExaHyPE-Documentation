75/*Initializes the boundary to a given face*/
76FaceBoundary::FaceBoundary(Face const& face):
77 left_edge (Edge(face.getCoord(0))),
78 right_edge (Edge(face.getCoord(0))),
79 bottom_edge(Edge(face.getCoord(1))),
80 top_edge (Edge(face.getCoord(1)))
81{
82 /*Copy position, indeces and coordinates from face*/
83 face.getIndexSize(index_size);
84 face.getCoord(coords);
85 position = face.getPosition();
86
87 /*Copy values from the boundary of the face*/
88 std::vector<int> range_eta; face.getIndexRange(coords[1],range_eta);
89 std::vector<int> range_xi; face.getIndexRange(coords[0],range_xi );
90
91 for(int c_int = 0 ; c_int < NCOORDS ; c_int++){
92 Coordinate c = static_cast<Coordinate>(c_int);
93
94 for (int i : range_eta){
95 bottom_edge.push_back(face(c, range_xi.front(), i));
96 top_edge .push_back(face(c, range_xi.back() , i));
97 }
98 for (int i : range_xi ){
99 left_edge .push_back(face(c, i, range_eta.front()));
100 right_edge.push_back(face(c, i, range_eta.back ()));
101 }
102 }
103
104 int minSize[3] = {1,1,1};
105 minSize[coords[0]] = 2; minSize[coords[1]] = 2;
106 for(int i = 0 ; i < 3 ; i++ ){
107 if(index_size[i] < minSize[i]){
108 std::stringstream ss;
109 ss << "Error: Init faceBoundary with not enough nodes one dimension "<< std::endl;
110 ss << coordToString(coords[0]) << " " << coordToString(coords[0]) << std::endl;
111 ss << index_size[0] << " " << index_size[1] << " " << index_size[2] << std::endl;
112 throw std::invalid_argument(ss.str());
113 }
114 }
115
116 checkConformity();
117}

119/*Initializes the boundary to a given set of geometric parameters*/
120FaceBoundary::FaceBoundary(double offset[3], double size[3],
121 int a_index_size[3], Coordinate a_coords[2]):
122 left_edge (Edge(a_coords[0])),
123 right_edge (Edge(a_coords[0])),
124 bottom_edge(Edge(a_coords[1])),
125 top_edge (Edge(a_coords[1]))
126{
127
128 /*Set position, indeces*/
129 std::copy_n(a_coords ,2, coords);
130 std::copy_n(a_index_size ,3, index_size );
131 /*Test input arguments*/
132 int minSize[3] = {1,1,1};
133 minSize[coords[0]] = 2; minSize[coords[1]] = 2;
134 for(int i = 0 ; i < 3 ; i++ ){
135 bool error=false;
136 std::stringstream ss;
137 if(index_size[i] < minSize[i]){
138 error = true;
139 ss << "Error: Init faceBoundary with not enough nodes in one dimension "<< std::endl;
140 }
141
142 if(!std::isfinite(offset[i]) || !std::isfinite(size[i])){
143 error = true;
144 ss << "Error: Init faceBoundary with not finite domain definition"<< std::endl;
145 ss << "Size: " << size[0] << " " << size[1] << " " << size[2] << std::endl;
146 ss << "Offset: "<< offset[0] << " " << offset[1] << " " << offset[2] << std::endl;
147 }
148
149 if(error){
150 ss << (*this) << std::endl;
151 throw std::invalid_argument(ss.str());
152 }
153 }
154 /*******************/
155 Coordinate eta = coords[1];
156 Coordinate xi = coords[0];
157 Coordinate mu = getNormal(eta, xi);
158
159 position = offset[mu];
160 /***********************/
161
162 int index_size_eta = index_size[coords[1]];
163 int index_size_xi = index_size[coords[0]];
164
165 left_edge .resize(index_size_xi * NCOORDS);
166 right_edge .resize(index_size_xi * NCOORDS);
167 bottom_edge.resize(index_size_eta * NCOORDS);
168 top_edge .resize(index_size_eta * NCOORDS);
169
170
171 /*init top and bottom line*/
172 {
173 idx2 idx(NCOORDS, index_size_eta);
174 double deta = 1.0/double(index_size_eta-1) * size[eta];
175 for (int i = 0 ; i < index_size_eta ; i++){
176 double coord_eta = i * deta + offset[eta];
177
178 bottom_edge(eta,i) = coord_eta;
179 bottom_edge(xi ,i) = offset[xi];
180 bottom_edge(mu ,i) = position;
181
182 top_edge (eta,i) = coord_eta;
183 top_edge (xi ,i) = offset[xi] + size[xi];
184 top_edge (mu ,i) = position;
185 }
186 }
187
188 /*init left and right edge*/
189 {
190 idx2 idx(NCOORDS, index_size_xi);
191 double dxi = 1.0 / double(index_size_xi-1) * size[xi];
192 for (int i = 0 ; i < index_size_xi ; i++){
193 double coord_xi = i * dxi + offset[xi];
194
195 left_edge (eta,i) = offset[eta];
196 left_edge (xi ,i) = coord_xi;
197 left_edge (mu ,i) = position;
198
199 right_edge(eta,i) = offset[eta] + size[eta];
200 right_edge(xi ,i) = coord_xi;
201 right_edge(mu ,i) = position;
202 }
203 }

279/*Splitts a Face Boundary along an edge defined by intersection_path*/
280void FaceBoundary::split(Coordinate c,
281 int intersection_index,
282 Edge cutting_edge,
283 FaceBoundary& left, FaceBoundary& right){
284
285 //Check if coordinate c is part of Face Boundary
286 if(c != coords[0] && c!= coords[1]){
287 std::stringstream ss;
288 ss << "Splitting Face Boundary in wrong dimension" << std::endl;
289 ss << (*this) << std::endl;
290 ss << coordToString(c) << std::endl;
291 throw std::invalid_argument(ss.str());
292 }
293
294 //We assume we split from left to right, along eta:
295 Coordinate eta = c;
296 Coordinate xi = c == coords[0] ? coords[1]: coords[0];
297
298 // To keep the Face Boundary valid
299 // the intersection path needs to have the same size as its paralell edges
300 Edge o_edge = getEdge(eta,false);
301
302 if(o_edge.size() != cutting_edge.size()){
303 std::stringstream ss;
304 ss << "Trying to split Face Boundary with intersection path of wrong length" << std::endl;
305 ss << (*this) << std::endl;
306 ss << "Old edge" << std::endl;
307 ss << o_edge << std::endl;
308 ss << "Cutting edge" << std::endl;
309 ss << cutting_edge << std::endl;
310 throw std::invalid_argument(ss.str());
311 }
312
313 // We assume the first node of the intersection lies on the bottom edge of the face boundary
314 // same with the last node and the top edge
315 // Both edges will be split at these nodes
316
317 Edge top_edge (getEdge(xi,true));
318 Edge bottom_edge(getEdge(xi,false));
319 double intersection_node_top [3];
320 double intersection_node_bottom[3];
321
322 for(int d = 0 ; d< NCOORDS ; d++){
323 intersection_node_top[d] = cutting_edge(d, cutting_edge.numNodes()-1);
324 intersection_node_bottom[d] = cutting_edge(d, 0);
325 }
326
327 Edge left_new_bottom(bottom_edge.getDirection());
328 Edge left_new_top(top_edge.getDirection());
329 Edge right_new_bottom(bottom_edge.getDirection());
330 Edge right_new_top(top_edge.getDirection());
331
332 //split top and bottome edge
333
334 top_edge .split(intersection_node_top , left_new_top, right_new_top);
335 bottom_edge.split(intersection_node_bottom, left_new_bottom, right_new_bottom);
336
337 left_new_top = left_new_top.remesh(intersection_index + 1);
338 left_new_bottom = left_new_bottom.remesh(intersection_index + 1);
339
340 right_new_top = right_new_top.remesh(top_edge.numNodes() - intersection_index);
341 right_new_bottom = right_new_bottom.remesh(top_edge.numNodes() - intersection_index);
342
343
344 //set resulting face boundaries
345 if(eta == coords[1]){
346 left = FaceBoundary(coords,position,
347 getEdge(eta,false),cutting_edge,
348 left_new_bottom,left_new_top);
349
350 right = FaceBoundary(coords,position,
351 cutting_edge,getEdge(eta,true),
352 right_new_bottom,right_new_top);
353 }else{
354 left = FaceBoundary(coords,position,
355 left_new_bottom,left_new_top,
356 getEdge(eta,false),cutting_edge);
357
358 right = FaceBoundary(coords,position,
359 right_new_bottom,right_new_top,
360 cutting_edge,getEdge(eta,true));
361 }
362};

365void FaceBoundary::checkConformity(){
366 bool conform = true;
367 conform &= left_edge.getDirection() == right_edge.getDirection();
368 conform &= top_edge.getDirection() == bottom_edge.getDirection();
369
370 conform &= left_edge.numNodes() == right_edge.numNodes();
371 conform &= top_edge.numNodes() == bottom_edge.numNodes();
372
373 conform &= left_edge.getDirection() != top_edge.getDirection();
374
375 Coordinate eta = top_edge.getDirection() == coords[0] ? coords[0] : coords[1];
376 Coordinate xi = left_edge.getDirection() == coords[0] ? coords[0] : coords[1];
377
378 conform &= index_size[eta] == top_edge.numNodes();
379 conform &= index_size[xi] == left_edge.numNodes();
380
381 if(!conform){
382 std::stringstream ss;
383 ss << "faceBoundary is not conform" << std::endl;
384 ss << (*this) << std::endl;
385 ss << "left edge" << std::endl;
386 ss << left_edge << std::endl;
387 ss << "right edge" << std::endl;
388 ss << right_edge << std::endl;
389 ss << "top edge" << std::endl;
390 ss << top_edge << std::endl;
391 ss << "bottom edge" << std::endl;
392 ss << bottom_edge << std::endl;
393 throw std::range_error(ss.str());
394 }
395}
