163 if ( _deadlockTimeOut.count()>0 ) {

168 } else {
169 logWarning(
170 "writeTimeOutWarning(...)",
171 "deadlock detection switched off as deadlock time out is set to " << _deadlockTimeOut.count() << ", i.e. you will continue to get warnings, but code will not shut down"
172 );
173 }
174
175 if (
176 _timeOutWarning<_deadlockTimeOut/2
177 or
178 _deadlockTimeOut.count()<=0
179 ) {
180 std::chrono::seconds newTimeOutWarning = _timeOutWarning*2;
181 logWarning(
182 "writeTimeOutWarning(...)",
183 "increase time out warning threshold from " << std::to_string(_timeOutWarning.count()) << " seconds to " << std::to_string(newTimeOutWarning.count()) << " seconds to avoid flood of warning messages"
184 );
185 _timeOutWarning = newTimeOutWarning;
186 _globalTimeOutWarning = _globalTimeOutWarning+_timeOutWarning;
187 }

210
211 int resultlen;
212 char* string = new char[MPI_MAX_ERROR_STRING]; // (char *)malloc(MPI_MAX_ERROR_STRING * sizeof(char));

216 MPI_Error_class(result, &errorclass);
217
218 out << "mpi error class: " << errorclass << "="

221 switch ( errorclass ) {

223 case MPI_ERR_BUFFER: out << "MPI_ERR_BUFFER [invalid buffer pointer]"; break;

228 case MPI_ERR_RANK: out << "MPI_ERR_RANK [invalid rank]"; break;
229 case MPI_ERR_REQUEST: out << "MPI_ERR_REQUEST [invalid request handle]"; break;
230 case MPI_ERR_ROOT: out << "MPI_ERR_ROOT [invalid root argument]"; break;
231 case MPI_ERR_GROUP: out << "MPI_ERR_GROUP [invalid group]"; break;
232 case MPI_ERR_OP: out << "MPI_ERR_OP [invalid operation]"; break;
233 case MPI_ERR_TOPOLOGY: out << "MPI_ERR_TOPOLOGY [invalid topology]"; break;
