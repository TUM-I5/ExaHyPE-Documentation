17 logTraceInWith7Arguments("rollOver", vertex.getValue(), vertex.getRhs(), vertex.getNumberOfRestrictions(), vertex.getNewRestrictedRhs(), vertex.getDelta(), restrictRhsAndInjectSolution, fas);
18
19 /*
20 About to start CG. So, let's reset value, and send whatever is
21 in newRestrictedRhs into rhs. This should have been set during
22 the final stage of the DG Solver (in the coupling action set).
23 We can also reset delta and newRestrictedRhs.
24 */
25 if ( restrictRhsAndInjectSolution ) {
26 vertex.setU(0.0);
27 vertex.setOldU( 0.0 );
28 vertex.setDelta(0.0);
29 vertex.setRhs( vertex.getNewRestrictedRhs() );
30 }
31
32 // Ditto but for FAS
33 else if ( restrictRhsAndInjectSolution and fas ) {
34 auto rescaledNewSolution =
35 1.0 / vertex.getNumberOfRestrictions()
36 *
37 vertex.getSumOfInjectedValues();
38
39 vertex.setU(
40 rescaledNewSolution
41 );
42 vertex.setOldU(
43 rescaledNewSolution
44 );
45 vertex.setRhs(
46 vertex.getNewRestrictedRhs()
47 );
48 }
49
50 /*
51 About to start DG. We have just finished CG and we are
52 interpolating back up to DG. So, we need to set delta
53 so that we have something to interpolate. For now,
54 we just take whatever we find in the value field. We
55 can set all the other vertex fields to 0.0.
56 */
57 else {
58 vertex.setDelta(
59 vertex.getU() - vertex.getOldU()
60 );
61 vertex.setU(0.0);
62 vertex.setRhs( 0.0 );
63 }
64
65 vertex.setSumOfInjectedValues( 0.0 );
66 vertex.setNumberOfRestrictions( 0 );
67 vertex.setNewRestrictedRhs( 0.0 );
68 logTraceOutWith5Arguments("rollOver", vertex.getU(), vertex.getRhs(), vertex.getNumberOfRestrictions(), vertex.getNewRestrictedRhs(), vertex.getDelta());
69}
