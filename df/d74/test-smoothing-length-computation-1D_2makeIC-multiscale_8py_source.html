70def get_coords(box_min, box_max, npart):
71 """
72 Generate coordinates for particles between box_min and box_max
73 """
74
75 dx = (box_max - box_min) / npart
76
77 new_coords = np.zeros((npart, 3))
78 # Get uniform coordinates.
79 for i in range(npart):
80 new_coords[i, 0] = box_min + (i + 0.5) * dx
81
82 # Now randomize them a bit.
83 # Taking a completely randomly sampled particle distribution may lead
84 # to trouble the codes aren't set up to handle, such as particles which
85 # are either too close or too far from each other. So use a somewhat
86 # regular distribution instead, by randomly displacing an initially
87 # uniform particle distribution.
88
89 # rng.random returns random numbers in [0, 1). Scale that.
90 displacement = (2.0 * rng.random((npart, 1)) - 1.0) * max_displacement * dx
91 new_coords[:, 0] += displacement[:, 0]
92
93 return new_coords
94

101coords[npart : 2 * npart, 0] = new_coords[:, 0]
