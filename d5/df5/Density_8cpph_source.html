128template <typename Particle>
129void swift2::kernels::legacy::density_kernel(
130 Particle* localParticle,
131 const Particle* activeParticle
132) {
133 // macro expansions from swift2/legacy/kernels are cumbersome otherwise.
134 using namespace swift2::kernels::legacy::kernelHydro;
135
136 const double hydroDimensions = localParticle->getHydroDimensions();
137
138 /* Retrieve basic properties of local and active particles */
139 const double hi = localParticle->getSmoothingLength();
140
141 /* Distance between the particles */
142 const tarch::la::Vector<Dimensions, double>
143 dx = localParticle->getX() - activeParticle->getX();
144 const double r = tarch::la::norm2(dx);
145
146 const double mj = activeParticle->getMass();
147 const double hi_inv = 1. / hi;
148 // Normalized distance to evaluate kernel
149 const double qi = r * hi_inv;
150
151 // Evaluate kernel and its derivative
152 double wi, dwi_dx;
153 kernel_deval(qi, wi, dwi_dx);
154 const double dwi_dh = -(hydroDimensions * wi + qi * dwi_dx);
155
156 // Increment density
157 localParticle->setDensity(localParticle->getDensity() + mj * wi);
158
159 // Increment drho/dh. Note that minus sign was absorbed into dwi_dh
160 localParticle->setRho_dh(localParticle->getRho_dh() + mj * dwi_dh);
161
162 // Increment fractional number density
163 localParticle->setWcount(localParticle->getWcount() + wi);
164
165 // Increment dWcount/dh. Note that minus sign was absorbed into dwi_dh
166 localParticle->setWcount_dh(localParticle->getWcount_dh() + dwi_dh);
167
168
169 // Compute dv dot r
170 // Velocity terms for viscosity
171 tarch::la::Vector<Dimensions, double> dv;
172
173 /* Now we need to compute the div terms */
174 const double r_inv = r ? 1.0 / r : 0.0;
175 const double fac_i = mj * dwi_dx * r_inv;
176
177 dv = localParticle->getV() - activeParticle->getV();
178 const double dvdr = tarch::la::dot(dv, dx);
179 localParticle->setDiv_v(localParticle->getDiv_v() - fac_i * dvdr);
180
181 /* Compute dv cross r */
182 /* 2D Term only has 1 component (z) */
183#if Dimensions < 3
184 const double curlvr2D = dv[0] * dx[1] - dv[1] * dx[0];
185 localParticle->setRot_v(localParticle->getRot_v() + fac_i * curlvr2D);
186#else
187 const tarch::la::Vector<3, double> curlvr = {
188 dv[1] * dx[2] - dv[2] * dx[1],
189 dv[2] * dx[0] - dv[0] * dx[2],
190 dv[0] * dx[1] - dv[1] * dx[0]};
191 localParticle->setRot_v(localParticle->getRot_v() + fac_i * curlvr);
192#endif
193
194#if PeanoDebug > 0
195 localParticle->setDensityNeighbourCount(
