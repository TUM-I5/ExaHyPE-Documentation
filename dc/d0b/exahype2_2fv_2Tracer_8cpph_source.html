185template <int SourceIndex, int... SourceIndices, typename Particle>
186void exahype2::fv::updateParticleState(
187 Particle& particle,
188 double timeStepSize,
189 const peano4::datamanagement::CellMarker& marker,
190 int voxelsPerAxis,
191 int unknownsPerVoxel,
192 const double* __restrict__ Q,
193 tarch::la::Vector<Dimensions, double> DomainSize,
194 tarch::la::Vector<Dimensions, double> DomainOffset,
195 tarch::la::Vector<Dimensions, double> PeriodicBC
196) {
197 // 🔹 Step 1: Interpolation (Store values directly in the particle)
198 projectValuesOntoParticle_piecewiseLinear<
199 SourceIndex,
200 SourceIndices...>(marker, voxelsPerAxis, unknownsPerVoxel, Q, particle);
201
202 // 🔹 Step 2: Compute Forces (e.g., gravity, drag) - To be implemented
203 // Example: Applying only gravity for now
204 particle.setData(5, 0.0);
205 particle.setData(6, -9.81);
206
207 // 🔹 Step 3: Perform Integration
208 tarch::la::Vector<Dimensions, double> newPosition;
209
210 newPosition = explicitEulerIntegration(particle, timeStepSize);
211
212 // 🔹 Step 4: Handle Boundary Collisions (if periodic BC is not active)
213 if (tarch::la::max(PeriodicBC) == 1.0) {
214 handleBoundaryCollisions(particle, newPosition, DomainSize, DomainOffset);
215 }
216
217 // 🔹 Step 5: Update Particle Position
218 particle.setX(newPosition);
219}

221template <typename Particle>
222tarch::la::Vector<Dimensions, double> exahype2::fv::explicitEulerIntegration(Particle& particle, double timeStepSize) {
223 tarch::la::Vector<Dimensions, double> newPosition;
224 // Set initial acceleration for testing purposes
225 for (int d = 0; d < Dimensions; ++d) {
226 double velocity = particle.getData(3 + d);
227 double acceleration = particle.getData(5 + d);
228 velocity += acceleration * timeStepSize; // Update velocity
229 velocity = std::min(std::max(velocity, -10.0), 10.0);
230
231 particle.setData(3 + d, velocity);
232
233 newPosition[d] = particle.getX(d) + velocity * timeStepSize + 0.5 * acceleration * timeStepSize * timeStepSize;
234 }
235
236 return newPosition;
237}

239template <typename Particle>
240void exahype2::fv::handleBoundaryCollisions(
241 Particle& particle,
242 tarch::la::Vector<Dimensions, double>& newPosition,
243 tarch::la::Vector<Dimensions, double> DomainSize,
244 tarch::la::Vector<Dimensions, double> DomainOffset
245) {
246 for (int d = 0; d < Dimensions; d++) {
247 double lowerBound = DomainOffset[d];
248 double upperBound = DomainOffset[d] + DomainSize[d];
249
250 double velocity = particle.getData(3 + d);
251
252 if (newPosition[d] < lowerBound) {
253 // Reflect at lower boundary
254 newPosition[d] = lowerBound + (lowerBound - newPosition[d]);
255 particle.setData(3 + d, -velocity * 0.9); // Reverse & damp velocity
256
257 } else if (newPosition[d] > upperBound) {
258 // Reflect at upper boundary
259 newPosition[d] = upperBound - (newPosition[d] - upperBound);
260 particle.setData(3 + d, -velocity * 0.9); // Reverse & damp velocity
261 }
262 }
