101Block Block::resample(std::function<void(Block const&,
102 int,int,int,
103 Block&)> func,
104 int in_z , int in_y , int in_x,
105 int out_z, int out_y, int out_x) const{
106
107 if((index_size[0]-1) % (in_z-1) != 0 ||
108 (index_size[1]-1) % (in_y-1) != 0 ||
109 (index_size[2]-1) % (in_x-1) != 0){
110 std::stringstream ss;
111 ss << "Resampling with wrong modes: " << std::endl;
112 ss << "Modes: " << std::endl;
113 ss << in_z << " " << in_y << " " << in_x << std::endl;
114 ss << "IndexSizes: " << std::endl;
115 ss << index_size[0] << " " << index_size[1] << " " << index_size[2] << std::endl;
116 throw std::domain_error( ss.str());
117 }
118
119 /*the number of implicit elements*/
120 int elements_z = (index_size[0]-1) / (in_z-1);
121 int elements_y = (index_size[1]-1) / (in_y-1);
122 int elements_x = (index_size[2]-1) / (in_x-1);
123
124 /*the firest index of implicit elements*/
125 int start_elements_z = (index_offset[0]) / (in_z-1);
126 int start_elements_y = (index_offset[1]) / (in_y-1);
127 int start_elements_x = (index_offset[2]) / (in_x-1);
128
129 Coordinate coords[3]; getCoord(coords);
130
131 int i_indexSize [3] = {elements_z * (out_z-1) + 1,
132 elements_y * (out_y-1) + 1,
133 elements_x * (out_x-1) + 1};
134
135 int i_indexOffset[3] = {start_elements_z * (out_z-1),
136 start_elements_y * (out_y-1),
137 start_elements_x * (out_x-1)};
138
139 std::vector<double> result_nodes;
140
141 result_nodes.resize(i_indexSize[2] *
142 i_indexSize[1] *
143 i_indexSize[0] * NCOORDS);
144
145 double position[3];
146 idx4 id(NCOORDS,nodes.size(),nodes.size(),nodes.size());
147 Block result(coords,i_indexOffset,i_indexSize,result_nodes);
148
149
150 for(int e_z = 0; e_z < elements_z; e_z++){
151 for(int e_y = 0; e_y < elements_y; e_y++){
152 for(int e_x = 0; e_x < elements_x; e_x++){
153 //std::cout << "e_z: " << e_z<< "e_y :" << e_y<< "e_x :" << e_x << std::endl;
154 func((*this),start_elements_z + e_z,
155 start_elements_y + e_y,
156 start_elements_x + e_x,
157 result);
158 }
159 }
160 }
161 return result;
162}

185void Block::interpolate(const double node[3], double result[3] , int strip) const{
186 std::vector<int> ranges[3];
187 if (strip % 2 == 1) strip++;
188 for (Coordinate c : coords){
189 int range_min = std::min(std::max(index_offset[c],int(std::ceil(node[c])-strip/2)),
190 index_offset[c]+index_size[c] - strip);
191 int range_max = strip + range_min - 1;
192 getIndexRange(c,range_min,range_max + 1,ranges[c]);
193 }
194 Interpolation::Lagrange::interpolate3D(node,(*this),ranges[0],ranges[1],ranges[2],result);
195}

216Block operator+(Block lhs, const Block& rhs){
217 Coordinate coords_left[3],coords_right[3];
218 lhs.getCoord(coords_left); rhs.getCoord(coords_right);
219
220 Coordinate coords[3];
221 std::copy_n(coords_left,3,coords);
222
223 int new_index_offset[3];
224 int new_index_size [3];
225
226 /*find fitting coordinates*/
227 int r_index[3];
228 r_index[0] =
229 coords_right[0] == coords_left[0] ? 0 :
230 coords_right[1] == coords_left[0] ? 1 : 2;
231
232 r_index[1] =
233 coords_right[0] == coords_left[1] ? 0 :
234 coords_right[1] == coords_left[1] ? 1 : 2;
235
236 r_index[2] = 3 - r_index[0] - r_index[1];
237 /**************************/
238
239 //*******Check if merge is possible **********//
240 int index_offset_left[3],index_offset_right[3];
241 lhs.getIndexOffset(index_offset_left); rhs.getIndexOffset(index_offset_right);
242
243 //******* For the offset **********//
244 int same_offset = 0;
245 int const_coord_1 = -1;
246 int const_coord_2 = -1;
247 int merge_at_coord = -1; //the coordiante we will merge along
248 if(index_offset_left[0] == index_offset_right[r_index[0]]){
249 same_offset++;
250 new_index_offset[0] = index_offset_left[0];
251 const_coord_1 = 0;
252 }else{
253 merge_at_coord = 0;
254 }
255
256 if(index_offset_left[1] == index_offset_right[r_index[1] ]){
257 same_offset++;
258 new_index_offset[1] = index_offset_left[1];
259 const_coord_1 = const_coord_1 == -1 ? 1 : const_coord_1;
260 const_coord_2 = const_coord_2 == -1 ? 1 : const_coord_2;
261 }else{
262 merge_at_coord = 1;
263 }
264
265 if(index_offset_left[2] == index_offset_right[r_index[2] ]){
266 same_offset++;
267 new_index_offset[2] = index_offset_left[2];
268 const_coord_2 = 2;
269 }else{
270 merge_at_coord = 2;
271 }
272
273 if(same_offset != 2){
274 std::stringstream ss;
275 ss << "Trying to merge non fitting bocks (offset)" << std::endl;
276 ss << "lhs" << std::endl;
277 ss << lhs.toString() << std::endl;
278 ss << "rhs" << std::endl;
279 ss << rhs.toString() << std::endl;
280 throw std::domain_error(ss.str());
281 }
282
283 //******* For the size **********//
284
285 int index_size_left[3],index_size_right[3];
286 lhs.getIndexSize(index_size_left); rhs.getIndexSize(index_size_right);
287
288 int same_size = 0;
289
290 if(merge_at_coord != 0 && index_size_left[0] == index_size_right[r_index[0]]){
291 same_size++;
292 new_index_size[0] = index_size_left[0];
293 }
294
295 if(merge_at_coord != 1 && index_size_left[1] == index_size_right[r_index[1]]){
296 same_size++;
297 new_index_size[1] = index_size_left[1];
298 }
299
300 if(merge_at_coord != 2 && index_size_left[2] == index_size_right[r_index[2]]){
301 same_size++;
302 new_index_size[2] = index_size_left[2];
303 }
304
305 if(same_size != 2){
306 std::stringstream ss;
307 ss << "Trying to merge non fitting bocks (size)" << std::endl;
308 ss << "lhs" << std::endl;
309 ss << lhs.toString() << std::endl;
310 ss << "rhs" << std::endl;
311 ss << rhs.toString() << std::endl;
312 throw std::domain_error(ss.str());
313 }
314
315 new_index_size[merge_at_coord]=
316 index_size_left[merge_at_coord] + index_size_right[r_index[merge_at_coord]] - 1 ;
317
318 std::vector<double> new_nodes;
319 new_nodes.resize(new_index_size[0]*new_index_size[1]*new_index_size[2]*NCOORDS);
320
321 /*****************************/
322
323 //******* Merge blocks **********//
324 Coordinate c1 = static_cast<Coordinate>(const_coord_1 );
325 Coordinate c2 = static_cast<Coordinate>(const_coord_2 );
326 Coordinate m = static_cast<Coordinate>(merge_at_coord);
327
328 /*Decide which block comes first*/
329 /*left comes first*/
330 if (index_offset_left[m] < index_offset_right[r_index[m]]){
331 /*Interface Matches*/
332 if(index_offset_left[m] + index_size_left[m] - 1
333 == index_offset_right[r_index[m]]){
334 new_index_offset[m] = index_offset_left[m];
335 }else{
336 throw std::domain_error("Interfaces don't fit for block merge");
337 }
338 /*right comes first*/
339 }else{
340 /*Interface Matches*/
341 if(index_offset_left[m]
342 ==index_offset_right[r_index[m]]+index_size_right[r_index[m]]-1){
343 new_index_offset[m] = index_offset_right[r_index[m]];
344 }else{
345 throw std::domain_error("Interfaces don't fit for block merge");
346 }
347 }
348
349 Block result(coords,new_index_offset,new_index_size,new_nodes);
350 std::vector<int> range_c1; lhs.getIndexRange(c1,range_c1);
351 std::vector<int> range_c2; lhs.getIndexRange(c2,range_c2);
352 std::vector<int> range_ml; lhs.getIndexRange(m ,range_ml);
353 std::vector<int> range_mr; rhs.getIndexRange(m ,range_mr);
354
355 for (int i : range_c1 ){
356 for (int j : range_c2){
357 for (int k : range_ml){
358 for (int d = 0 ; d < NCOORDS; d++){
359 result(d, c1 ,i, c2 ,j, m,k) = lhs (d, c1 ,i, c2 ,j, m,k);
360 }
361 }
362 for (int k : range_mr){
363 for (int d = 0 ; d < NCOORDS; d++){
364 result(d, c1 ,i, c2 ,j, m, k) = rhs (d, c1 ,i, c2 ,j, m, k);
365 }
366 }
367 }
368 }
369 /******************************/
370 return result;
371}
