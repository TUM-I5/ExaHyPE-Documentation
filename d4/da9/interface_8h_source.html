60 double invertProjection(Coordinate coord, double coords[3]){
61
62 Coordinate mu = coord;
63 Coordinate xi,eta;
64 getPlane(mu,xi,eta);
65
66 BlockBoundary boundary = root->traverseProjection(coords);
67
68 Face upper = boundary.getFace(mu, true);
69 Face lower = boundary.getFace(mu, false);
70
71
72 std::vector<int> range_xi; lower.getIndexRange(xi,range_xi);
73 std::vector<int> range_eta;lower.getIndexRange(eta,range_eta);
74 double offset_xi = lower( xi,xi,range_xi.front(),eta,range_eta.front());
75 double offset_eta = lower(eta,xi,range_xi.front(),eta,range_eta.front());
76 double size_xi = lower( xi,xi,range_xi.back() ,eta,range_eta.front()) - offset_xi;
77 double size_eta = lower( eta,xi,range_xi.front(),eta,range_eta.back()) - offset_eta;
78
79
80 double index_xi = (coords[xi] - offset_xi ) / size_xi * (range_xi.back() - range_xi.front()) + range_xi.front();
81 double index_eta = (coords[eta] - offset_eta) / size_eta * (range_eta.back() - range_eta.front()) + range_eta.front();
82
83 int index_xi_l = std::floor(index_xi);
84 int index_eta_l = std::floor(index_eta);
85 int index_xi_c = std::ceil(index_xi);
86 int index_eta_c = std::ceil(index_eta);
87
88 //interpolate
89 //lagrange coefficients
90 double l_xi_c = (index_xi - index_xi_l) / double(index_xi_c - index_xi_l);
91 double l_xi_l = (index_xi - index_xi_c) / double(index_xi_l - index_xi_c);
92 double l_eta_c = (index_eta - index_eta_l) / double(index_eta_c - index_eta_l);
93 double l_eta_l = (index_eta - index_eta_c) / double(index_eta_l - index_eta_c);
94
95 double mu_u = l_xi_c * l_eta_c * upper(mu,xi,index_xi_c,eta,index_eta_c) +
96 l_xi_c * l_eta_l * upper(mu,xi,index_xi_c,eta,index_eta_l) +
97 l_xi_l * l_eta_c * upper(mu,xi,index_xi_l,eta,index_eta_c) +
98 l_xi_l * l_eta_l * upper(mu,xi,index_xi_l,eta,index_eta_l);
99
100 double mu_l = l_xi_c * l_eta_c * lower(mu,xi,index_xi_c,eta,index_eta_c) +
101 l_xi_c * l_eta_l * lower(mu,xi,index_xi_c,eta,index_eta_l) +
102 l_xi_l * l_eta_c * lower(mu,xi,index_xi_l,eta,index_eta_c) +
103 l_xi_l * l_eta_l * lower(mu,xi,index_xi_l,eta,index_eta_l);
104
105 double rel_pos_mu;
106 rel_pos_mu = (coords[mu] - mu_l)/ (mu_u-mu_l);
107 double rel_index = (rel_pos_mu * boundary.getIndexSize(mu) + boundary.getIndexOffset(mu))/grid[mu];
108 return rel_index * size[mu] + offset[mu];
109 }
