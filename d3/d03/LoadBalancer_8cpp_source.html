77void toolbox::loadbalancing::LoadBalancer::updateInterRankLoadBalancing() {
78 const int numberOfRanks = tarch::mpi::Rank::getInstance().getNumberOfRanks();
79 if (numberOfRanks <= 1 or _statistics.getGlobalNumberOfTrees() > 1) {
80 _state = State::IntraRankDistribution;
81 return;
82 }
83
84 if (hasSplitRecently() or not _statistics.hasConsistentViewOfWorld()
85 or not tarch::mpi::Rank::getInstance().isGlobalMaster()) {
86 return; // Nothing to do yet
87 }
88
89 const int numberOfCells = _statistics.getGlobalNumberOfInnerUnrefinedCells();
90 const int minTreeSize = _configuration->getMinTreeSize(_state);
91 const int numberOfTreesToAccommodateMinTreeSize = std::max(numberOfCells / minTreeSize, 1);
92 if (numberOfTreesToAccommodateMinTreeSize < numberOfRanks) {
93 return; // Nothing to do (yet)
94 }
95
96 const int numberOfCellsPerTree = std::max(static_cast<int>(std::round(numberOfCells / numberOfRanks)), 1);
97 logDebug(
98 "updateInterRankLoadBalancing()",
99 "Mesh has "
100 << numberOfCells << " cells, so split " << numberOfRanks - 1 << " times "
101 << "("
102 << "MinTreeSize=" << minTreeSize << ", NumberOfTreesToAccommodateMinTreeSize="
103 << numberOfTreesToAccommodateMinTreeSize << ", NumberOfCellsPerTree=" << numberOfCellsPerTree << ")"
104 );
105
106 for (int targetRank = 1; targetRank < numberOfRanks; targetRank++) {
107 int numberOfCellsPerTreeWithRemainder = numberOfCellsPerTree;
108 if (numberOfCells % numberOfRanks >= targetRank) {
109 numberOfCellsPerTreeWithRemainder++;
110 }
111 triggerSplit(numberOfCellsPerTreeWithRemainder, targetRank);
112 }
113}

115void toolbox::loadbalancing::LoadBalancer::updateIntraRankLoadBalancing() {
116 const int numberOfThreads = tarch::multicore::Core::getInstance().getNumberOfThreads();
117 const int maxPermittedTreesPerRank = _configuration->getMaxLocalTreesPerRank(_state);
118 if (numberOfThreads <= 1 or _statistics.getLocalNumberOfTrees() >= maxPermittedTreesPerRank) {
119 _state = State::Stagnation;
120 return;
121 }
122
123 const int minTreeSize = _configuration->getMinTreeSize(_state);
124 const int numberOfCells = _statistics.getLocalNumberOfInnerUnrefinedCells();
125 const int numberOfTreesToAccommodateMinTreeSize = std::max(numberOfCells / minTreeSize, 1);
126 const int numberOfAllowedTrees = std::min({numberOfTreesToAccommodateMinTreeSize, maxPermittedTreesPerRank, numberOfThreads});
127 if (numberOfTreesToAccommodateMinTreeSize < numberOfThreads or not _statistics.hasConsistentViewOfWorld()) {
128 return; // Nothing to do (yet)
129 }
130
131 unsigned int numberOfSplits = numberOfAllowedTrees - 1;
132 if (numberOfSplits > 0) {
133 const int numberOfCellsPerTree = std::max(static_cast<int>(std::round(numberOfCells / numberOfAllowedTrees)), 1);
134 logDebug(
135 "updateIntraRankLoadBalancing()",
136 "Mesh has "
137 << numberOfCells << " cells, so split " << numberOfSplits << " times "
138 << "("
139 << "MinTreeSize=" << minTreeSize << ", MaxPermittedTreesPerRank=" << maxPermittedTreesPerRank
140 << ", NumberOfTreesToAccommodateMinTreeSize=" << numberOfTreesToAccommodateMinTreeSize
141 << ", NumberOfCellsPerTree=" << numberOfCellsPerTree << ")"
142 );
143 unsigned int remainder = numberOfCells - (numberOfCellsPerTree * numberOfAllowedTrees);
144 while (numberOfSplits) {
145 if (remainder > 0) {
146 triggerSplit(numberOfCellsPerTree + 1, tarch::mpi::Rank::getInstance().getRank());
147 remainder--;
148 } else {
149 triggerSplit(numberOfCellsPerTree, tarch::mpi::Rank::getInstance().getRank());
150 }
151 numberOfSplits--;
152 }
153 }
154}

156void toolbox::loadbalancing::LoadBalancer::triggerSplit(int numberOfCells, int targetRank) {
157 const int sourceTree = *(peano4::parallel::SpacetreeSet::getInstance().getLocalSpacetrees().begin());
158 const bool success = peano4::parallel::SpacetreeSet::getInstance().split(
159 sourceTree,
160 peano4::SplitInstruction{numberOfCells, _configuration->getMode(_state)},
161 targetRank
162 );
163
164 if (not success) {
165 logError("triggerSplit()", "Wanted to split tree " << sourceTree << " but failed");
166 }
167
168 _blacklist.triggeredSplit(sourceTree);
169 _statistics.incLocalNumberOfSplits();
170}

179std::string toolbox::loadbalancing::LoadBalancer::getStatistics() {
180 std::ostringstream msg;
181
182 int sumLocal = 0;
183 int sumRemote = 0;
184 std::set<int> idsOfLocalSpacetrees = peano4::parallel::SpacetreeSet::getInstance().getLocalSpacetrees();
185
186 msg << idsOfLocalSpacetrees.size() << " tree(s): ";
187
188 for (auto p : idsOfLocalSpacetrees) {
189 msg
190 << "(#" << p << ":"
191 << peano4::parallel::SpacetreeSet::getInstance().getGridStatistics(p).getNumberOfLocalUnrefinedCells() << "/"
192 << peano4::parallel::SpacetreeSet::getInstance().getGridStatistics(p).getNumberOfRemoteUnrefinedCells() << ")";
193 sumLocal += peano4::parallel::SpacetreeSet::getInstance().getGridStatistics(p).getNumberOfLocalUnrefinedCells();
194 sumRemote += peano4::parallel::SpacetreeSet::getInstance().getGridStatistics(p).getNumberOfRemoteUnrefinedCells();
195 }
196 if (idsOfLocalSpacetrees.empty()) {
197 msg << "no trees hosted on this node - therefore no load";
198 } else {
199 msg << " total=" << sumLocal << "/" << sumRemote << " (local/virtual)";
200 }
201
202 return msg.str();
