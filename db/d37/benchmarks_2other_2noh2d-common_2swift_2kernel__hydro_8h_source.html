257__attribute__((always_inline)) INLINE static void kernel_deval(
258 float u, float *restrict W, float *restrict dW_dx) {
259
260 /* Go to the range [0,1[ from [0,H[ */
261 const float x = u * kernel_gamma_inv;
262
263 /* Pick the correct branch of the kernel */
264 const int temp = (int)(x * kernel_ivals_f);
265 const int ind = temp > kernel_ivals ? kernel_ivals : temp;
266 const float *const coeffs = &kernel_coeffs[ind * (kernel_degree + 1)];
267
268 /* First two terms of the polynomial ... */
269 float w = coeffs[0] * x + coeffs[1];
270 float dw_dx = coeffs[0];
271
272 /* ... and the rest of them */
273 for (int k = 2; k <= kernel_degree; k++) {
274 dw_dx = dw_dx * x + w;
275 w = x * w + coeffs[k];
276 }
277
278 w = MAX(w, 0.f);
279 dw_dx = MIN(dw_dx, 0.f);
280
281 /* Return everything */
282 *W = w * kernel_constant * kernel_gamma_inv_dim;
283 *dW_dx = dw_dx * kernel_constant * kernel_gamma_inv_dim_plus_one;
284}

369__attribute__((always_inline)) INLINE static void kernel_eval_dWdx(
370 float u, float *restrict dW_dx) {
371
372 /* Go to the range [0,1[ from [0,H[ */
373 const float x = u * kernel_gamma_inv;
374
375 /* Pick the correct branch of the kernel */
376 const int temp = (int)(x * kernel_ivals_f);
377 const int ind = temp > kernel_ivals ? kernel_ivals : temp;
378 const float *const coeffs = &kernel_coeffs[ind * (kernel_degree + 1)];
379
380 /* First two terms of the polynomial ... */
381 float dw_dx = ((float)kernel_degree * coeffs[0] * x) +
382 (float)(kernel_degree - 1) * coeffs[1];
383
384 /* ... and the rest of them */
385 for (int k = 2; k < kernel_degree; k++)
386 dw_dx = dw_dx * x + (float)(kernel_degree - k) * coeffs[k];
387
388 dw_dx = MIN(dw_dx, 0.f);
389
390 /* Return everything */
391 *dW_dx = dw_dx * kernel_constant * kernel_gamma_inv_dim_plus_one;
392}

407__attribute__((always_inline, const)) INLINE static float potential_dh(
408 const float u, const float h_inv) {
409
410 /* Ratio of r to kernel support
411 * Recall that in our definition the kernel edge is H = kernel_gamma * h. */
412 const float q = fminf(u * kernel_gamma_inv, 1.f);
413
414 /* -24 q^7 + 105 q^6 -168 q^5 + 105 q^4 - 21 q^2 */
415 float dphi_dh = -24.f * q + 105.f;
416 dphi_dh = dphi_dh * q - 168.f;
417 dphi_dh = dphi_dh * q + 105.f;
418 dphi_dh = dphi_dh * q;
419 dphi_dh = dphi_dh * q - 21.f;
420 dphi_dh = dphi_dh * q;
421 dphi_dh = dphi_dh * q;
422
423 return dphi_dh * h_inv * h_inv * kernel_gamma_inv * 0.25f;
424}

441__attribute__((always_inline)) INLINE static void kernel_deval_vec(
442 vector *u, vector *w, vector *dw_dx) {
443
444 /* Go to the range [0,1[ from [0,H[ */
445 vector x;
446 x.v = vec_mul(u->v, kernel_gamma_inv_vec.v);
447
448 /* Load x and get the interval id. */
449 vector ind;
450 ind.m =
451 vec_ftoi(vec_fmin(vec_mul(x.v, kernel_ivals_vec.v), kernel_ivals_vec.v));
452
453 /* load the coefficients. */
454 vector c[kernel_degree + 1];
455 for (int k = 0; k < VEC_SIZE; k++)
456 for (int j = 0; j < kernel_degree + 1; j++)
457 c[j].f[k] = kernel_coeffs[ind.i[k] * (kernel_degree + 1) + j];
458
459 /* Init the iteration for Horner's scheme. */
460 w->v = vec_fma(c[0].v, x.v, c[1].v);
461 dw_dx->v = c[0].v;
462
463 /* And we're off! */
464 for (int k = 2; k <= kernel_degree; k++) {
465 dw_dx->v = vec_fma(dw_dx->v, x.v, w->v);
466 w->v = vec_fma(x.v, w->v, c[k].v);
467 }
468
469 /* Return everything */
470 w->v =
471 vec_mul(w->v, vec_mul(kernel_constant_vec.v, kernel_gamma_inv_dim_vec.v));
472 dw_dx->v = vec_mul(dw_dx->v, vec_mul(kernel_constant_vec.v,
473 kernel_gamma_inv_dim_plus_one_vec.v));
474}

534__attribute__((always_inline)) INLINE static void kernel_deval_1_vec(
535 vector *u, vector *w, vector *dw_dx) {
536
537 /* Go to the range [0,1[ from [0,H[ */
538 vector x;
539 x.v = vec_mul(u->v, kernel_gamma_inv_vec.v);
540
541#ifdef WENDLAND_C2_KERNEL
542 /* Init the iteration for Horner's scheme. */
543 w->v = vec_fma(wendland_const_c0.v, x.v, wendland_const_c1.v);
544 dw_dx->v = wendland_const_c0.v;
545
546 /* Calculate the polynomial interleaving vector operations */
547 dw_dx->v = vec_fma(dw_dx->v, x.v, w->v);
548 w->v = vec_fma(x.v, w->v, wendland_const_c2.v);
549
550 dw_dx->v = vec_fma(dw_dx->v, x.v, w->v);
551 w->v = vec_fma(x.v, w->v, wendland_const_c3.v);
552
553 dw_dx->v = vec_fma(dw_dx->v, x.v, w->v);
554 w->v = vec_mul(x.v, w->v); /* wendland_const_c4 is zero. */
555
556 dw_dx->v = vec_fma(dw_dx->v, x.v, w->v);
557 w->v = vec_fma(x.v, w->v, wendland_const_c5.v);
558#elif defined(CUBIC_SPLINE_KERNEL)
559 vector w2, dw_dx2;
560 mask_t mask_reg;
561
562 /* Form a mask for one part of the kernel. */
563 /* Only need the mask for one region as the vec_blend defaults to the vector
564 * when the mask is 0.*/
565 vec_create_mask(mask_reg, vec_cmp_gte(x.v, cond.v)); /* 0.5 < x < 1 */
566
567 /* Work out w for both regions of the kernel and combine the results together
568 * using a mask. */
569
570 /* Init the iteration for Horner's scheme. */
571 w->v = vec_fma(cubic_1_const_c0.v, x.v, cubic_1_const_c1.v);
572 w2.v = vec_fma(cubic_2_const_c0.v, x.v, cubic_2_const_c1.v);
573 dw_dx->v = cubic_1_const_c0.v;
574 dw_dx2.v = cubic_2_const_c0.v;
575
576 /* Calculate the polynomial interleaving vector operations. */
577 dw_dx->v = vec_fma(dw_dx->v, x.v, w->v);
578 dw_dx2.v = vec_fma(dw_dx2.v, x.v, w2.v);
579 w->v = vec_mul(x.v, w->v); /* cubic_1_const_c2 is zero. */
580 w2.v = vec_fma(x.v, w2.v, cubic_2_const_c2.v);
581
582 dw_dx->v = vec_fma(dw_dx->v, x.v, w->v);
583 dw_dx2.v = vec_fma(dw_dx2.v, x.v, w2.v);
584 w->v = vec_fma(x.v, w->v, cubic_1_const_c3.v);
585 w2.v = vec_fma(x.v, w2.v, cubic_2_const_c3.v);
586
587 /* Blend both kernel regions into one vector (mask out unneeded values). */
588 /* Only need the mask for one region as the vec_blend defaults to the vector
589 * when the mask is 0.*/
590 w->v = vec_blend(mask_reg, w->v, w2.v);
591 dw_dx->v = vec_blend(mask_reg, dw_dx->v, dw_dx2.v);
592
593#else
594#error \
595 "Vectorisation not supported for this kernel!!! Choose a different one or configure with --disable-hand-vec."
596#endif
597
598 /* Return everyting */
599 w->v =
