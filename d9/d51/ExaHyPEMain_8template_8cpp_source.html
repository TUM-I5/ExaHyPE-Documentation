73bool selectNextAlgorithmicStep() {
74 static bool gridConstructed = false;
75 static bool gridInitialised = false;
76 static double nextMaxPlotTimeStamp = {{SUBNAMESPACE}}FirstPlotTimeStamp;
77 static double nextMinPlotTimeStamp = {{SUBNAMESPACE}}FirstPlotTimeStamp;
78 static double nextMaxCheckpointTimeStamp = {{SUBNAMESPACE}}FirstCheckpointTimeStamp;
79 static double nextMinCheckpointTimeStamp = {{SUBNAMESPACE}}FirstCheckpointTimeStamp;
80 static bool haveJustWrittenSnapshot = false;
81 static tarch::la::Vector<Dimensions, double> minH = tarch::la::Vector<Dimensions, double>(std::numeric_limits<double>::max());
82 static int globalNumberOfTrees = 0;
83 bool continueToSolve = true;
84
85 if (tarch::hasNonCriticalAssertionBeenViolated() and not haveReceivedNonCriticalAssertion) {
86 peano4::parallel::Node::getInstance().setNextProgramStep(
87 {{SUBNAMESPACE}}repositories::StepRepository::toProgramStep({{SUBNAMESPACE}}repositories::StepRepository::Steps::PlotSolution)
88 );
89 haveReceivedNonCriticalAssertion = true;
90 logError("selectNextAlgorithmicStep()", "Noncritical assertion has been triggered in code. Dump final state and terminate.");
91 } else if (tarch::hasNonCriticalAssertionBeenViolated()) {
92 continueToSolve = false;
93 } else if (gridConstructed and not gridInitialised) {
94 peano4::parallel::Node::getInstance().setNextProgramStep(
95 {{SUBNAMESPACE}}repositories::StepRepository::toProgramStep({{SUBNAMESPACE}}repositories::StepRepository::Steps::InitGrid)
96 );
97 gridInitialised = true;
98 } else if (not gridConstructed) {
99 if (tarch::la::max(peano4::parallel::SpacetreeSet::getInstance().getGridStatistics().getMinH()) < tarch::la::max(minH)) {
100 minH = peano4::parallel::SpacetreeSet::getInstance().getGridStatistics().getMinH();
101 logDebug("selectNextAlgorithmicStep()", "Mesh has refined, so reset minH to " << minH << " and postpone further refinement.");
102 peano4::parallel::Node::getInstance().setNextProgramStep({{SUBNAMESPACE}}repositories::StepRepository::toProgramStep(
103 {{SUBNAMESPACE}}repositories::StepRepository::Steps::CreateGridWithoutRefinement
104 ));
105 } else if ({{SUBNAMESPACE}}repositories::loadBalancer.getGlobalNumberOfTrees() > globalNumberOfTrees) {
106 logDebug("selectNextAlgorithmicStep()", "Mesh has rebalanced recently, so postpone further refinement.");
107 peano4::parallel::Node::getInstance().setNextProgramStep({{SUBNAMESPACE}}repositories::StepRepository::toProgramStep(
108 {{SUBNAMESPACE}}repositories::StepRepository::Steps::CreateGridWithoutRefinement
109 ));
110 globalNumberOfTrees = {{SUBNAMESPACE}}repositories::loadBalancer.getGlobalNumberOfTrees();
111 } else if (
112 peano4::parallel::SpacetreeSet::getInstance().getGridStatistics().getStationarySweeps() > 5
113 and
114 // Ensure that a proper creation has been ran before, so the mesh had the opportunity to refine further if it has not done yet.
115 {{SUBNAMESPACE}}repositories::StepRepository::toStepEnum(peano4::parallel::Node::getInstance().getCurrentProgramStep()) == {{SUBNAMESPACE}}repositories::StepRepository::Steps::CreateGridWithRefinement
116 ) {
117 logDebug("selectNextAlgorithmicStep()", "Grid has been stationary for quite some time. Terminate grid construction.");
118 gridConstructed = true;
119 } else {
120 logDebug("selectNextAlgorithmicStep()", "Mesh rebalancing seems to be stationary, so study whether to refine mesh further in next sweep.");
121 peano4::parallel::Node::getInstance().setNextProgramStep({{SUBNAMESPACE}}repositories::StepRepository::toProgramStep(
122 {{SUBNAMESPACE}}repositories::StepRepository::Steps::CreateGridWithRefinement)
123 );
124 globalNumberOfTrees = {{SUBNAMESPACE}}repositories::loadBalancer.getGlobalNumberOfTrees();
125 }
126 } else {
127 if (
128 {{SUBNAMESPACE}}TimeInBetweenPlots > 0.0
129 and
130 {{SUBNAMESPACE}}repositories::getMinTimeStamp() < {{SUBNAMESPACE}}MinTerminalTime
131 and
132 {{SUBNAMESPACE}}repositories::getMaxTimeStamp() < {{SUBNAMESPACE}}MaxTerminalTime
133 and
134 ({{SUBNAMESPACE}}repositories::getMinTimeStamp() >= nextMinPlotTimeStamp or {{SUBNAMESPACE}}repositories::getMaxTimeStamp() >= nextMaxPlotTimeStamp)
135 and
136 {{SUBNAMESPACE}}repositories::mayPlot()
137 ) {
138 if ({{SUBNAMESPACE}}repositories::getMinTimeStamp() >= nextMinPlotTimeStamp) {
139 nextMinPlotTimeStamp += {{SUBNAMESPACE}}TimeInBetweenPlots;
140 }
141 if ({{SUBNAMESPACE}}repositories::getMaxTimeStamp() >= nextMaxPlotTimeStamp) {
142 nextMaxPlotTimeStamp += {{SUBNAMESPACE}}TimeInBetweenPlots;
143 }
144
145 if (nextMinPlotTimeStamp < {{SUBNAMESPACE}}repositories::getMinTimeStamp()) {
146 logWarning(
147 "selectNextAlgorithmicStep()",
148 "Code is asked to plot every dt="
149 << {{SUBNAMESPACE}}TimeInBetweenPlots << ", but this seems to be less than the time step size of the solvers. "
150 << "So postpone next plot to t=" << ({{SUBNAMESPACE}}repositories::getMinTimeStamp() + {{SUBNAMESPACE}}TimeInBetweenPlots)
151 );
152 nextMinPlotTimeStamp = {{SUBNAMESPACE}}repositories::getMinTimeStamp() + {{SUBNAMESPACE}}TimeInBetweenPlots;
153 } else if (nextMaxPlotTimeStamp < {{SUBNAMESPACE}}repositories::getMaxTimeStamp()) {
154 logWarning(
155 "selectNextAlgorithmicStep()",
156 "Code is asked to plot every dt="
157 << {{SUBNAMESPACE}}TimeInBetweenPlots << ", but this seems to be less than the time step size of the solvers. "
158 << "So postpone next plot to t=" << ({{SUBNAMESPACE}}repositories::getMaxTimeStamp() + {{SUBNAMESPACE}}TimeInBetweenPlots)
159 );
160 nextMaxPlotTimeStamp = {{SUBNAMESPACE}}repositories::getMaxTimeStamp() + {{SUBNAMESPACE}}TimeInBetweenPlots;
161 }
162
163 nextMaxPlotTimeStamp = std::max(nextMaxPlotTimeStamp, nextMinPlotTimeStamp);
164
165 peano4::parallel::Node::getInstance().setNextProgramStep(
166 {{SUBNAMESPACE}}repositories::StepRepository::toProgramStep({{SUBNAMESPACE}}repositories::StepRepository::Steps::PlotSolution)
167 );
168 haveJustWrittenSnapshot = true;
169 } else if (
170 {{SUBNAMESPACE}}TimeInBetweenCheckpoints > 0.0
171 and
172 {{SUBNAMESPACE}}repositories::getMinTimeStamp() < {{SUBNAMESPACE}}MinTerminalTime
173 and
174 {{SUBNAMESPACE}}repositories::getMaxTimeStamp() < {{SUBNAMESPACE}}MaxTerminalTime
175 and
176 ({{SUBNAMESPACE}}repositories::getMinTimeStamp() >= nextMinCheckpointTimeStamp or {{SUBNAMESPACE}}repositories::getMaxTimeStamp() >= nextMaxCheckpointTimeStamp)
177 and
178 {{SUBNAMESPACE}}repositories::mayPlot()
179 ) {
180 if ({{SUBNAMESPACE}}repositories::getMinTimeStamp() >= nextMinCheckpointTimeStamp) {
181 nextMinCheckpointTimeStamp += {{SUBNAMESPACE}}TimeInBetweenCheckpoints;
182 }
183 if ({{SUBNAMESPACE}}repositories::getMaxTimeStamp() >= nextMaxCheckpointTimeStamp) {
184 nextMaxCheckpointTimeStamp += {{SUBNAMESPACE}}TimeInBetweenCheckpoints;
185 }
186
187 if (nextMinCheckpointTimeStamp < {{SUBNAMESPACE}}repositories::getMinTimeStamp()) {
188 logWarning(
189 "selectNextAlgorithmicStep()",
190 "Code is asked to checkpoint every dt="
191 << {{SUBNAMESPACE}}TimeInBetweenCheckpoints << ", but this seems to be less than the time step size of the solvers. "
192 << "So postpone next checkpoint to t=" << ({{SUBNAMESPACE}}repositories::getMinTimeStamp() + {{SUBNAMESPACE}}TimeInBetweenCheckpoints)
193 );
194 nextMinCheckpointTimeStamp = {{SUBNAMESPACE}}repositories::getMinTimeStamp() + {{SUBNAMESPACE}}TimeInBetweenCheckpoints;
195 } else if (nextMaxCheckpointTimeStamp < {{SUBNAMESPACE}}repositories::getMaxTimeStamp()) {
196 logWarning(
197 "selectNextAlgorithmicStep()",
198 "Code is asked to checkpoint every dt="
199 << {{SUBNAMESPACE}}TimeInBetweenCheckpoints << ", but this seems to be less than the time step size of the solvers. "
200 << "So postpone next checkpoint to t=" << ({{SUBNAMESPACE}}repositories::getMaxTimeStamp() + {{SUBNAMESPACE}}TimeInBetweenCheckpoints)
201 );
202 nextMaxCheckpointTimeStamp = {{SUBNAMESPACE}}repositories::getMaxTimeStamp() + {{SUBNAMESPACE}}TimeInBetweenCheckpoints;
203 }
204
205 nextMaxCheckpointTimeStamp = std::max(nextMaxCheckpointTimeStamp, nextMinCheckpointTimeStamp);
206
207 peano4::parallel::Node::getInstance().setNextProgramStep(
208 {{SUBNAMESPACE}}repositories::StepRepository::toProgramStep({{SUBNAMESPACE}}repositories::StepRepository::Steps::CheckpointSolution)
209 );
210 haveJustWrittenSnapshot = true;
211 } else if ({{SUBNAMESPACE}}repositories::getMinTimeStamp() < {{SUBNAMESPACE}}MinTerminalTime and {{SUBNAMESPACE}}repositories::getMaxTimeStamp() < {{SUBNAMESPACE}}MaxTerminalTime) {
212 peano4::parallel::Node::getInstance().setNextProgramStep(
213 {{SUBNAMESPACE}}repositories::StepRepository::toProgramStep({{SUBNAMESPACE}}repositories::StepRepository::Steps::TimeStep)
214 );
215 continueToSolve = true;
216 haveJustWrittenSnapshot = false;
217 } else {
218 if (not haveJustWrittenSnapshot and {{SUBNAMESPACE}}TimeInBetweenPlots > 0.0 and {{SUBNAMESPACE}}repositories::mayPlot()) {
219 peano4::parallel::Node::getInstance().setNextProgramStep(
220 {{SUBNAMESPACE}}repositories::StepRepository::toProgramStep({{SUBNAMESPACE}}repositories::StepRepository::Steps::PlotSolution)
221 );
222 haveJustWrittenSnapshot = true;
223 nextMinPlotTimeStamp = std::numeric_limits<double>::max();
224 nextMaxPlotTimeStamp = std::numeric_limits<double>::max();
225 } else if (not haveJustWrittenSnapshot and {{SUBNAMESPACE}}TimeInBetweenPlots > 0.0 and not {{SUBNAMESPACE}}repositories::mayPlot()) {
226 continueToSolve = true; // don't want to terminate immediately but to wait for incomplete time steps to complete
227 } else {
228 continueToSolve = false;
229 }
230 }
231 }
232
233 return continueToSolve;
234}
