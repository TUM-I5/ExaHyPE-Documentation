19bool tarch::accelerator::Device::init() {
20 logDebug("init()", "-----------------------------------------");
21 CHECK_CUDA_ERROR(cudaGetDeviceCount(&numberOfDevices));
22 for (int device = 0; device < numberOfDevices; device++) {
23 CHECK_CUDA_ERROR(cudaSetDevice(device));
24 CHECK_CUDA_ERROR(cudaFree(0));
25 cudaDeviceProp* prop = &cuda::DevicePropertiesCache::get();
26 std::size_t free = 0, total = 0;
27 CHECK_CUDA_ERROR(cudaMemGetInfo(&free, &total));
28
29 constexpr int MB = 1024 * 1024;
30 logInfo("init()", "[" << device << "]: " << prop->name);
31 logDebug("init()", " " << prop->major << "." << prop->minor << " CC");
32 logDebug("init()", " " << free / MB << " MB free of " << total / MB << " MB total global memory");
33 logDebug("init()", " " << prop->totalConstMem << " bytes constant memory");
34 logDebug("init()", " " << prop->multiProcessorCount << " SMs");
35 logDebug("init()", " " << prop->maxBlocksPerMultiProcessor << " blocks per SM");
36 logDebug("init()", " " << prop->maxThreadsPerMultiProcessor << " threads per SM");
37 logDebug("init()", " " << prop->regsPerMultiprocessor << " registers per SM");
38 logDebug("init()", " " << prop->sharedMemPerMultiprocessor << " bytes shared memory per SM");
39 logDebug("init()", " " << prop->maxThreadsPerBlock << " threads per block");
40 logDebug("init()", " " << prop->regsPerBlock << " registers per block");
41 logDebug("init()", " " << prop->sharedMemPerBlock << " bytes shared memory per block");
42 logDebug("init()", " " << prop->warpSize << " threads per warp");
43 logDebug("init()", " " << prop->asyncEngineCount << " asynchronous engines");
44 }
45 logDebug("init()", "-----------------------------------------");
46
47 return isInitialised();
48}
