13peano4::datamanagement::CellMarker::CellMarker(const peano4::grid::GridTraversalEvent& event):
14 _centre(event.getX()),
15 _h(event.getH()),
16 _hasBeenRefined(event.getHasBeenRefined() != 0),
17 _willBeRefined(event.getWillBeRefined() != 0),

20 _isOneVertexHanging(false),

23 _invokingSpacetreeIsNotInvolvedInAnyDynamicLoadBalancing(

28 _isOneVertexHanging |= event.getVertexDataFrom(i)

30
31 _isOneVertexCreatedOrDestroyed |= event.getVertexDataTo(i)
32 == peano4::grid::TraversalObserver::CreateOrDestroyPersistentGridEntity;
33 _isOneVertexCreatedOrDestroyed |= event.getVertexDataFrom(i)
34 == peano4::grid::TraversalObserver::CreateOrDestroyPersistentGridEntity;
35 }
36 _relativePositionOfCellWithinFatherCell = event.getRelativePositionToFather();
37
38 bool couldBeEnclave = _invokingSpacetreeIsNotInvolvedInAnyDynamicLoadBalancing and _areAllVerticesInsideDomain
39 and not _isOneVertexHanging;
40
41 _willBeEnclave = couldBeEnclave and not _willBeRefined and not _isOneVertexCreatedOrDestroyed;
42 _hasBeenEnclave = couldBeEnclave and not _hasBeenRefined and not _isOneVertexCreatedOrDestroyed;
43}

52bool peano4::datamanagement::CellMarker::isContained(
53 const tarch::la::Vector<Dimensions, double>& x,
54 const tarch::la::Vector<Dimensions, double>& cellCentre,
55 const tarch::la::Vector<Dimensions, double>& h,
56 double tolerance
57) {
58 bool result = true;
59 for (int d = 0; d < Dimensions; d++) {
60 result &= tarch::la::greaterEquals(x(d), cellCentre(d) - h(d) / 2.0, tolerance);
61 result &= tarch::la::smallerEquals(x(d), cellCentre(d) + h(d) / 2.0, tolerance);
62 }
63 return result;

80tarch::la::Vector<Dimensions, double> peano4::datamanagement::CellMarker::getInvokingParentCellsCentre() const {
81 tarch::la::Vector<Dimensions, double> result = _centre;
82 for (int d = 0; d < Dimensions; d++) {
83 assertion1(_relativePositionOfCellWithinFatherCell(d) >= 0, toString());
84 assertion1(_relativePositionOfCellWithinFatherCell(d) <= 2, toString());
85 if (_relativePositionOfCellWithinFatherCell(d) == 0) {
86 result(d) += _h(d);
87 }
88 if (_relativePositionOfCellWithinFatherCell(d) == 2) {
89 result(d) -= _h(d);
90 }
91 }
92 return result;
93}

164void peano4::datamanagement::CellMarker::initDatatype() {
165 peano4::grid::GridTraversalEvent dummyEvent;
166
167 peano4::datamanagement::CellMarker instances[] = {
168 peano4::datamanagement::CellMarker(dummyEvent),
169 peano4::datamanagement::CellMarker(dummyEvent)
170 };
171
172 MPI_Datatype subtypes[] = {
173 MPI_DOUBLE,
174 MPI_DOUBLE,
175 MPI_BYTE,
176 MPI_BYTE,
177 MPI_BYTE,
178 MPI_BYTE,
179 MPI_BYTE,
180 MPI_BYTE,
181 MPI_BYTE,
182 MPI_INT
183 };
184
185 int blocklen[] = {Dimensions, Dimensions, 1, 1, 1, 1, 1, 1, Dimensions};
186
187 const int NumberOfAttributes = 9;
188
189 MPI_Aint baseFirstInstance;
190 MPI_Aint baseSecondInstance;
191 MPI_Get_address(&instances[0], &baseFirstInstance);
192 MPI_Get_address(&instances[1], &baseSecondInstance);
193 MPI_Aint disp[NumberOfAttributes];
194 int currentAddress = 0;
195 MPI_Get_address(&(instances[0]._centre), &disp[currentAddress]);
196 currentAddress++;
197 MPI_Get_address(&(instances[0]._h), &disp[currentAddress]);
198 currentAddress++;
199 MPI_Get_address(&(instances[0]._hasBeenRefined), &disp[currentAddress]);
200 currentAddress++;
201 MPI_Get_address(&(instances[0]._willBeRefined), &disp[currentAddress]);
202 currentAddress++;
203 MPI_Get_address(&(instances[0]._isLocal), &disp[currentAddress]);
204 currentAddress++;
205 MPI_Get_address(&(instances[0]._areAllVerticesRefined), &disp[currentAddress]);
206 currentAddress++;
207 MPI_Get_address(&(instances[0]._isOneVertexHanging), &disp[currentAddress]);
208 currentAddress++;
209 MPI_Get_address(&(instances[0]._areAllVerticesInsideDomain), &disp[currentAddress]);
210 currentAddress++;
211 MPI_Get_address(&(instances[0]._invokingSpacetreeIsNotInvolvedInAnyDynamicLoadBalancing), &disp[currentAddress]);
212 currentAddress++;
213 MPI_Get_address(&(instances[0]._relativePositionOfCellWithinFatherCell), &disp[currentAddress]);
214 currentAddress++;
215
216 MPI_Aint offset = disp[0] - baseFirstInstance;
217 MPI_Aint extent = baseSecondInstance - baseFirstInstance - offset;
218 for (int i = NumberOfAttributes - 1; i >= 0; i--) {
219 disp[i] = disp[i] - disp[0];
220 }
221
222 int errorCode = 0;
223 MPI_Datatype tmpType;
224 errorCode += MPI_Type_create_struct(NumberOfAttributes, blocklen, disp, subtypes, &tmpType);
225 errorCode += MPI_Type_create_resized(tmpType, offset, extent, &Datatype);
226 errorCode += MPI_Type_commit(&Datatype);
227 errorCode += MPI_Type_free(&tmpType);
228 if (errorCode)
229 std::cerr << "error constructing MPI datatype in " << __FILE__ << ":" << __LINE__ << std::endl;
230}
