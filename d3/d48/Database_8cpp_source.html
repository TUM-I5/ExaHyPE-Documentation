265std::string toolbox::particles::assignmentchecks::internal::Database::
266 particleHistory(const ParticleIdentifier& identifier) {
267
268 tarch::multicore::MultiReadSingleWriteLock
269 lock(_semaphore, tarch::multicore::MultiReadSingleWriteLock::Read);
270
271 std::ostringstream msg;
272 msg << "\n============================\n"
273 << identifier.toString() << "\n============================\n";
274
275 auto search = _data.find(identifier);
276 if (search != _data.end()) {
277
278#if PeanoDebug > 3
279 assertion3(
280 search->first.particleID == identifier.particleID,
281 "Particle identifier ID not identical",
282 search->first.particleID,
283 identifier.particleID
284 );
285 assertion3(
286 search->first.treeID == identifier.treeID,
287 "Particle treeID not identical",
288 search->first.treeID,
289 identifier.treeID
290 );
291 for (int i = 0; i < Dimensions; i++) {
292 assertion3(
293 search->first.vertexInd(i) == identifier.vertexInd(i),
294 "Particle treeID not identical",
295 search->first.vertexInd(i),
296 identifier.vertexInd(i)
297 );
298 }
299 assertion3(
300 search->first.vertexDepth == identifier.vertexDepth,
301 "Particle depth not identical",
302 search->first.vertexDepth,
303 identifier.vertexDepth
304 );
305#endif
306
307 ParticleEvents& history = search->second;
308 auto ev = history.crbegin();
309
310 // mesh sweep index of the previous event.
311 int prevMeshSweepInd = -1;
312
313 while (ev != history.crend()) {
314
315 int meshSweepInd = ev->meshSweepIndex;
316
317 if (meshSweepInd != prevMeshSweepInd) {
318 // We're starting new sweep. Print header.
319 msg << "\nsweep #" << meshSweepInd << " ("
320 << _meshSweepData.at(meshSweepInd).getName() << "):";
321 }
322 msg << "\n\t->" << ev->toString();
323
324 prevMeshSweepInd = meshSweepInd;
325 ev++;
326 }
327 }
328
329 msg << "\n";
330
331 return msg.str();
332}

335toolbox::particles::assignmentchecks::internal::Event toolbox::particles::
336 assignmentchecks::internal::Database::
337 addEvent(ParticleIdentifier identifier, Event& event) {
338
339 logDebug("addEvent()", "Adding event for " << identifier.toString());
340
341 tarch::multicore::MultiReadSingleWriteLock
342 lock(_semaphore, tarch::multicore::MultiReadSingleWriteLock::Write);
343
344 // Take note of the current mesh sweep index.
345 event.meshSweepIndex = getCurrentMeshSweepIndex();
346
347 // Try finding existing previous event.
348 auto search = _data.find(identifier);
349
350 if (search == _data.end()) {
351 // This is a new particle.
352
353 ParticleEvents newEventHistory = ParticleEvents();
354 newEventHistory.push_back(event);
355 _data.insert(
356 std::pair<ParticleIdentifier, ParticleEvents>(identifier, newEventHistory)
357 );
358 logDebug(
359 "addEvent(...) ",
360 "add new particle history thread for " + identifier.toString()
361 + ",Event:" + event.toString()
362 );
363 // there is no previous event.
364 return Event(Event::Type::NotFound);
365
366 } else {
367
368#if PeanoDebug > 3
369 assertion3(
370 search->first.particleID == identifier.particleID,
371 "Particle identifier ID not identical",
372 search->first.particleID,
373 identifier.particleID
374 );
375 assertion3(
376 search->first.treeID == identifier.treeID,
377 "Particle treeID not identical",
378 search->first.treeID,
379 identifier.treeID
380 );
381 for (int i = 0; i < Dimensions; i++) {
382 assertion3(
383 search->first.vertexInd(i) == identifier.vertexInd(i),
384 "Particle treeID not identical",
385 search->first.vertexInd(i),
386 identifier.vertexInd(i)
387 );
388 }
389 assertion3(
390 search->first.vertexDepth == identifier.vertexDepth,
391 "Particle depth not identical",
392 search->first.vertexDepth,
393 identifier.vertexDepth
394 );
395#endif
396
397
398 ParticleEvents& history = search->second;
399
400 // First: Do we need to downsize the history?
401 if (history.size() >= _maxParticleEventsToKeepTrackOf) {
402
403 // make copy, not reference
404 Event lastEvent = history.back();
405
406#if PeanoDebug > 3
407 if (lastEvent.type == Event::Type::NotFound) {
408 logDebug(
409 "addEvent(...)",
410 "Particle histroy shouldn't contain an invalid event"
411 );
412 }
413#endif
414
415 // In certain cases, especially for virtual particles, the last saved
416 // event may be a substitution already. We don't want to keep
417 // prepending the string with the notifier, so check for it.
418 //if (not lastEvent.trace.starts_with("substitute-for-whole-trajectory/")) {
419 if (lastEvent.trace.compare(0, std::string("substitute-for-whole-trajectory/").length(), "substitute-for-whole-trajectory/") != 0) {
420 lastEvent.trace = "substitute-for-whole-trajectory/" + lastEvent.trace;
421 }
422
423 logDebug(
424 "addEvent(...) ",
425 "shortening history of " << identifier.toString(
426 ) << " to " << lastEvent.toString()
427 );
428
429 history.clear();
430 history.push_back(lastEvent);
431 }
432
433 // Add the new event now.
434 Event previousEvent = history.back();
435 history.push_back(event);
436 logDebug(
437 "addEvent(...) ",
438 "added event " + event.toString() + " " + identifier.toString()
439 );
440
441 return previousEvent;
442 }
443}

485 }
486 assertion3(

488 "Particle depth not identical",

490 identifier.vertexDepth
491 );
492#endif
493
494 ParticleEvents& history = search->second;
495
496 // Get previous event
497 Event previousEventOnThisTree = Event(Event::Type::NotFound);

519void toolbox::particles::assignmentchecks::internal::Database::
520 shiftIdentifierVertexInd(
521 ParticleIdentifier& identifier,
522 tarch::la::Vector<Dimensions, long long>& newVertexInd,
523 const int newVertexDepth

528
529 auto search = _data.find(identifier);
530
531 assertion1(
532 search != _data.end(),
533 "Particle not found through its identifier"
534 );
535
536#if PeanoDebug > 3
537 assertion3(
538 search->first.particleID == identifier.particleID,
539 "Particle identifier ID not identical",
540 search->first.particleID,
541 identifier.particleID
542 );
543 assertion3(
544 search->first.treeID == identifier.treeID,
545 "Particle treeID not identical",
546 search->first.treeID,
547 identifier.treeID
548 );
549 for (int i = 0; i < Dimensions; i++) {
550 assertion3(
551 search->first.vertexInd(i) == identifier.vertexInd(i),
552 "Particle treeID not identical",
553 search->first.vertexInd(i),
554 identifier.vertexInd(i)
555 );
556 }
557 assertion3(
558 search->first.vertexDepth == identifier.vertexDepth,
559 "Particle depth not identical",
560 search->first.vertexDepth,
561 identifier.vertexDepth
562 );
563#endif
564
565 // Do we need to shift the coordinates of the identifier? Use the one the
566 // search gives you back, because we allow for a fuzzy search. You need to
567 // compare to the actually stored coordinates, not to whatever you think they
568 // currently may be.
569 ParticleIdentifier key = search->first;
570 ParticleEvents& history = search->second;
571
572 // delete old entry in database and add the new one.
573 logDebug(
574 "shiftIdentifierCoordinates(...) ",
575 "shifting particle identifier from "
576 << key.vertexInd << " to " << newVertexInd;
577 );
578
579 ParticleEvents historyCopy = history;
580 _data.erase(key);
581 ParticleIdentifier newIdentifier = ParticleIdentifier(
582 identifier.particleName,
583 identifier.particleX,
584 newVertexInd,
585 newVertexDepth,
586 identifier.particleID,
587 identifier.treeID
588 );
589 // Important: if particle already exists with this identifier (e.g. because
590 // if was on this vertex before), map.insert() will not replace the existing
591 // entry. Use insert_or_assign instead, which replaces it.
592 // TODO: Instead of replacing, append history.
593 _data.insert_or_assign(newIdentifier, historyCopy);
594}

618toolbox::particles::assignmentchecks::internal::Event toolbox::particles::
619 assignmentchecks::internal::Database::getPreviousEvent(
620 ParticleEvents& particleHistory,
621 int spacetreeId,
622 size_t nFirstEventsToSkip
623 ) {
624
625 tarch::multicore::MultiReadSingleWriteLock
626 lock(_semaphore, tarch::multicore::MultiReadSingleWriteLock::Read);
627
628 if (particleHistory.size() <= nFirstEventsToSkip) {
629 return Event(Event::Type::NotFound);
630 }
631
632 auto it = particleHistory.crbegin();
633 std::advance(it, nFirstEventsToSkip);
634
635 if (it == particleHistory.crend()) {
636 return Event(Event::Type::NotFound);
637 }
638
639 if (spacetreeId == Database::AnyTree)
640 return *it;
641
642 while (it != particleHistory.crend()) {
643 if (it->treeId == spacetreeId)
644 return *it;
645 it++;
646 }
647
648 return Event(Event::Type::NotFound);
