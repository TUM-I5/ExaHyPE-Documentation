5"""
6We import the required modules for this project.
7We always need the 'peano4' module as this is our core project.

11import peano4
12import exahype2
13
14"""
15We specify the space dimensions here.
16We support either 2- or 3-dimensional problems.
17"""
18dimensions = 2
19
20"""
21The number of finite volumes per axis in one patch.
22"""
23patch_size = 16
24
25"""
26The number of levels the mesh is refined.
27"""
28depth = 4
29
30"""
31The simulation end time.
32"""
33end_time = 1.0
34
35"""
36Choose domain size and offset.
37"""
38size = [10.0, 10.0, 10.0]
39offset = [0.0, 0.0, 0.0]
40
41"""
42Choose how often a snapshot is written.
43"""
44time_in_between_two_snapshots = end_time / 20
45# time_in_between_two_snapshots = 0 # Set to 0 to disable I/O
46
47"""
48Switch between 'Release', 'Debug', 'Asserts', 'Trace', 'Stats'.
49"""
50compile_mode = "Release"
51
52"""
53We first create a new ExaHyPE 2 project.
54For this, we specify the (nested) namespaces, the name of our main file and our executable name.
55"""
56my_project = exahype2.Project(
57 namespace=["tutorials", "exahype2", "symhype", "acoustic"],
58 project_name="Acoustic" + str(dimensions) + "D",
59 directory=".",
60 executable="Acoustic" + str(dimensions) + "D." + str(compile_mode),
61)
62
63"""
64Add the Finite Volumes solver using named arguments.
65This is the way you can add further PDE terms.
66This requires the 'BlockStructured' toolbox and 'ExaHyPE' to be built.
67"""
68my_solver = exahype2.solvers.fv.godunov.GlobalAdaptiveTimeStep(
69 name="FVSolver",
70 patch_size=patch_size,
71 unknowns=dimensions + 1, # [p, u, v, (w)]
72 auxiliary_variables=0,
73 min_volume_h=(1.1 * min(size[0:dimensions]) / (3.0**depth)),
74 max_volume_h=(1.1 * min(size[0:dimensions]) / (3.0**depth)),
75 time_step_relaxation=0.5,
76)
77
78"""
79We want to define our PDE symbolically.
80For this we use the 'symhype' package (not to be confused with 'sympy') from 'ExaHyPE'.
81"""
82my_pde = exahype2.symhype.FirstOrderConservativePDEFormulation(
83 unknowns=dimensions + 1, auxiliary_variables=0, dimensions=dimensions
84)
85p = my_pde.name_Q_entry(0, "p")
86v = my_pde.name_Q_entries(1, dimensions, "v")
87
88rho = sympy.symbols("rho")
89c = sympy.symbols("c")
90K0 = rho * c * c
91
92"""
93Define the equation system
94"""
95# Flux [unknowns, dimensions]
96if dimensions == 2:
97 my_pde.F[0, 0] = K0 * v[0]
98 my_pde.F[1, 0] = 1.0 / rho * p
99 my_pde.F[2, 0] = 0
100
101 my_pde.F[0, 1] = K0 * v[1]
102 my_pde.F[1, 1] = 0
103 my_pde.F[2, 1] = 1.0 / rho * p
104else:
105 my_pde.F[0, 0] = K0 * v[0]
106 my_pde.F[1, 0] = 1.0 / rho * p
107 my_pde.F[2, 0] = 0
108 my_pde.F[3, 0] = 0
109
110 my_pde.F[0, 1] = K0 * v[1]
111 my_pde.F[1, 1] = 0
112 my_pde.F[2, 1] = 1.0 / rho * p
113 my_pde.F[3, 1] = 0
114
115 my_pde.F[0, 2] = K0 * v[2]
116 my_pde.F[1, 2] = 0
117 my_pde.F[2, 2] = 0
118 my_pde.F[3, 2] = 1.0 / rho * p
119
120# Eigenvalues [unknowns, dimensions]
121if dimensions == 2:
122 my_pde.eigenvalues[0] = -c
123 my_pde.eigenvalues[1] = 0
124 my_pde.eigenvalues[2] = c
125else:
126 my_pde.eigenvalues[0] = -c
127 my_pde.eigenvalues[1] = 0
128 my_pde.eigenvalues[2] = 0
129 my_pde.eigenvalues[3] = c
130
131"""
132Since 'my_pde' only holds the PDE without initial- or boundary conditions,
133we still need to properly define initial- and boundary conditions.
134This gives us then a complete description of a 'scenario'.
135"""
136
137# Initial conditions
138my_pde.initial_values[0] = 0
139my_pde.initial_values[1] = 0
140my_pde.initial_values[2] = 0
141if dimensions == 3:
142 my_pde.initial_values[3] = 0
143
144# Boundary conditions
145my_pde.boundary_values[0] = 0
146my_pde.boundary_values[1] = 0
147my_pde.boundary_values[2] = 0
148if dimensions == 3:
149 my_pde.boundary_values[3] = 0
150
151# Source term
152sigma = sympy.symbols("sigma")
153t0 = sympy.symbols("t0")
154M0 = sympy.symbols("M0")
155t = sympy.symbols("t")
156
157max_h = sympy.symbols("MaxAdmissibleVolumeH")
158force = M0 * sympy.exp(-((t - t0) * (t - t0)) / (2.0 * sigma * sigma))
159
160if dimensions == 2:
161 point_source = sympy.sqrt((5 - my_pde.x[0]) ** 2 + (5 - my_pde.x[1]) ** 2)
162 my_pde.sources[0] = sympy.Piecewise((force, point_source <= max_h), (0, True)) # p
163else:
164 point_source = sympy.sqrt((5 - my_pde.x[0]) ** 2 + (5 - my_pde.x[1]) ** 2 + (5 - my_pde.x[2]) ** 2)
165 my_pde.sources[0] = sympy.Piecewise((force, point_source <= max_h), (0, True)) # p
166
167my_pde.sources[1] = 0
168my_pde.sources[2] = 0
169if dimensions == 3:
170 my_pde.sources[3] = 0
171
172my_pde.substitute_expression(rho, 2.7)
173my_pde.substitute_expression(c, 6.0)
174my_pde.substitute_expression(sigma, 0.1149)
175my_pde.substitute_expression(t0, 0.7)
176my_pde.substitute_expression(M0, 1000.0)
177
178"""
179Specify which implementation our solvers uses.
180Here we want to set the implementation we get from our symbolically defined PDE,
181i.e., we get the C++ implementation which is generated by ExaHyPE's 'symhype' package.
182"""
183my_solver.set_implementation(
184 initial_conditions=my_pde.implementation_of_initial_conditions(),
185 boundary_conditions=my_pde.implementation_of_boundary_conditions(),
186 flux=my_pde.implementation_of_flux(),
187 max_eigenvalue=my_pde.implementation_of_max_eigenvalue(),
188 source_term=my_pde.implementation_of_sources(),
189)
190
191"""
192To see which variables (unknowns + auxiliary variables) we can visualise,
193let's add a plot description for the variables to our solver.
194"""
195my_solver.plot_description = my_pde.unknown_identifier_for_plotter()
196
197"""
198Add the solver to our project
199"""
200my_project.add_solver(my_solver)
201
202"""
203Configure some global parameters
204"""
205my_project.set_global_simulation_parameters(
206 dimensions=dimensions,
207 size=size[0:dimensions],
208 offset=offset[0:dimensions],
209 min_end_time=end_time,
210 max_end_time=end_time,
211 first_plot_time_stamp=0.0,
212 time_in_between_plots=time_in_between_two_snapshots,
213 periodic_BC=[False, False, False],
214)

220my_project.set_output_path("solutions")
