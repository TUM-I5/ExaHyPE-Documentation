45 def _getValueFromLine( self, line, key, verbose ):
46 result = 0.0
47 try:
48 rhs = line.split( key + "=" )[1]
49 if "," in rhs:
50 rhs = rhs.split(",")[0]
51 result = float( rhs )

71 def entry_holds_interesting_data(self,
72 key,
73 verbose
74 ):
75 """!
76
77 Return true if entry for key is not empty and not all values are exactly the same
78
79 """
80 if verbose:
81 print( "search for entry for key {}".format(key) )
82 if not key in self.mesh_association_data.keys():
83 if verbose:
84 print( "no key {}".format(key) )
85 return False
86 if len(self.mesh_association_data[key])==0:
87 if verbose:
88 print( "no entry for key {}".format(key) )
89 return False
90 first_mesh_association_data_entry = self.mesh_association_data[key][0]
91 for values in self.mesh_association_data[key]:
92 if first_mesh_association_data_entry!=values:
93 return True
