143 int normal,
144 bool extrapolateLinearly,
145 bool interpolateLinearlyBetweenRealAndRestrictedVolumes
146 ) {
147 tarch::la::DynamicMatrix P1d(
148 3,
149 3,
150 {{1.0 / 3.0, 2.0 / 3.0, 0.0},
151 {0.0, 3.0 / 3.0, 0.0},
152 {0.0, 2.0 / 3.0, 1.0 / 3.0}}
153 );
154 P1d.replicateRows(3, numberOfDoFsPerAxisInPatch, 1, true);
155 P1d.removeColumn(0);
156 P1d.removeColumn(numberOfDoFsPerAxisInPatch);
157
158 if (extrapolateLinearly) {
159 P1d(0, 0) = 2.0 / 3.0 + 1.0 / 3.0 * 2.0;
160 P1d(0, 1) = -1.0 / 3.0;
161 P1d(
162 numberOfDoFsPerAxisInPatch * 3 - 1,
163 numberOfDoFsPerAxisInPatch - 1
164 ) = 2.0 / 3.0 + 1.0 / 3.0 * 2.0;
165 P1d(
166 numberOfDoFsPerAxisInPatch * 3 - 1,
167 numberOfDoFsPerAxisInPatch - 2
168 ) = -1.0 / 3.0;
169 } else {
170 P1d(0, 0) = 1.0;
171 P1d(0, 1) = 0.0;
172 P1d(
173 numberOfDoFsPerAxisInPatch * 3 - 1,
174 numberOfDoFsPerAxisInPatch - 1
175 ) = 1.0;
176 P1d(
177 numberOfDoFsPerAxisInPatch * 3 - 1,
178 numberOfDoFsPerAxisInPatch - 2
179 ) = 0.0;
180 }
181
182 logDebug(
183 "createLinearInterpolationMatrix(...)",
184 "1d matrix: " << P1d.toString()
185 );
186
187 if (interpolateLinearlyBetweenRealAndRestrictedVolumes) {
188 return createInterpolationMatrixFrom1dTemplateByLinearInterpolationAlongNormal(
189 P1d,
190 numberOfDoFsPerAxisInPatch,
191 normal
192 );
193 } else {
194 return createInterpolationMatrixFrom1dTemplateByInsertingZeroColsAndRows(
195 P1d,
196 numberOfDoFsPerAxisInPatch,
197 normal
198 );
