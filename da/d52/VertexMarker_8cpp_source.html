89tarch::la::Vector<Dimensions, double> peano4::datamanagement::VertexMarker::x(int i) const {
90 tarch::la::Vector<Dimensions, double> result(_cellCentre);
91 std::bitset<Dimensions> myset(i);
92 for (int d = 0; d < Dimensions; d++) {

173tarch::la::Vector<Dimensions, double> peano4::datamanagement::VertexMarker::getInvokingParentCellsCentre() const {
174 tarch::la::Vector<Dimensions, double> result = _cellCentre;
175 for (int d = 0; d < Dimensions; d++) {
176 assertion1(_relativePositionOfCellWithinFatherCell(d) >= 0, toString());
177 assertion1(_relativePositionOfCellWithinFatherCell(d) <= 2, toString());
178 if (_relativePositionOfCellWithinFatherCell(d) == 0) {
179 result(d) += _h(d);
180 }
181 if (_relativePositionOfCellWithinFatherCell(d) == 2) {
182 result(d) -= _h(d);
183 }
184 }
185 return result;
186}
